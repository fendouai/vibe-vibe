## 事件驱动架构

事件驱动架构是现代交易系统的首选：

```python
from abc import ABC, abstractmethod
from typing import Dict, Any
from dataclasses import dataclass
from datetime import datetime
from enum import Enum

class EventType(Enum):
    """事件类型枚举"""
    MARKET = "market"           # 市场数据事件
    SIGNAL = "signal"          # 交易信号事件
    ORDER = "order"            # 订单事件
    FILL = "fill"              # 成交事件
    POSITION = "position"      # 持仓更新事件
    RISK = "risk"              # 风险事件

@dataclass
class Event:
    """事件基类"""
    type: EventType
    timestamp: datetime
    data: Dict[str, Any]

    def __post_init__(self):
        if isinstance(self.type, str):
            self.type = EventType(self.type)

class EventBus:
    """事件总线 - 组件间通信中枢"""
    def __init__(self):
        self.subscribers: Dict[EventType, list] = {
            event_type: [] for event_type in EventType
        }

    def subscribe(self, event_type: EventType, handler: callable):
        """订阅事件"""
        self.subscribers[event_type].append(handler)

    def unsubscribe(self, event_type: EventType, handler: callable):
        """取消订阅"""
        if handler in self.subscribers[event_type]:
            self.subscribers[event_type].remove(handler)

    def publish(self, event: Event):
        """发布事件"""
        for handler in self.subscribers[event.type]:
            try:
                handler(event)
            except Exception as e:
                print(f"事件处理错误 {event.type}: {e}")

# 全局事件总线实例
event_bus = EventBus()
```

## 核心组件设计

### 1. 数据源组件

```python
class MarketDataSource(ABC):
    """市场数据源抽象基类"""

    @abstractmethod
    def connect(self):
        """连接数据源"""
        pass

    @abstractmethod
    def disconnect(self):
        """断开连接"""
        pass

    @abstractmethod
    def subscribe(self, symbol: str):
        """订阅标的"""
        pass

    @abstractmethod
    def unsubscribe(self, symbol: str):
        """取消订阅"""
        pass

class DataFeed(MarketDataSource):
    """数据feed组件"""

    def __init__(self, event_bus: EventBus):
        self.event_bus = event_bus
        self.connected = False
        self.subscribed_symbols = set()

    def connect(self):
        """连接到数据源"""
        print("📡 连接市场数据源...")
        self.connected = True
        print("✅ 数据源已连接")

    def disconnect(self):
        """断开连接"""
        print("📡 断开市场数据源...")
        self.connected = False
        print("✅ 数据源已断开")

    def subscribe(self, symbol: str):
        """订阅标的"""
        if symbol not in self.subscribed_symbols:
            self.subscribed_symbols.add(symbol)
            print(f"📊 订阅: {symbol}")

    def unsubscribe(self, symbol: str):
        """取消订阅"""
        if symbol in self.subscribed_symbols:
            self.subscribed_symbols.remove(symbol)
            print(f"📊 取消订阅: {symbol}")

    def on_market_data(self, symbol: str, data: dict):
        """处理市场数据"""
        event = Event(
            type=EventType.MARKET,
            timestamp=datetime.now(),
            data={'symbol': symbol, **data}
        )
        self.event_bus.publish(event)
```

### 2. 策略引擎组件

```python
class Strategy(ABC):
    """策略抽象基类"""

    def __init__(self, name: str, event_bus: EventBus):
        self.name = name
        self.event_bus = event_bus
        self.positions = {}  # 当前持仓
        self.signals = {}   # 交易信号

        # 订阅市场数据事件
        self.event_bus.subscribe(EventType.MARKET, self.on_market_data)

    @abstractmethod
    def on_market_data(self, event: Event):
        """处理市场数据 - 子类实现"""
        pass

    def generate_signal(self, symbol: str, direction: str, strength: float):
        """生成交易信号"""
        signal = {
            'strategy': self.name,
            'symbol': symbol,
            'direction': direction,  # 'long' or 'short'
            'strength': strength,     # 0-1
            'timestamp': datetime.now()
        }

        event = Event(
            type=EventType.SIGNAL,
            timestamp=datetime.now(),
            data=signal
        )
        self.event_bus.publish(event)
        return signal

    def get_position(self, symbol: str) -> float:
        """获取持仓"""
        return self.positions.get(symbol, 0)

class MomentumStrategy(Strategy):
    """动量策略"""

    def __init__(self, name: str, event_bus: EventBus, lookback: int = 20):
        super().__init__(name, event_bus)
        self.lookback = lookback
        self.price_history = {}  # 价格历史

    def on_market_data(self, event: Event):
        """计算动量信号"""
        symbol = event.data['symbol']
        price = event.data.get('close')

        if price is None:
            return

        # 更新价格历史
        if symbol not in self.price_history:
            self.price_history[symbol] = []
        self.price_history[symbol].append(price)

        # 保持足够的历史数据
        if len(self.price_history[symbol]) > self.lookback:
            self.price_history[symbol].pop(0)

        # 计算动量
        if len(self.price_history[symbol]) >= self.lookback:
            prices = self.price_history[symbol]
            momentum = (prices[-1] - prices[0]) / prices[0]

            # 生成信号
            if momentum > 0.02:  # 动量超过2%
                self.generate_signal(symbol, 'long', min(momentum * 10, 1.0))
            elif momentum < -0.02:
                self.generate_signal(symbol, 'short', min(abs(momentum) * 10, 1.0))
```

### 3. 风险管理组件

```python
class RiskManager:
    """风险管理器"""

    def __init__(self, event_bus: EventBus, max_position_size: float = 0.1,
                 max_daily_loss: float = 0.05, max_leverage: float = 2.0):
        self.event_bus = event_bus
        self.max_position_size = max_position_size  # 单个仓位最大比例
        self.max_daily_loss = max_daily_loss       # 日最大亏损比例
        self.max_leverage = max_leverage           # 最大杠杆

        self.daily_pnl = 0.0
        self.positions = {}

        # 订阅事件
        self.event_bus.subscribe(EventType.FILL, self.on_fill)

    def check_order(self, order: dict) -> tuple[bool, str]:
        """检查订单是否通过风控"""
        symbol = order['symbol']
        quantity = order['quantity']
        price = order.get('price', 0)

        # 检查仓位限制
        position_value = self.positions.get(symbol, 0) * price
        portfolio_value = sum(self.positions.values()) * price  # 简化计算

        if portfolio_value > 0:
            position_ratio = position_value / portfolio_value
            if position_ratio > self.max_position_size:
                return False, f"仓位超限: {position_ratio:.2%} > {self.max_position_size:.2%}"

        # 检查日亏损限制
        if self.daily_pnl < -self.max_daily_loss:
            return False, f"日亏损超限: {self.daily_pnl:.2%} > -{self.max_daily_loss:.2%}"

        # 检查杠杆
        total_exposure = sum(abs(v) for v in self.positions.values())
        if total_exposure > self.max_leverage:
            return False, f"杠杆超限: {total_exposure:.2f}x > {self.max_leverage}x"

        return True, "通过"

    def on_fill(self, event: Event):
        """处理成交事件"""
        fill = event.data
        symbol = fill['symbol']

        # 更新持仓
        if fill['side'] == 'buy':
            self.positions[symbol] = self.positions.get(symbol, 0) + fill['quantity']
        else:
            self.positions[symbol] = self.positions.get(symbol, 0) - fill['quantity']

        # 更新盈亏
        self.daily_pnl += fill.get('pnl', 0)
```

### 4. 订单路由组件

```python
class OrderRouter:
    """订单路由器"""

    def __init__(self, event_bus: EventBus, broker_adapter):
        self.event_bus = event_bus
        self.broker = broker_adapter
        self.pending_orders = {}  # 待成交订单
        self.order_id = 0

        # 订阅信号事件
        self.event_bus.subscribe(EventType.SIGNAL, self.on_signal)
        self.event_bus.subscribe(EventType.FILL, self.on_fill)

    def on_signal(self, event: Event):
        """处理交易信号"""
        signal = event.data
        self.route_order(signal)

    def route_order(self, signal: dict):
        """路由订单到broker"""
        self.order_id += 1

        order = {
            'id': self.order_id,
            'symbol': signal['symbol'],
            'side': 'buy' if signal['direction'] == 'long' else 'sell',
            'quantity': self.calculate_quantity(signal),
            'order_type': 'market',
            'signal': signal
        }

        self.pending_orders[self.order_id] = order

        # 发送订单到broker
        self.broker.place_order(order)

    def calculate_quantity(self, signal: dict) -> float:
        """计算机位数量"""
        # 简单实现：固定数量
        return 100

    def on_fill(self, event: Event):
        """处理成交事件"""
        fill = event.data
        order_id = fill.get('order_id')

        if order_id in self.pending_orders:
            del self.pending_orders[order_id]
```

## 系统主程序

```python
class TradingSystem:
    """交易系统主类"""

    def __init__(self, config: dict):
        # 初始化事件总线
        self.event_bus = EventBus()

        # 初始化各组件
        self.data_feed = DataFeed(self.event_bus)
        self.broker = None  # 根据配置初始化broker
        self.risk_manager = RiskManager(self.event_bus)
        self.order_router = OrderRouter(self.event_bus, self.broker)
        self.strategies = []

        self.running = False

    def add_strategy(self, strategy: Strategy):
        """添加策略"""
        self.strategies.append(strategy)

    def start(self):
        """启动系统"""
        print("="*50)
        print("🚀 启动交易系统...")
        print("="*50)

        self.running = True

        # 连接数据源
        self.data_feed.connect()

        # 订阅交易标的数据
        for symbol in ['AAPL', 'TSLA', 'BTCUSDT']:
            self.data_feed.subscribe(symbol)

        print("✅ 系统启动完成")
        print(f"   运行策略: {[s.name for s in self.strategies]}")
        print(f"   订阅标的: {list(self.data_feed.subscribed_symbols)}")

    def stop(self):
        """停止系统"""
        print("\n🛑 停止交易系统...")
        self.running = False

        # 取消订阅
        for symbol in list(self.data_feed.subscribed_symbols):
            self.data_feed.unsubscribe(symbol)

        # 断开连接
        self.data_feed.disconnect()

        print("✅ 系统已停止")

# 使用示例
if __name__ == "__main__":
    # 创建系统
    system = TradingSystem({'broker': 'alpaca'})

    # 添加策略
    momentum = MomentumStrategy('momentum_001', system.event_bus)
    system.add_strategy(momentum)

    # 启动
    system.start()

    # 运行一段时间
    import time
    time.sleep(60)

    # 停止
    system.stop()
```

## 架构设计原则

| 原则     | 说明               | 重要性 |
| -------- | ------------------ | ------ |
| 单一职责 | 每个组件只做一件事 | ⭐⭐⭐ |
| 事件解耦 | 组件通过事件通信   | ⭐⭐⭐ |
| 可配置   | 避免硬编码参数     | ⭐⭐   |
| 可扩展   | 容易添加新策略     | ⭐⭐   |
| 容错性   | 处理异常不崩溃     | ⭐⭐⭐ |

## 总结

交易系统架构的核心是**事件驱动**：

- 事件总线连接所有组件
- 策略产生信号，风险管理检查信号
- 订单路由将信号转化为真实订单
- 成交反馈更新持仓和风险状态
