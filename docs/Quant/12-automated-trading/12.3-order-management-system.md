# 12.3 订单管理系统

## 订单管理系统的作用

订单管理系统（Order Management System, OMS）是交易系统的核心组件，负责：

- 管理订单生命周期
- 维护订单队列
- 处理订单状态更新
- 追踪订单成交
- 处理部分成交
- 管理订单优先级

## 订单数据结构

```python
from dataclasses import dataclass, field
from datetime import datetime
from enum import Enum
from typing import Optional
import uuid

class OrderStatus(Enum):
    """订单状态"""
    PENDING = "pending"           # 待提交
    SUBMITTED = "submitted"       # 已提交
    PARTIAL_FILLED = "partial"   # 部分成交
    FILLED = "filled"            # 全部成交
    CANCELLED = "cancelled"      # 已取消
    REJECTED = "rejected"        # 被拒绝
    EXPIRED = "expired"          # 已过期

class OrderSide(Enum):
    """订单方向"""
    BUY = "buy"
    SELL = "sell"

class OrderType(Enum):
    """订单类型"""
    MARKET = "market"      # 市价单
    LIMIT = "limit"       # 限价单
    STOP = "stop"         # 止损单
    STOP_LIMIT = "stop_limit"  # 止损限价单

@dataclass
class Order:
    """订单数据类"""
    symbol: str
    side: OrderSide
    order_type: OrderType
    quantity: float
    price: Optional[float] = None          # 限价
    stop_price: Optional[float] = None     # 止损价
    time_in_force: str = "DAY"             # 有效期
    order_id: str = field(default_factory=lambda: str(uuid.uuid4()))
    status: OrderStatus = OrderStatus.PENDING
    filled_quantity: float = 0.0
    average_fill_price: float = 0.0
    created_at: datetime = field(default_factory=datetime.now)
    updated_at: datetime = field(default_factory=datetime.now)
    strategy_id: str = ""
    notes: str = ""

    @property
    def remaining_quantity(self) -> float:
        """剩余未成交数量"""
        return self.quantity - self.filled_quantity

    @property
    def is_active(self) -> bool:
        """订单是否活跃"""
        return self.status in [OrderStatus.PENDING, OrderStatus.SUBMITTED,
                               OrderStatus.PARTIAL_FILLED]

    def update_status(self, status: OrderStatus, filled_qty: float = None,
                      fill_price: float = None):
        """更新订单状态"""
        self.status = status
        self.updated_at = datetime.now()

        if filled_qty is not None:
            self.filled_quantity = filled_qty

        if fill_price is not None and filled_qty is not None:
            # 计算新的平均成交价
            total_value = (self.average_fill_price * self.filled_quantity +
                          fill_price * filled_qty)
            new_total_qty = self.filled_quantity + filled_qty
            if new_total_qty > 0:
                self.average_fill_price = total_value / new_total_qty

@dataclass
class Fill:
    """成交记录"""
    order_id: str
    symbol: str
    side: OrderSide
    quantity: float
    price: float
    commission: float = 0.0
    timestamp: datetime = field(default_factory=datetime.now)

    @property
    def notional(self) -> float:
        """名义金额"""
        return self.quantity * self.price

    @property
    def pnl(self) -> float:
        """实现盈亏（仅用于卖出）"""
        return 0  # 需要结合持仓成本计算
```

## 订单队列管理

```python
from collections import deque
from typing import Dict, List, Optional
import heapq

class OrderQueue:
    """订单队列"""

    def __init__(self, max_size: int = 1000):
        self.max_size = max_size
        self.pending_orders: Dict[str, Order] = {}
        self.order_heap: List[tuple] = []  # (priority, timestamp, order_id)
        self.filled_orders: Dict[str, Order] = {}
        self.cancelled_orders: Dict[str, Order] = {}

    def add_order(self, order: Order, priority: int = 0) -> bool:
        """添加订单到队列"""
        if len(self.pending_orders) >= self.max_size:
            print(f"❌ 队列已满 ({self.max_size})")
            return False

        self.pending_orders[order.order_id] = order

        # 添加到优先级堆
        heapq.heappush(
            self.order_heap,
            (priority, order.created_at.timestamp(), order.order_id)
        )

        print(f"✅ 订单已添加: {order.order_id[:8]}... {order.symbol} {order.side.value}")
        return True

    def get_next_order(self) -> Optional[Order]:
        """获取下一个待执行订单（按优先级）"""
        if not self.order_heap:
            return None

        _, _, order_id = heapq.heappop(self.order_heap)
        return self.pending_orders.get(order_id)

    def remove_order(self, order_id: str) -> Optional[Order]:
        """从队列移除订单"""
        if order_id in self.pending_orders:
            order = self.pending_orders.pop(order_id)
            print(f"🗑️ 订单已移除: {order_id[:8]}...")
            return order
        return None

    def update_order_status(self, order_id: str, status: OrderStatus,
                           filled_qty: float = None, fill_price: float = None):
        """更新订单状态"""
        if order_id not in self.pending_orders:
            return

        order = self.pending_orders[order_id]
        order.update_status(status, filled_qty, fill_price)

        if not order.is_active:
            self.pending_orders.pop(order_id)

            if status == OrderStatus.FILLED:
                self.filled_orders[order_id] = order
                print(f"📝 订单成交: {order_id[:8]}... {order.symbol} "
                      f"{order.filled_quantity}@{order.average_fill_price:.2f}")
            elif status == OrderStatus.CANCELLED:
                self.cancelled_orders[order_id] = order
                print(f"❌ 订单取消: {order_id[:8]}...")

    def get_order(self, order_id: str) -> Optional[Order]:
        """获取订单"""
        return (self.pending_orders.get(order_id) or
                self.filled_orders.get(order_id) or
                self.cancelled_orders.get(order_id))

    def get_active_orders(self, symbol: str = None) -> List[Order]:
        """获取活跃订单"""
        orders = list(self.pending_orders.values())

        if symbol:
            orders = [o for o in orders if o.symbol == symbol]

        return sorted(orders, key=lambda x: x.created_at)

    def cancel_all(self, symbol: str = None) -> List[str]:
        """取消所有订单"""
        active_orders = self.get_active_orders(symbol)
        cancelled_ids = []

        for order in active_orders:
            self.update_order_status(order.order_id, OrderStatus.CANCELLED)
            cancelled_ids.append(order.order_id)

        return cancelled_ids
```

## 订单执行处理器

```python
class ExecutionHandler:
    """订单执行处理器"""

    def __init__(self, broker_adapter, order_queue: OrderQueue):
        self.broker = broker_adapter
        self.order_queue = order_queue
        self.order_callbacks = []
        self.fill_callbacks = []

    def on_order_update(self, callback):
        """注册订单更新回调"""
        self.order_callbacks.append(callback)

    def on_fill(self, callback):
        """注册成交回调"""
        self.fill_callbacks.append(callback)

    def submit_order(self, order: Order, priority: int = 0) -> bool:
        """提交订单"""
        # 添加到队列
        if not self.order_queue.add_order(order, priority):
            return False

        # 更新状态为已提交
        self.order_queue.update_order_status(order.order_id, OrderStatus.SUBMITTED)

        # 发送到broker
        try:
            self._send_to_broker(order)
        except Exception as e:
            print(f"❌ 订单提交失败: {e}")
            self.order_queue.update_order_status(order.order_id, OrderStatus.REJECTED)
            return False

        return True

    def cancel_order(self, order_id: str) -> bool:
        """取消订单"""
        order = self.order_queue.get_order(order_id)
        if not order:
            print(f"❌ 订单不存在: {order_id[:8]}...")
            return False

        if not order.is_active:
            print(f"❌ 订单不可取消: {order.status.value}")
            return False

        try:
            self.broker.cancel_order(order_id)
            self.order_queue.update_order_status(order_id, OrderStatus.CANCELLED)
            return True
        except Exception as e:
            print(f"❌ 取消失败: {e}")
            return False

    def cancel_all_orders(self, symbol: str = None) -> List[str]:
        """取消所有订单"""
        return self.order_queue.cancel_all(symbol)

    def _send_to_broker(self, order: Order):
        """发送订单到broker"""
        # 根据订单类型构建broker订单
        broker_order = {
            'symbol': order.symbol,
            'side': order.side.value,
            'type': order.order_type.value,
            'quantity': order.quantity
        }

        if order.order_type == OrderType.LIMIT:
            broker_order['price'] = order.price
        elif order.order_type == OrderType.STOP:
            broker_order['stop_price'] = order.stop_price
        elif order.order_type == OrderType.STOP_LIMIT:
            broker_order['price'] = order.price
            broker_order['stop_price'] = order.stop_price

        self.broker.place_order(broker_order)

    def handle_broker_response(self, response: dict):
        """处理broker响应"""
        order_id = response.get('order_id')
        status = response.get('status')

        # 映射broker状态到系统状态
        status_map = {
            'new': OrderStatus.SUBMITTED,
            'partial': OrderStatus.PARTIAL_FILLED,
            'filled': OrderStatus.FILLED,
            'cancelled': OrderStatus.CANCELLED,
            'rejected': OrderStatus.REJECTED,
            'expired': OrderStatus.EXPIRED
        }

        new_status = status_map.get(status, OrderStatus.PENDING)

        filled_qty = response.get('filled_qty')
        fill_price = response.get('fill_price')

        self.order_queue.update_order_status(order_id, new_status, filled_qty, fill_price)

        # 触发回调
        for callback in self.order_callbacks:
            callback(order_id, new_status)

        # 如果是成交，触发成交回调
        if new_status in [OrderStatus.FILLED, OrderStatus.PARTIAL_FILLED]:
            for callback in self.fill_callbacks:
                callback(response)
```

## 订单优先级

```python
class PriorityCalculator:
    """订单优先级计算器"""

    @staticmethod
    def calculate_priority(order: Order, current_positions: dict) -> int:
        """
        计算机位优先级

        返回值越小优先级越高
        """
        priority = 1000  # 基础优先级

        # 平仓订单优先于开仓订单
        if order.side == OrderSide.SELL and current_positions.get(order.symbol, 0) > 0:
            priority -= 500  # 平仓优先
        elif order.side == OrderSide.BUY and current_positions.get(order.symbol, 0) < 0:
            priority -= 500  # 平空仓优先

        # 高价值订单优先
        notional = order.quantity * (order.price or 0)
        priority -= min(notional / 10000, 100)  # 最多减100

        # 策略优先级
        if 'high_priority' in order.strategy_id:
            priority -= 200

        # 时间因素（早提交的优先）
        # 已经在heap中考虑了时间

        return max(priority, 0)

# 使用示例
positions = {'AAPL': 100, 'TSLA': -50}
calc = PriorityCalculator()

order1 = Order('AAPL', OrderSide.SELL, OrderType.MARKET, 50)
order2 = Order('TSLA', OrderSide.BUY, OrderType.LIMIT, 100, price=200)

print(f"AAPL卖单优先级: {calc.calculate_priority(order1, positions)}")
print(f"TSLA买单优先级: {calc.calculate_priority(order2, positions)}")
```

## 滑点控制

```python
class SlippageController:
    """滑点控制器"""

    def __init__(self, max_slippage_pct: float = 0.005):
        self.max_slippage_pct = max_slippage_pct

    def check_slippage(self, order: Order, current_price: float) -> tuple[bool, float]:
        """
        检查滑点是否可接受

        Returns:
            (is_acceptable, max_acceptable_price)
        """
        if order.order_type == OrderType.MARKET:
            # 市价单：根据方向计算最大可接受价格
            if order.side == OrderSide.BUY:
                max_price = current_price * (1 + self.max_slippage_pct)
            else:
                max_price = current_price * (1 - self.max_slippage_pct)

            return True, max_price

        elif order.order_type == OrderType.LIMIT:
            # 限价单：检查限价是否合理
            deviation = abs(order.price - current_price) / current_price

            if deviation > self.max_slippage_pct * 2:
                return False, order.price

            return True, order.price

        return True, current_price

# 使用示例
controller = SlippageController(max_slippage_pct=0.005)

order = Order('AAPL', OrderSide.BUY, OrderType.MARKET, 100)
current_price = 175.00

acceptable, max_price = controller.check_slippage(order, current_price)
print(f"可接受: {acceptable}, 最大价格: ${max_price:.2f}")
```

## 完整OMS实现

```python
class OrderManagementSystem:
    """订单管理系统"""

    def __init__(self, broker_adapter, config: dict = None):
        self.config = config or {}

        # 初始化组件
        self.order_queue = OrderQueue(max_size=self.config.get('max_queue_size', 1000))
        self.execution_handler = ExecutionHandler(broker_adapter, self.order_queue)
        self.slippage_controller = SlippageController(
            max_slippage_pct=self.config.get('max_slippage', 0.005)
        )
        self.priority_calculator = PriorityCalculator()

        # 持仓状态
        self.positions: Dict[str, float] = {}
        self.realized_pnl: Dict[str, float] = {}

        # 注册内部回调
        self.execution_handler.on_fill(self._on_fill)

    def create_order(self, symbol: str, side: str, quantity: float,
                    order_type: str = 'market', price: float = None,
                    stop_price: float = None, strategy_id: str = '') -> Order:
        """创建订单"""
        return Order(
            symbol=symbol,
            side=OrderSide.BUY if side.lower() == 'buy' else OrderSide.SELL,
            order_type=OrderType[order_type.upper()],
            quantity=quantity,
            price=price,
            stop_price=stop_price,
            strategy_id=strategy_id
        )

    def place_order(self, order: Order, check_risk: bool = True) -> str:
        """
        下单

        Returns:
            order_id if successful, None otherwise
        """
        # 风控检查（可以接入风控管理器）
        # if check_risk and not self.risk_manager.check_order(order):
        #     return None

        # 计算机位优先级
        priority = self.priority_calculator.calculate_priority(order, self.positions)

        # 提交订单
        if self.execution_handler.submit_order(order, priority):
            return order.order_id

        return None

    def cancel_order(self, order_id: str) -> bool:
        """取消订单"""
        return self.execution_handler.cancel_order(order_id)

    def cancel_all_orders(self, symbol: str = None) -> List[str]:
        """取消所有订单"""
        return self.execution_handler.cancel_all_orders(symbol)

    def get_order(self, order_id: str) -> Optional[Order]:
        """查询订单"""
        return self.order_queue.get_order(order_id)

    def get_open_orders(self, symbol: str = None) -> List[Order]:
        """获取待成交订单"""
        return self.order_queue.get_active_orders(symbol)

    def get_positions(self) -> Dict[str, float]:
        """获取当前持仓"""
        return self.positions.copy()

    def get_realized_pnl(self) -> Dict[str, float]:
        """获取已实现盈亏"""
        return self.realized_pnl.copy()

    def _on_fill(self, fill_data: dict):
        """处理成交"""
        order_id = fill_data['order_id']
        symbol = fill_data['symbol']
        quantity = fill_data['quantity']
        side = OrderSide.BUY if fill_data['side'] == 'buy' else OrderSide.SELL

        # 更新持仓
        if side == OrderSide.BUY:
            self.positions[symbol] = self.positions.get(symbol, 0) + quantity
        else:
            self.positions[symbol] = self.positions.get(symbol, 0) - quantity

        print(f"📊 持仓更新: {symbol} = {self.positions[symbol]}")
```
