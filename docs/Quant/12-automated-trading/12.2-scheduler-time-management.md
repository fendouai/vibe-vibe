### 时间相关概念

```python
from datetime import datetime, time
import pytz

# 定义不同时区的交易时间
TRADING_HOURS = {
    'US': {
        'pre_market': time(4, 0),   # 04:00 EST
        'regular': (time(9, 30), time(16, 0)),
        'after_hours': time(20, 0), # 16:00后盘后
        'timezone': 'America/New_York'
    },
    'CN': {
        'morning': (time(9, 30), time(11, 30)),
        'afternoon': (time(13, 0), time(15, 0)),
        'timezone': 'Asia/Shanghai'
    },
    'CRYPTO': {
        'open': time(0, 0),  # 24小时
        'close': time(23, 59, 59),
        'timezone': 'UTC'
    }
}

def is_market_open(market: str, current_time: datetime = None) -> bool:
    """检查市场是否开盘"""
    if current_time is None:
        current_time = datetime.now()

    if market == 'CRYPTO':
        return True  # 7x24

    hours = TRADING_HOURS.get(market)
    if not hours:
        return False

    tz = pytz.timezone(hours['timezone'])
    local_time = current_time.astimezone(tz).time()

    if market == 'US':
        # 美股：只在regular hours交易
        reg_start, reg_end = hours['regular']
        return reg_start <= local_time <= reg_end
    elif market == 'CN':
        # A股分上下午
        morn_start, morn_end = hours['morning']
        aft_start, aft_end = hours['afternoon']
        return (morn_start <= local_time <= morn_end or
                aft_start <= local_time <= aft_end)

    return False
```

## 交易日历管理

```python
import pandas as pd
from datetime import datetime, timedelta

class TradingCalendar:
    """交易日历"""

    def __init__(self):
        # 中国A股交易日历（简单实现，实际应使用专业数据）
        self.cn_holidays = [
            '2024-01-01',  # 元旦
            '2024-02-10',  # 春节
            '2024-04-04',  # 清明
            '2024-05-01',  # 劳动节
            '2024-06-10',  # 端午节
            '2024-09-17',  # 中秋
            '2024-10-01',  # 国庆
        ]

    def is_trading_day(self, date: datetime) -> bool:
        """判断是否交易日"""
        # 周末检查
        if date.weekday() >= 5:  # 周六周日
            return False

        # 节假日检查
        date_str = date.strftime('%Y-%m-%d')
        if date_str in self.cn_holidays:
            return False

        return True

    def get_next_trading_day(self, date: datetime) -> datetime:
        """获取下一个交易日"""
        next_date = date + timedelta(days=1)
        while not self.is_trading_day(next_date):
            next_date += timedelta(days=1)
        return next_date

    def get_previous_trading_day(self, date: datetime) -> datetime:
        """获取上一个交易日"""
        prev_date = date - timedelta(days=1)
        while not self.is_trading_day(prev_date):
            prev_date -= timedelta(days=1)
        return prev_date

    def get_trading_days(self, start: datetime, end: datetime) -> list:
        """获取时间段内的所有交易日"""
        days = []
        current = start
        while current <= end:
            if self.is_trading_day(current):
                days.append(current)
            current += timedelta(days=1)
        return days

# 使用示例
calendar = TradingCalendar()
today = datetime.now()

print(f"今天是否交易日: {calendar.is_trading_day(today)}")
print(f"下一个交易日: {calendar.get_next_trading_day(today)}")
```

## 时间区处理

```python
from datetime import datetime, timezone
import pytz

def convert_timezone(dt: datetime, from_tz: str, to_tz: str) -> datetime:
    """转换时区"""
    # 如果datetime是naive，添加from_tz
    if dt.tzinfo is None:
        from_timezone = pytz.timezone(from_tz)
        dt = from_timezone.localize(dt)

    # 转换为目标时区
    to_timezone = pytz.timezone(to_tz)
    return dt.astimezone(to_timezone)

def get_utc_now() -> datetime:
    """获取当前UTC时间"""
    return datetime.now(timezone.utc)

def is_us_market_hours(dt: datetime = None) -> bool:
    """检查是否美股交易时间"""
    if dt is None:
        dt = get_utc_now()

    # 转换为美东时间
    est = pytz.timezone('America/New_York')
    us_time = dt.astimezone(est)

    # 检查是否在交易时间内 (9:30 - 16:00 EST)
    market_time = us_time.time()
    market_start = time(9, 30)
    market_end = time(16, 0)

    # 周一到周五
    if us_time.weekday() >= 5:
        return False

    return market_start <= market_time <= market_end

# 示例
print(f"当前UTC时间: {get_utc_now()}")
print(f"是否美股开盘: {is_us_market_hours()}")
```

## 定时任务类

```python
import schedule
from threading import Thread
from typing import Callable, Optional
from datetime import datetime
from enum import Enum

class TaskFrequency(Enum):
    """任务频率"""
    ONCE = "once"
    MINUTE = "minute"
    HOURLY = "hourly"
    DAILY = "daily"
    WEEKLY = "weekly"

class ScheduledTask:
    """定时任务"""

    def __init__(self, name: str, func: Callable, frequency: TaskFrequency,
                 time_spec: str = None):
        self.name = name
        self.func = func
        self.frequency = frequency
        self.time_spec = time_spec
        self.last_run = None
        self.enabled = True

    def setup(self):
        """设置调度规则"""
        if self.frequency == TaskFrequency.MINUTE:
            schedule.every().minute.do(self._run_wrapper)
        elif self.frequency == TaskFrequency.HOUR:
            schedule.every().hour.do(self._run_wrapper)
        elif self.frequency == TaskFrequency.DAILY:
            schedule.every().day.at(self.time_spec or "00:00").do(self._run_wrapper)
        elif self.frequency == TaskFrequency.WEEKLY:
            schedule.every().monday.at(self.time_spec or "00:00").do(self._run_wrapper)

    def _run_wrapper(self):
        """运行包装器"""
        if self.enabled:
            print(f"[{datetime.now()}] 执行任务: {self.name}")
            try:
                self.func()
                self.last_run = datetime.now()
            except Exception as e:
                print(f"任务执行错误 {self.name}: {e}")

class Scheduler:
    """调度器管理"""

    def __init__(self):
        self.tasks = []
        self.running = False
        self.thread = None

    def add_task(self, task: ScheduledTask):
        """添加任务"""
        task.setup()
        self.tasks.append(task)

    def remove_task(self, name: str):
        """移除任务"""
        schedule.clear(name)
        self.tasks = [t for t in self.tasks if t.name != name]

    def enable_task(self, name: str):
        """启用任务"""
        for task in self.tasks:
            if task.name == name:
                task.enabled = True

    def disable_task(self, name: str):
        """禁用任务"""
        for task in self.tasks:
            if task.name == name:
                task.enabled = False

    def start(self):
        """启动调度器"""
        self.running = True
        self.thread = Thread(target=self._run_loop, daemon=True)
        self.thread.start()
        print("✅ 调度器已启动")

    def stop(self):
        """停止调度器"""
        self.running = False
        if self.thread:
            self.thread.join()
        print("✅ 调度器已停止")

    def _run_loop(self):
        """调度循环"""
        while self.running:
            schedule.run_pending()
            import time
            time.sleep(1)

# 使用示例
scheduler = Scheduler()

# 添加各种任务
scheduler.add_task(ScheduledTask(
    name='数据更新',
    func=lambda: print("更新数据..."),
    frequency=TaskFrequency.MINUTE
))

scheduler.add_task(ScheduledTask(
    name='收盘处理',
    func=lambda: print("收盘处理..."),
    frequency=TaskFrequency.DAILY,
    time_spec="15:05"
))

scheduler.start()
```

## 交易时间状态机

```python
from enum import Enum

class MarketSession(Enum):
    """市场会话状态"""
    CLOSED = "closed"           # 休市
    PRE_MARKET = "pre_market"   # 盘前
    REGULAR = "regular"         # 盘中
    AFTER_HOURS = "after_hours" # 盘后

class MarketSessionManager:
    """市场会话管理器"""

    def __init__(self, market: str = 'US'):
        self.market = market
        self.session = MarketSession.CLOSED
        self.session_start = None

    def update(self, current_time: datetime = None):
        """更新市场会话状态"""
        if current_time is None:
            current_time = get_utc_now()

        if self.market == 'US':
            self._update_us_session(current_time)
        elif self.market == 'CN':
            self._update_cn_session(current_time)
        elif self.market == 'CRYPTO':
            self.session = MarketSession.REGULAR  # 24x7

    def _update_us_session(self, current_time: datetime):
        """更新美股会话"""
        est = pytz.timezone('America/New_York')
        local_time = current_time.astimezone(est).time()

        if local_time < time(4, 0):
            new_session = MarketSession.CLOSED
        elif local_time < time(9, 30):
            new_session = MarketSession.PRE_MARKET
        elif local_time <= time(16, 0):
            new_session = MarketSession.REGULAR
        else:
            new_session = MarketSession.AFTER_HOURS

        if new_session != self.session:
            print(f"会话切换: {self.session.value} -> {new_session.value}")
            self.session = new_session
            self.session_start = current_time

    def _update_cn_session(self, current_time: datetime):
        """更新A股会话"""
        cst = pytz.timezone('Asia/Shanghai')
        local_time = current_time.astimezone(cst).time()

        if local_time < time(9, 30):
            new_session = MarketSession.CLOSED
        elif local_time < time(11, 30):
            new_session = MarketSession.REGULAR
        elif local_time < time(13, 0):
            new_session = MarketSession.CLOSED  # 午休
        elif local_time <= time(15, 0):
            new_session = MarketSession.REGULAR
        else:
            new_session = MarketSession.CLOSED

        if new_session != self.session:
            print(f"会话切换: {self.session.value} -> {new_session.value}")
            self.session = new_session
            self.session_start = current_time

    def can_trade(self) -> bool:
        """是否可以交易"""
        return self.session == MarketSession.REGULAR

    def should_collect_data(self) -> bool:
        """是否应该收集数据"""
        return self.session in [MarketSession.REGULAR, MarketSession.AFTER_HOURS]

# 使用示例
session_mgr = MarketSessionManager('US')
session_mgr.update()
print(f"当前会话: {session_mgr.session.value}")
print(f"可交易: {session_mgr.can_trade()}")
```
