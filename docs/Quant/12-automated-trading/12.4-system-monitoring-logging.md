# 12.4 系统监控与日志

## 日志的重要性

交易系统的日志是问题排查和合规审计的关键：

- **问题诊断**: 系统异常时，日志是唯一的追踪线索
- **性能分析**: 通过日志分析系统瓶颈
- **合规审计**: 监管要求保留完整的交易记录
- **策略复盘**: 还原交易决策过程

## Python日志模块

```python
import logging
from logging.handlers import RotatingFileHandler, TimedRotatingFileHandler
import sys
from datetime import datetime

def setup_logger(name: str, log_file: str = None, level: int = logging.INFO,
                max_bytes: int = 10*1024*1024, backup_count: int = 10):
    """
    配置日志器

    Parameters:
    - name: 日志器名称
    - log_file: 日志文件路径
    - level: 日志级别
    - max_bytes: 单个日志文件最大大小
    - backup_count: 保留的备份文件数量
    """
    logger = logging.getLogger(name)
    logger.setLevel(level)

    # 清除已有的处理器
    logger.handlers.clear()

    # 日志格式
    formatter = logging.Formatter(
        fmt='%(asctime)s | %(levelname)-8s | %(name)s | %(message)s',
        datefmt='%Y-%m-%d %H:%M:%S'
    )

    # 控制台处理器
    console_handler = logging.StreamHandler(sys.stdout)
    console_handler.setLevel(level)
    console_handler.setFormatter(formatter)
    logger.addHandler(console_handler)

    # 文件处理器（按大小轮转）
    if log_file:
        file_handler = RotatingFileHandler(
            log_file,
            maxBytes=max_bytes,
            backupCount=backup_count,
            encoding='utf-8'
        )
        file_handler.setLevel(level)
        file_handler.setFormatter(formatter)
        logger.addHandler(file_handler)

    return logger

# 创建系统日志器
system_logger = setup_logger('trading_system', 'logs/system.log')
trade_logger = setup_logger('trades', 'logs/trades.log')
risk_logger = setup_logger('risk', 'logs/risk.log')
```

## 分层日志架构

```python
class TradingLogger:
    """交易系统日志管理器"""

    def __init__(self, log_dir: str = 'logs'):
        self.log_dir = log_dir

        # 系统日志
        self.system = setup_logger('system', f'{log_dir}/system.log')

        # 交易日志
        self.trade = setup_logger('trade', f'{log_dir}/trades.log')

        # 订单日志
        self.order = setup_logger('order', f'{log_dir}/orders.log')

        # 风险日志
        self.risk = setup_logger('risk', f'{log_dir}/risk.log')

        # 性能日志
        self.performance = setup_logger('performance', f'{log_dir}/performance.log')

        # 错误日志
        self.error = setup_logger('error', f'{log_dir}/error.log')

    def log_trade(self, symbol: str, side: str, quantity: float, price: float,
                  order_id: str = None, strategy: str = None):
        """记录交易"""
        self.trade.info(
            f"TRADE | order={order_id or 'N/A'} | {side} {quantity} {symbol} @ {price} | strategy={strategy or 'N/A'}"
        )

    def log_signal(self, symbol: str, direction: str, strength: float,
                   strategy: str, price: float):
        """记录信号"""
        self.system.info(
            f"SIGNAL | {strategy} | {symbol} | {direction} ({strength:.2f}) @ {price}"
        )

    def log_order(self, order_id: str, symbol: str, order_type: str,
                  status: str, message: str = ''):
        """记录订单"""
        self.order.info(
            f"ORDER | {order_id} | {symbol} | {order_type} | {status} | {message}"
        )

    def log_risk_event(self, event_type: str, message: str, details: dict = None):
        """记录风险事件"""
        detail_str = f" | {details}" if details else ""
        self.risk.warning(
            f"RISK_EVENT | {event_type} | {message}{detail_str}"
        )

    def log_performance(self, portfolio_value: float, daily_return: float,
                       sharpe: float = None, drawdown: float = None):
        """记录性能指标"""
        msg = f"VALUE={portfolio_value:.2f} | RETURN={daily_return:.2%}"
        if sharpe:
            msg += f" | SHARPE={sharpe:.2f}"
        if drawdown:
            msg += f" | DD={drawdown:.2%}"
        self.performance.info(msg)

    def log_error(self, component: str, error: Exception, context: dict = None):
        """记录错误"""
        ctx_str = f" | {context}" if context else ""
        self.error.error(
            f"ERROR | {component} | {type(error).__name__}: {str(error)}{ctx_str}",
            exc_info=True
        )

# 使用示例
logger = TradingLogger('logs')

# 记录交易
logger.log_trade('AAPL', 'BUY', 100, 175.50, 'order-123', 'momentum_v1')

# 记录信号
logger.log_signal('TSLA', 'LONG', 0.85, 'momentum_v1', 245.00)

# 记录订单
logger.log_order('order-123', 'AAPL', 'LIMIT', 'FILLED')

# 记录风险事件
logger.log_risk_event('POSITION_LIMIT', 'AAPL仓位超限', {'current': 0.15, 'max': 0.10})

# 记录性能
logger.log_performance(105000, 0.012, sharpe=1.85, drawdown=-0.03)

# 记录错误
try:
    1/0
except Exception as e:
    logger.log_error('order_executor', e, {'order_id': 'order-123'})
```

## 交易审计日志

```python
from datetime import datetime
import json
from pathlib import Path

class AuditLogger:
    """交易审计日志"""

    def __init__(self, audit_dir: str = 'logs/audit'):
        self.audit_dir = Path(audit_dir)
        self.audit_dir.mkdir(parents=True, exist_ok=True)

        # 按日期组织审计日志
        self.current_date = datetime.now().strftime('%Y-%m-%d')
        self.audit_file = self.audit_dir / f"audit_{self.current_date}.jsonl"

    def _ensure_new_file(self):
        """检查是否需要创建新文件（日期变化）"""
        today = datetime.now().strftime('%Y-%m-%d')
        if today != self.current_date:
            self.current_date = today
            self.audit_file = self.audit_dir / f"audit_{self.current_date}.jsonl"

    def log(self, event_type: str, event_data: dict):
        """
        记录审计事件

        event_type: ORDER_PLACED, ORDER_FILLED, ORDER_CANCELLED, etc.
        """
        self._ensure_new_file()

        record = {
            'timestamp': datetime.now().isoformat(),
            'event_type': event_type,
            **event_data
        }

        with open(self.audit_file, 'a') as f:
            f.write(json.dumps(record) + '\n')

    def log_order_placed(self, order: dict):
        """记录订单提交"""
        self.log('ORDER_PLACED', {
            'order_id': order.get('order_id'),
            'symbol': order.get('symbol'),
            'side': order.get('side'),
            'quantity': order.get('quantity'),
            'price': order.get('price'),
            'order_type': order.get('order_type'),
            'strategy_id': order.get('strategy_id', '')
        })

    def log_order_filled(self, fill: dict):
        """记录订单成交"""
        self.log('ORDER_FILLED', {
            'order_id': fill.get('order_id'),
            'symbol': fill.get('symbol'),
            'side': fill.get('side'),
            'quantity': fill.get('quantity'),
            'price': fill.get('price'),
            'commission': fill.get('commission', 0)
        })

    def log_position_change(self, symbol: str, old_position: float,
                           new_position: float, reason: str):
        """记录持仓变化"""
        self.log('POSITION_CHANGED', {
            'symbol': symbol,
            'old_position': old_position,
            'new_position': new_position,
            'change': new_position - old_position,
            'reason': reason
        })

    def query(self, event_type: str = None, start_time: datetime = None,
              end_time: datetime = None, symbol: str = None) -> list:
        """查询审计日志"""
        results = []

        # 如果查询跨日期，需要合并多个文件
        dates = []
        if start_time and end_time:
            current = start_time
            while current.date() <= end_time.date():
                dates.append(current.strftime('%Y-%m-%d'))
                current = datetime.combine(current.date() + __import__('datetime').timedelta(days=1),
                                          datetime.min.time())
        else:
            dates = [self.current_date]

        for date in dates:
            audit_file = self.audit_dir / f"audit_{date}.jsonl"
            if not audit_file.exists():
                continue

            with open(audit_file, 'r') as f:
                for line in f:
                    record = json.loads(line)

                    # 过滤条件
                    if event_type and record['event_type'] != event_type:
                        continue
                    if symbol and record.get('symbol') != symbol:
                        continue

                    results.append(record)

        return results

# 使用示例
audit = AuditLogger()

# 记录订单提交
audit.log_order_placed({
    'order_id': 'order-123',
    'symbol': 'AAPL',
    'side': 'BUY',
    'quantity': 100,
    'price': 175.50,
    'order_type': 'LIMIT',
    'strategy_id': 'momentum_v1'
})

# 记录成交
audit.log_order_filled({
    'order_id': 'order-123',
    'symbol': 'AAPL',
    'side': 'BUY',
    'quantity': 100,
    'price': 175.50,
    'commission': 0.50
})

# 查询今天的AAPL相关事件
today_orders = audit.query(symbol='AAPL')
```

## 系统健康监控

```python
import psutil
import time
from threading import Thread
from typing import Dict, Callable

class SystemMonitor:
    """系统健康监控"""

    def __init__(self, check_interval: int = 60):
        self.check_interval = check_interval
        self.running = False
        self.thread = None
        self.alert_callbacks = []
        self.metrics_history = []

        # 告警阈值
        self.thresholds = {
            'cpu_percent': 80.0,
            'memory_percent': 85.0,
            'disk_percent': 90.0,
            'latency_ms': 1000
        }

    def add_alert_callback(self, callback: Callable):
        """添加告警回调"""
        self.alert_callbacks.append(callback)

    def check_health(self) -> dict:
        """检查系统健康状态"""
        metrics = {
            'timestamp': datetime.now(),
            'cpu_percent': psutil.cpu_percent(interval=1),
            'memory_percent': psutil.virtual_memory().percent,
            'disk_percent': psutil.disk_usage('/').percent,
            'network_connections': len(psutil.net_connections()),
            'thread_count': psutil.Process().num_threads()
        }

        self.metrics_history.append(metrics)

        # 保留最近1000条记录
        if len(self.metrics_history) > 1000:
            self.metrics_history = self.metrics_history[-1000:]

        # 检查阈值
        for metric, value in metrics.items():
            if metric == 'timestamp':
                continue
            threshold = self.thresholds.get(metric)
            if threshold and value > threshold:
                self._trigger_alert(metric, value, threshold)

        return metrics

    def _trigger_alert(self, metric: str, value: float, threshold: float):
        """触发告警"""
        alert = {
            'timestamp': datetime.now(),
            'metric': metric,
            'value': value,
            'threshold': threshold,
            'severity': 'warning' if value < threshold * 1.1 else 'critical'
        }

        print(f"⚠️ 告警: {metric}={value:.1f} (阈值: {threshold})")

        for callback in self.alert_callbacks:
            try:
                callback(alert)
            except Exception as e:
                print(f"告警回调错误: {e}")

    def start(self):
        """启动监控"""
        self.running = True
        self.thread = Thread(target=self._monitor_loop, daemon=True)
        self.thread.start()
        print("✅ 系统监控已启动")

    def stop(self):
        """停止监控"""
        self.running = False
        if self.thread:
            self.thread.join()
        print("✅ 系统监控已停止")

    def _monitor_loop(self):
        """监控循环"""
        while self.running:
            self.check_health()
            time.sleep(self.check_interval)

    def get_metrics_summary(self, last_n: int = 100) -> dict:
        """获取指标摘要"""
        recent = self.metrics_history[-last_n:]

        if not recent:
            return {}

        return {
            'cpu_avg': sum(m['cpu_percent'] for m in recent) / len(recent),
            'cpu_max': max(m['cpu_percent'] for m in recent),
            'memory_avg': sum(m['memory_percent'] for m in recent) / len(recent),
            'memory_max': max(m['memory_percent'] for m in recent),
        }

# 使用示例
def send_alert(alert):
    """发送告警"""
    print(f"🚨 告警通知: {alert['metric']} 异常!")

monitor = SystemMonitor(check_interval=30)
monitor.add_alert_callback(send_alert)
monitor.start()
```

## 性能监控

```python
import time
from functools import wraps
from contextlib import contextmanager

def timing_decorator(func):
    """函数执行时间装饰器"""
    @wraps(func)
    def wrapper(*args, **kwargs):
        start = time.perf_counter()
        result = func(*args, **kwargs)
        elapsed = (time.perf_counter() - start) * 1000  # ms
        print(f"⏱️ {func.__name__}: {elapsed:.2f}ms")
        return result
    return wrapper

@contextmanager
def timing_context(name: str):
    """代码块执行时间上下文管理器"""
    start = time.perf_counter()
    try:
        yield
    finally:
        elapsed = (time.perf_counter() - start) * 1000
        print(f"⏱️ {name}: {elapsed:.2f}ms")

class PerformanceTracker:
    """性能追踪器"""

    def __init__(self):
        self.timings = {}
        self.call_counts = {}

    def record(self, name: str, elapsed_ms: float):
        """记录执行时间"""
        if name not in self.timings:
            self.timings[name] = []
            self.call_counts[name] = 0

        self.timings[name].append(elapsed_ms)
        self.call_counts[name] += 1

    def get_stats(self, name: str) -> dict:
        """获取统计信息"""
        if name not in self.timings or not self.timings[name]:
            return {}

        timings = self.timings[name]
        return {
            'count': self.call_counts[name],
            'total_ms': sum(timings),
            'avg_ms': sum(timings) / len(timings),
            'min_ms': min(timings),
            'max_ms': max(timings)
        }

    def report(self) -> str:
        """生成性能报告"""
        lines = ["="*60, "性能报告", "="*60]

        for name in sorted(self.timings.keys()):
            stats = self.get_stats(name)
            lines.append(
                f"{name:30s} | "
                f"调用: {stats['count']:5d} | "
                f"平均: {stats['avg_ms']:8.2f}ms | "
                f"最大: {stats['max_ms']:8.2f}ms"
            )

        return "\n".join(lines)

# 使用示例
perf = PerformanceTracker()

# 使用装饰器
@timing_decorator
def fetch_data():
    time.sleep(0.1)

# 使用上下文管理器
with timing_context('数据处理'):
    time.sleep(0.2)

# 记录
perf.record('data_fetch', 105.3)
perf.record('data_process', 203.8)

print(perf.report())
```
