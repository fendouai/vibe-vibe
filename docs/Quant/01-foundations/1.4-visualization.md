# 1.4 数据可视化：Matplotlib、Seaborn 基础

## 为什么可视化很重要

在量化交易中，可视化用于：

- **探索性分析**：理解数据特征
- **策略验证**：查看策略信号和表现
- **报告展示**：清晰展示交易结果

## Matplotlib 基础

### 基本图表

```python
import matplotlib.pyplot as plt
import numpy as np

# 最简单的折线图
x = [1, 2, 3, 4, 5]
y = [100, 102, 98, 105, 103]

plt.figure(figsize=(10, 6))
plt.plot(x, y)
plt.title('Stock Price')
plt.xlabel('Day')
plt.ylabel('Price ($)')
plt.grid(True)
plt.show()
```

## 金融图表

### K线图（Candlestick Chart）

```python
import matplotlib.pyplot as plt
import pandas as pd
import yfinance as yf

# 获取数据
df = yf.download('AAPL', start='2023-01-01', end='2023-03-31')
df.columns = df.columns.get_level_values(0)

# 绘制K线图
fig, ax = plt.subplots(figsize=(12, 6))

# 计算颜色
colors = ['green' if df['Close'][i] >= df['Open'][i] else 'red'
          for i in range(len(df))]

# 绘制
for i in range(len(df)):
    color = 'green' if df['Close'].iloc[i] >= df['Open'].iloc[i] else 'red'
    # 绘制实体
    ax.add_patch(plt.Rectangle(
        (i - 0.4, min(df['Open'].iloc[i], df['Close'].iloc[i])),
        0.8,
        abs(df['Close'].iloc[i] - df['Open'].iloc[i]),
        facecolor=color, edgecolor=color, alpha=0.8
    ))
    # 绘制上下影线
    ax.plot([i, i], [df['Low'].iloc[i], df['High'].iloc[i]], color=color)

ax.set_xlim(-0.5, len(df) - 0.5)
ax.set_ylim(df['Low'].min() * 0.99, df['High'].max() * 1.01)
plt.title('AAPL Candlestick Chart')
plt.show()
```

### 收益率分布图

```python
import matplotlib.pyplot as plt
import pandas as pd
import yfinance as yf
from scipy import stats

# 获取数据
df = yf.download('AAPL', start='2022-01-01', end='2023-12-31')
df.columns = df.columns.get_level_values(0)
returns = df['Close'].pct_change().dropna()

# 绘制分布图
fig, axes = plt.subplots(1, 2, figsize=(14, 5))

# 直方图
axes[0].hist(returns, bins=50, edgecolor='black', alpha=0.7)
axes[0].axvline(returns.mean(), color='red', linestyle='--', label=f'Mean: {returns.mean():.4f}')
axes[0].set_title('Daily Return Distribution')
axes[0].set_xlabel('Return')
axes[0].set_ylabel('Frequency')
axes[0].legend()

# 箱线图
axes[1].boxplot(returns)
axes[1].set_title('Return Box Plot')
axes[1].set_ylabel('Return')

plt.tight_layout()
plt.show()
```

### 多股票对比图

```python
import matplotlib.pyplot as plt
import pandas as pd
import yfinance as yf

# 获取多只股票
stocks = ['AAPL', 'GOOGL', 'MSFT', 'AMZN']
data = yf.download(stocks, start='2022-01-01', end='2023-12-31')['Close']
data.columns = data.columns.get_level_values(0)

# 计算归一化价格（起点100）
normalized = data / data.iloc[0] * 100

# 绘图
plt.figure(figsize=(12, 6))
for stock in stocks:
    plt.plot(normalized.index, normalized[stock], label=stock)

plt.title('Normalized Stock Performance (Base=100)')
plt.xlabel('Date')
plt.ylabel('Normalized Price')
plt.legend()
plt.grid(True, alpha=0.3)
plt.show()
```

## Seaborn 高级可视化

Seaborn 基于 Matplotlib，提供了更美观的默认样式。

```python
import seaborn as sns
import matplotlib.pyplot as plt
import pandas as pd
import numpy as np

# 设置风格
sns.set_style('darkgrid')
sns.set_palette('husl')

# 关系图
fig, axes = plt.subplots(2, 2, figsize=(14, 10))

# 1. 散点图 + 回归线
x = np.random.randn(100)
y = x * 0.8 + np.random.randn(100) * 0.3
sns.regplot(x=x, y=y, ax=axes[0, 0])
axes[0, 0].set_title('Scatter with Regression')

# 2. 热力图（股票相关性）
stocks = ['AAPL', 'GOOGL', 'MSFT', 'AMZN', 'TSLA']
data = yf.download(stocks, start='2022-01-01', end='2023-12-31')['Close']
data.columns = data.columns.get_level_values(0)
returns = data.pct_change().dropna()
corr = returns.corr()
sns.heatmap(corr, annot=True, cmap='RdYlGn', center=0, ax=axes[0, 1])
axes[0, 1].set_title('Stock Return Correlation')

# 3. pairplot（多变量关系）
sample_data = returns[['AAPL', 'GOOGL', 'MSFT']].sample(100)
sns.pairplot(sample_data, diag_kind='kde', ax=axes[1, 0])

# 4. 时间序列（带置信区间）
dates = pd.date_range('2023-01-01', periods=100)
ts = pd.Series(np.cumsum(np.random.randn(100)), index=dates)
sns.lineplot(data=ts, ax=axes[1, 1])
axes[1, 1].set_title('Time Series with Trend')

plt.tight_layout()
plt.show()
```

## 交互式图表（Plotly）

Plotly 提供交互式图表，适合在 Jupyter Notebook 中使用。

```python
import plotly.graph_objects as go
import pandas as pd
import yfinance as yf

# 获取数据
df = yf.download('AAPL', start='2023-01-01', end='2023-03-31')
df.columns = df.columns.get_level_values(0)

# 创建K线图
fig = go.Figure(data=[go.Candlestick(
    x=df.index,
    open=df['Open'],
    high=df['High'],
    low=df['Low'],
    close=df['Close']
)])

fig.update_layout(
    title='AAPL Candlestick Chart',
    yaxis_title='Price',
    xaxis_rangeslider_visible=False
)

fig.show()  # 在Jupyter中显示交互式图表
```

## 常见图表类型

| 图表类型 | 用途           | 函数                                 |
| -------- | -------------- | ------------------------------------ |
| 折线图   | 价格走势、均线 | `plt.plot()`, `sns.lineplot()`       |
| K线图    | 显示OHLC       | candlestick_ohlc                     |
| 直方图   | 收益率分布     | `plt.hist()`, `sns.histplot()`       |
| 箱线图   | 异常值检测     | `plt.boxplot()`, `sns.boxplot()`     |
| 散点图   | 资产关系       | `plt.scatter()`, `sns.scatterplot()` |
| 热力图   | 相关性矩阵     | `sns.heatmap()`                      |
| 面积图   | 累计收益       | `plt.fill_between()`                 |

# 1. AAPL 收盘价和均线（MA5, MA20, MA60）
# 2. 成交量柱状图
# 3. MACD 指标
# 4. RSI 指标

```python
import pandas as pd
import yfinance as yf
import matplotlib.pyplot as plt

# 获取数据
df = yf.download('AAPL', start='2022-01-01', end='2023-12-31')
df.columns = df.columns.get_level_values(0)

# 计算指标
df['MA5'] = df['Close'].rolling(5).mean()
df['MA20'] = df['Close'].rolling(20).mean()
df['MA60'] = df['Close'].rolling(60).mean()

# MACD
df['EMA12'] = df['Close'].ewm(span=12).mean()
df['EMA26'] = df['Close'].ewm(span=26).mean()
df['MACD'] = df['EMA12'] - df['EMA26']
df['Signal'] = df['MACD'].ewm(span=9).mean()

# RSI
delta = df['Close'].diff()
gain = (delta.where(delta > 0, 0)).rolling(14).mean()
loss = (-delta.where(delta < 0, 0)).rolling(14).mean()
rs = gain / loss
df['RSI'] = 100 - (100 / (1 + rs))

# 绘图
fig, axes = plt.subplots(4, 1, figsize=(14, 16),
                         gridspec_kw={'height_ratios': [3, 1, 1, 1]})

# 1. 价格和均线
axes[0].plot(df.index, df['Close'], label='Price', alpha=0.7)
axes[0].plot(df.index, df['MA5'], label='MA5', alpha=0.5)
axes[0].plot(df.index, df['MA20'], label='MA20', alpha=0.5)
axes[0].plot(df.index, df['MA60'], label='MA60', alpha=0.5)
axes[0].set_title('AAPL Price and Moving Averages')
axes[0].legend()
axes[0].grid(True, alpha=0.3)

# 2. 成交量
colors = ['green' if df['Close'].iloc[i] >= df['Open'].iloc[i] else 'red'
          for i in range(len(df))]
axes[1].bar(df.index, df['Volume'], color=colors, alpha=0.5)
axes[1].set_title('Volume')
axes[1].grid(True, alpha=0.3)

# 3. MACD
axes[2].plot(df.index, df['MACD'], label='MACD', color='blue')
axes[2].plot(df.index, df['Signal'], label='Signal', color='orange')
axes[2].bar(df.index, df['MACD'] - df['Signal'], color='gray', alpha=0.3)
axes[2].set_title('MACD')
axes[2].legend()
axes[2].grid(True, alpha=0.3)

# 4. RSI
axes[3].plot(df.index, df['RSI'], label='RSI', color='purple')
axes[3].axhline(y=70, color='red', linestyle='--', alpha=0.5)
axes[3].axhline(y=30, color='green', linestyle='--', alpha=0.5)
axes[3].fill_between(df.index, 30, 70, alpha=0.1)
axes[3].set_title('RSI (14)')
axes[3].set_ylim(0, 100)
axes[3].legend()
axes[3].grid(True, alpha=0.3)

plt.tight_layout()
plt.show()
```
