# 1.2 NumPy 数值计算：向量、矩阵、广播

## NumPy 是什么

NumPy 是 Python 科学计算的基础库，提供了：

- 高效的多维数组（ndarray）
- 向量化运算
- 丰富的数学函数

量化交易中，大量的价格计算、收益率运算都依赖 NumPy。

## 创建数组

```python
import numpy as np

# 从列表创建
prices = np.array([100, 102, 98, 105, 103])
print(prices)        # [100 102  98 105 103]
print(type(prices))  # <class 'numpy.ndarray'>

# 创建特殊数组
zeros = np.zeros(5)           # [0. 0. 0. 0. 0.]
ones = np.ones(5)            # [1. 1. 1. 1. 1.]
range_arr = np.arange(0, 10, 2)  # [0 2 4 6 8]
linspace = np.linspace(0, 1, 5)  # [0.   0.25 0.5  0.75 1.  ]
```

## 数组的基本属性

```python
prices = np.array([100, 102, 98, 105, 103])

print(prices.ndim)     # 1（维度）
print(prices.shape)    # (5,)（形状）
print(prices.dtype)    # int64（数据类型）
print(prices.size)     # 5（元素个数）
```

## 索引和切片

```python
prices = np.array([100, 102, 98, 105, 103])

# 基本索引
print(prices[0])      # 100（第一个元素）
print(prices[-1])     # 103（最后一个元素）

# 切片
print(prices[1:3])    # [102 98]（索引1到2）
print(prices[:3])     # [100 102 98]（开头到索引2）
print(prices[2:])     # [98 105 103]（索引2到结尾）

# 布尔索引（重要！）
mask = prices > 100
print(mask)           # [False  True False  True  True]
print(prices[mask])   # [102 105 103]（只保留 >100 的）
```

## 向量化运算（核心！）

向量化是 NumPy 最重要的特性。

### 基本运算

```python
prices = np.array([100, 102, 98, 105, 103])

# 标量运算（作用于每个元素）
print(prices + 1)     # [101 103 99 106 104]
print(prices * 2)     # [200 204 196 210 206]
print(prices ** 2)    # [10000 10404  9604 11025 10609]
```

### 数组之间的运算

```python
a = np.array([1, 2, 3])
b = np.array([4, 5, 6])

print(a + b)   # [5 7 9]
print(a - b)   # [-3 -3 -3]
print(a * b)   # [4 10 18]
print(a / b)   # [0.25 0.4  0.5]
```

## 广播（Broadcasting）

广播是 NumPy 的强大特性，让不同形状的数组可以运算。

```python
# 场景：计算每只股票的收益率
prices_yesterday = np.array([100, 200, 50])   # 昨日收盘价
prices_today = np.array([105, 190, 52])       # 今日收盘价

# 收益率 = (今日 - 昨日) / 昨日
returns = (prices_today - prices_yesterday) / prices_yesterday
print(returns)  # [ 0.05  -0.05   0.04]
```

### 实战：计算股价数组的收益率序列

```python
# 股价数组
prices = np.array([100, 102, 101, 105, 103, 108, 107])

# 计算收益率：prices[1]/prices[0] - 1, prices[2]/prices[1] - 1, ...
returns = prices[1:] / prices[:-1] - 1
print(returns)  # [ 0.02 -0.0098  0.0396 -0.019  0.0485 -0.0093]
```

## 统计函数

```python
prices = np.array([100, 102, 98, 105, 103])

print(np.mean(prices))      # 101.6（均值）
print(np.std(prices))       # 2.653...（标准差）
print(np.var(prices))       # 7.04（方差）
print(np.min(prices))       # 98（最小值）
print(np.max(prices))       # 105（最大值）
print(np.sum(prices))       # 508（求和）

# 累计函数
print(np.cumsum(prices))    # [100 202 300 405 508]
print(np.cumprod(prices))   # [100 10200 999600 ...]
```

## 滚动窗口计算

NumPy 没有直接的滚动窗口函数，但可以用切片模拟：

```python
def rolling_mean(arr, window):
    """计算滚动均值"""
    return np.array([arr[i:i+window].mean() for i in range(len(arr) - window + 1)])

def rolling_std(arr, window):
    """计算滚动标准差"""
    return np.array([arr[i:i+window].std() for i in range(len(arr) - window + 1)])

prices = np.array([100, 102, 98, 105, 103, 108, 107, 110, 109, 112])
window = 3

print("滚动均值:", rolling_mean(prices, window))
print("滚动标准差:", rolling_std(prices, window))
```

> **注意**：对于滚动窗口计算，Pandas 的 `rolling()` 方法更高效，我们会在 Pandas 章节学习。

## 矩阵运算

```python
# 创建 2x3 矩阵
A = np.array([[1, 2, 3],
              [4, 5, 6]])

# 创建 3x2 矩阵
B = np.array([[1, 2],
              [3, 4],
              [5, 6]])

# 矩阵乘法
C = A @ B  # 或 np.dot(A, B)
print(C)
# [[22 28]
#  [49 64]]

# 转置
print(A.T)
# [[1 4]
#  [2 5]
#  [3 6]]
```

## 常用金融计算

### 复利计算

```python
def compound_return(initial, rate, periods):
    """计算复利终值"""
    return initial * (1 + rate) ** periods

# 10000元，年化10%，5年后
final = compound_return(10000, 0.10, 5)
print(f"终值: ¥{final:,.2f}")  # ¥16,105.10
```

### 对数收益率

```python
def log_returns(prices):
    """计算对数收益率"""
    return np.log(prices[1:] / prices[:-1])

prices = np.array([100, 102, 98, 105, 103])
log_ret = log_returns(prices)
print(f"对数收益率: {log_ret}")
print(f"总对数收益: {log_ret.sum():.4f}")  # 对数收益率可累加
```

prices = np.array([100, 103, 102, 105, 108, 107, 110, 112, 111, 115])

# 目标1：每日收益率
daily_returns = prices[1:] / prices[:-1] - 1
print(f"每日收益率: {daily_returns}")

three_day_returns = prices[3:] / prices[:-3] - 1
print(f"3日收益率: {three_day_returns}")

cumulative = prices[-1] / prices[0] - 1

log_ret = np.log(prices[1:] / prices[:-1])
print(f"对数收益累加: {log_ret.sum():.2%}")
