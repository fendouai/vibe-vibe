# 11.2 Alpaca API实盘交易

## Alpaca简介

Alpaca是一家美国持牌券商，提供免费的股票和加密货币交易API。对于量化交易者来说，Alpaca的优势在于：

- **免费使用**: 无平台费，佣金极低
- **REST API**: 简单易用
- **WebSocket**: 支持实时数据
- **纸交易**: 免费的模拟交易环境
- **Python SDK**: 官方SDK支持

```python
def place_market_order(symbol, qty, side):
    """
    下市价单

    Parameters:
    - symbol: 股票代码 (如 'AAPL')
    - qty: 数量
    - side: 'buy' 或 'sell'
    """
    order = trading_client.submit_order(
        symbol=symbol,
        qty=qty,
        side=side,
        type='market',
        time_in_force=TimeInForce.DAY

    print(f"   数量: {order.qty}")
    print(f"   方向: {order.side}")
    print(f"   状态: {order.status}")
```

```python
from alpaca.data import Bar, Quote, Trade
from datetime import datetime, timedelta

def get_historical_bars(symbol, start, end=None, timeframe='1Day', limit=100):
    """
    获取历史K线数据

    Parameters:
    - symbol: 股票代码
    - start: 开始时间
    - end: 结束时间
    - timeframe: '1Min', '5Min', '15Min', '1Hour', '1Day'
    - limit: 最大数据条数
    """
    from alpaca.data.timeframe import TimeFrame

    timeframe_map = {
        '1Min': TimeFrame.Minute,
        '5Min': TimeFrame.FiveMinute,
        '15Min': TimeFrame.FifteenMinute,
        '1Hour': TimeFrame.Hour,
        '1Day': TimeFrame.Day
    }

    bars = data_client.get_stock_bars(
        symbol_or_symbols=[symbol],
        start=start,
        end=end,
        timeframe=timeframe_map.get(timeframe, TimeFrame.Day),
        limit=limit
    ).df

    return bars

# 示例：获取苹果最近100天的日K
```

## 实时行情 (WebSocket)

```python
import asyncio
from alpaca.data import StockHistoricalDataClient
from alpaca.data.stream import StockDataStream

async def stream_stock_data(symbols, callback):
    """
    实时订阅股票数据

    Parameters:
    - symbols: 股票代码列表
    - callback: 数据回调函数
    """
    data_stream = StockDataStream(API_KEY, API_SECRET)

    # 订阅 Trades
    async def handle_trade(trade):
        callback('trade', trade)

    # 订阅 Quotes
    async def handle_quote(quote):
        callback('quote', quote)

    # 订阅 Bars
    async def handle_bar(bar):
        callback('bar', bar)

    # 注册处理器
    data_stream.subscribe_trades(handle_trade, *symbols)
    data_stream.subscribe_quotes(handle_quote, *symbols)
    data_stream.subscribe_bars(handle_bar, *symbols)

    print(f"📡 开始接收 {symbols} 的实时数据...")
    await data_stream.consume()

# 使用示例
def on_data(data_type, data):
    if data_type == 'trade':

    elif data_type == 'quote':
        print(f"报价: {data.symbol} 买${data.bid_price} 卖${data.ask_price}")
```

## 完整策略执行示例

```python
class AlpacaTradingBot:
    def __init__(self, api_key, api_secret, paper=True):
        self.trading_client = TradingClient(api_key, api_secret, paper=paper)
        self.data_client = StockHistoricalDataClient(api_key, api_secret)
        self.position = None
        self.order = None

    def get_account(self):
        return self.trading_client.get_account()

    def get_current_price(self, symbol):
        """获取当前市场价格"""
        bars = self.data_client.get_stock_latest_bar([symbol])
        return bars[symbol].close

    def check_signal(self, symbol):
        """检查交易信号 - 这里用一个简单示例"""
        # 获取最近20天数据
        from datetime import timedelta
        end = datetime.now()
        start = end - timedelta(days=30)

        bars = self.data_client.get_stock_bars(
            [symbol],
            start=start,
            end=end,
            limit=20
        ).df

        if len(bars) < 20:
            return None

        # 简单策略：收盘价上穿20日均线买入，下穿卖出
        bars['ma20'] = bars['close'].rolling(20).mean()
        latest = bars.iloc[-1]
        prev = bars.iloc[-2]

        if prev['close'] < prev['ma20'] and latest['close'] > latest['ma20']:
            return 'buy'
        elif prev['close'] > prev['ma20'] and latest['close'] < latest['ma20']:
            return 'sell'

        return None

    def execute_trade(self, symbol, signal):
        """执行交易"""
        if signal == 'buy' and not self.position:
            # 买入
            account = self.get_account()
            buying_power = float(account.buying_power)
            current_price = self.get_current_price(symbol)

            # 用一半的买入能力
            qty = int(buying_power * 0.5 / current_price / 100) * 100

            if qty > 0:
                order = self.trading_client.submit_order(
                    symbol=symbol,
                    qty=qty,
                    side=OrderSide.BUY,
                    type='market',
                    time_in_force=TimeInForce.DAY
                )
                self.position = {'symbol': symbol, 'qty': qty, 'entry_price': current_price}
                print(f"✅ 买入 {qty} 股 {symbol} @ ${current_price}")

        elif signal == 'sell' and self.position:
            # 卖出
            order = self.trading_client.submit_order(
                symbol=symbol,
                qty=self.position['qty'],
                side=OrderSide.SELL,
                type='market',
                time_in_force=TimeInForce.DAY
            )
            current_price = self.get_current_price(symbol)
            pnl = (current_price - self.position['entry_price']) * self.position['qty']
            print(f"✅ 卖出 {self.position['qty']} 股 {symbol} @ ${current_price} (PnL: ${pnl:.2f})")
            self.position = None

    def run(self, symbol, interval_seconds=60):
        """运行交易机器人"""
        import time

        print(f"🤖 开始运行交易机器人: {symbol}")
        print(f"   间隔: {interval_seconds}秒")

        while True:
            try:
                signal = self.check_signal(symbol)

                if signal:
                    print(f"\n📊 检测到信号: {signal}")
                    self.execute_trade(symbol, signal)
                else:
                    print(f".", end='', flush=True)

                time.sleep(interval_seconds)

            except KeyboardInterrupt:
                print("\n\n🛑 机器人已停止")
                break
            except Exception as e:
                print(f"\n❌ 错误: {e}")
                time.sleep(5)

# bot = AlpacaTradingBot('YOUR_API_KEY', 'YOUR_API_SECRET', paper=True)
```

## 注意事项

1. **纸交易优先**: 一定要先在纸交易环境测试
2. **订单执行时间**: 市价单不能保证精确价格
3. **隔夜持仓**: 注意保证金要求
4. **API限制**: 遵守速率限制
