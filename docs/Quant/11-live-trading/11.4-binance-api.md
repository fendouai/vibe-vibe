# 11.4 Binance API数字货币

## Binance简介

Binance（币安）是全球最大的加密货币交易所，提供丰富的API接口。相比传统券商，Binance API的特点：

- **交易品种丰富**: 数百种加密货币交易对
- **API功能完善**: 支持现货、合约、杠杆、理财等
- **文档完善**: 官方文档详细易懂
- **高流动性**: 主流币种流动性极好
- **手续费低**: maker返利机制
- **7x24小时交易**: 没有休市概念

## API类型

Binance提供多种API：

| API类型     | 说明          | 用途           |
| ----------- | ------------- | -------------- |
| REST API    | 请求/响应模式 | 查询数据、下单 |
| WebSocket   | 实时推送      | 行情、成交推送 |
| Futures API | 合约交易      | 期货、永续合约 |
| Margin API  | 杠杆交易      | 借币、杠杆交易 |

## 获取市场数据

### K线数据（Klines）

```python
def get_klines(symbol, interval='1h', limit=100):
    """
    获取K线数据

    Parameters:
    - symbol: 交易对 (如 'BTCUSDT')
    - interval: 时间间隔 ('1m', '5m', '15m', '1h', '4h', '1d')
    - limit: 数据条数 (max 1000)
    """
    klines = client.get_klines(
        symbol=symbol,
        interval=interval,
        limit=limit
    )

    # 转换为列表格式
    data = []
    for k in klines:
        data.append({
            'open_time': k[0],
            'open': float(k[1]),
            'high': float(k[2]),
            'low': float(k[3]),
            'close': float(k[4]),
            'volume': float(k[5]),
            'close_time': k[6],
            'quote_volume': float(k[7]),
            'trades': int(k[8])
        })

    return data

# 获取BTC最近100根1小时K线
btc_klines = get_klines('BTCUSDT', '1h', 100)
print(f"获取到 {len(btc_klines)} 根K线")
print(f"最新收盘价: ${btc_klines[-1]['close']:,.2f}")

# 转换为pandas DataFrame
import pandas as pd

df = pd.DataFrame(btc_klines)
df['open_time'] = pd.to_datetime(df['open_time'], unit='ms')
df.set_index('open_time', inplace=True)
print(df.tail())
```

def get_order_book(symbol, limit=20):

def get_account_balance():
    """获取账户余额"""
    account = client.get_account()

    balances = []
    for balance in account['balances']:
        free = float(balance['free'])
        locked = float(balance['locked'])
        if free > 0 or locked > 0:
            balances.append({
                'asset': balance['asset'],
                'free': free,
                'locked': locked,
                'total': free + locked
            })

    return balances

# 获取所有非零余额
balances = get_account_balance()

for b in balances[:10]:
    print(f"  {b['asset']}: {b['free']:.8f} (锁定: {b['locked']:.8f})")

## 订单管理

```python
def get_order(symbol, order_id):
    """查询指定订单"""
    try:
        order = client.get_order(symbol=symbol, orderId=order_id)
        return order
    except Exception as e:
        print(f"查询失败: {e}")
        return None

def get_open_orders(symbol=None):
    """获取待成交订单"""
    if symbol:
        orders = client.get_open_orders(symbol=symbol)
    else:
        orders = client.get_open_orders()

    print(f"待成交订单 ({len(orders)}个):")
    for o in orders:
        print(f"  {o['symbol']}: {o['side']} {o['origQty']} @ ${float(o['price']):,.2f}")

    return orders

def cancel_order(symbol, order_id):
    """取消指定订单"""
    try:
        result = client.cancel_order(symbol=symbol, orderId=order_id)
        print(f"✅ 订单 {order_id} 已取消")
        return result
    except Exception as e:
        print(f"❌ 取消失败: {e}")
        return None

def cancel_all_orders(symbol):
    """取消所有待成交订单"""
    result = client.cancel_all_open_orders(symbol=symbol)
    print(f"✅ 所有 {symbol} 待成交订单已取消")
    return result
```

## WebSocket实时数据

```python
from binance.websocket.websocket_client import BinanceWebsocketManager

def on_message(ws, message):
    """处理WebSocket消息"""
    print(message)

# 创建WebSocket管理器
ws_manager = BinanceWebsocketManager()
ws_manager.start()

# 订阅K线数据
def subscribe_kline(symbol, interval='1m'):
    """订阅K线数据"""
    @ws_manager.on('kline')
    def handle_kline(msg):
        if msg['e'] == 'kline':
            k = msg['k']
            print(f"{k['s']} {k['i']}: O={k['o']} H={k['h']} L={k['l']} C={k['c']}")

    ws_manager.subscribe_kline(symbol, interval)
    return handle_kline

# 订阅Trade数据
def subscribe_trade(symbol):
    """订阅成交数据"""
    @ws_manager.on('trade')
    def handle_trade(msg):
        print(f"成交: {msg['s']} {msg['p']} x {msg['q']} @ ${float(msg['p']):,.2f}")

    ws_manager.subscribe_trade(symbol)
    return handle_trade

# 示例
subscribe_kline('BTCUSDT', '1m')
subscribe_trade('BTCUSDT')

# 运行一段时间
import time
time.sleep(60)

# 停止
ws_manager.stop()
```

## 安全最佳实践

| 实践            | 说明                          |
| --------------- | ----------------------------- |
| 只用API Key权限 | 只开启需要的权限（读取/交易） |
