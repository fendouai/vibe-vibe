# 11.3 Interactive Brokers

## Interactive Brokers简介

Interactive Brokers（IBKR）是全球最大的在线券商之一，提供极其强大的交易API。IB的优势在于：

- **市场覆盖广**: 美股、港股、A股、期货、外汇、期权、加密货币等
- **强大的订单类型**: 支持几乎所有复杂订单
- **低佣金**: 业内最低佣金之一
- **TWS交易平台**: 功能强大的图形化交易平台
- **丰富的API**: 支持Python、Java、C++、C#等多种语言

## IB API架构

Interactive Brokers提供两种主要API：

| API类型        | 说明                    | 适用场景   |
| -------------- | ----------------------- | ---------- |
| TWS API        | 通过TWS或IB Gateway运行 | 桌面用户   |
| IB Gateway API | 轻量级网关              | 服务器部署 |

## 连接IB

```python
from ib_insync import *

# 连接IB Gateway或TWS
ib = IB()

# 连接到本地IB Gateway (默认端口 4001 纸交易, 4002 实盘)
# 如果使用TWS，默认端口 7496 纸交易, 7497 实盘
ib.connect('127.0.0.1', 4001, clientId=1)

# 检查连接状态
print(f"已连接: {ib.isConnected()}")
print(f"服务器时间: {ib.reqCurrentTime()}")

# 账户信息
account_summary = ib.accountSummary()
print("\n账户摘要:")
for item in account_summary:
    print(f"  {item.tag}: {item.value} {item.currency}")
```

def get_stock_bars(symbol, duration='1 D', bar_size='1 hour'):
    """获取股票K线数据"""
    contract = Stock(symbol, 'SMART', 'USD')

    bars = ib.reqHistoricalData(
        contract,

        barSizeSetting=bar_size,
        whatToShow='TRADES',
        useRTH=True,
        formatDate=1

# 转换为pandas DataFrame
import pandas as pd

def bars_to_df(bars):
    """将K线数据转换为DataFrame"""
    df = pd.DataFrame(bars)
    df['date'] = pd.to_datetime(df['date'])
    df.set_index('date', inplace=True)
    return df

# 获取苹果公司数据

def get_account_info():
    """获取账户信息"""
    summary = ib.accountSummary()

    info = {}
    for item in summary:
        if item.tag in ['NetLiquidation', 'Cash', 'BuyingPower', 'Equity']:
            info[item.tag] = float(item.value)

    return info

def place_market_order(symbol, quantity, action):
    """
    下市价单

    Parameters:
    - symbol: 股票代码
    - quantity: 数量
    - action: 'BUY' 或 'SELL'
    """
    contract = Stock(symbol, 'SMART', 'USD')
    order = MarketOrder(action, quantity)

    # 下单
    trade = ib.placeOrder(contract, order)

    # 等待订单状态更新
    ib.sleep(1)

    print(f"订单已提交:")
    print(f"  股票: {symbol}")
    print(f"  数量: {quantity}")
    print(f"  方向: {action}")
    print(f"  状态: {trade.orderStatus.status}")
    print(f"  订单ID: {trade.order.orderId}")

    return trade

# 买入100股苹果
trade = place_market_order('AAPL', 100, 'BUY')
```

def get_open_orders():
    """获取待成交订单"""
    orders = ib.openOrders()

    print("=== 待成交订单 ===")
    for order in orders:
        print(f"\n订单ID: {order.orderId}")
        print(f"  股票: {order.contract.symbol}")
        print(f"  数量: {order.totalQuantity}")
        print(f"  类型: {order.orderType}")
        print(f"  方向: {order.action}")
        print(f"  状态: {order.status}")

    return orders

def cancel_order(order_id):
    """取消指定订单"""
    for order in ib.openOrders():
        if order.orderId == order_id:
            ib.cancelOrder(order)
            print(f"订单 {order_id} 已取消")
            return True
    print(f"未找到订单 {order_id}")
    return False

def cancel_all_orders():
    """取消所有待成交订单"""
    ib.cancelAllOrders()
    print("所有待成交订单已取消")
```

## 错误处理

```python
from ib_insync import IBError

def safe_connect(host='127.0.0.1', port=4001, client_id=1, max_retries=3):
    """带重试的连接"""
    ib = IB()

    for attempt in range(max_retries):
        try:
            ib.connect(host, port, clientId=client_id)
            print(f"✅ 成功连接到 IB (尝试 {attempt + 1})")
            return ib
        except ConnectionRefusedError:
            print(f"❌ 连接被拒绝 (尝试 {attempt + 1}/{max_retries})")
            ib.sleep(5)
        except Exception as e:
            print(f"❌ 连接错误: {e}")
            ib.sleep(5)

    raise ConnectionError("无法连接到IB")

# 使用
try:
    ib = safe_connect()
    # 执行交易
finally:
    if ib.isConnected():
        ib.disconnect()
```
