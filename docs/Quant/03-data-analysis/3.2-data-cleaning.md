### 处理方法

```python
# 方法1：删除缺失值
df_clean = df.dropna()
print(f"删除前行数: {len(df)}, 删除后: {len(df_clean)}")

# 方法2：前向填充 (Forward Fill)
# 用前一个有效值填充
df_filled = df.fillna(method='ffill')
# 适用于：价格数据（用昨天的收盘价填充今天的停牌价）

# 方法3：后向填充 (Backward Fill)
# 用后一个有效值填充
df_filled = df.fillna(method='bfill')

# 方法4：插值填充
df_interpolated = df.interpolate(method='linear')
# 适用于：数值连续的情况

# 方法5：均值/中位数填充
df['close'] = df['close'].fillna(df['close'].mean())

# 方法6：只填充价格列（保留其他列的NaN用于标记）
df['close'] = df['close'].fillna(method='ffill')
df['volume'] = df['volume'].fillna(0)  # 停牌时成交量为0
```

### 停牌标记处理

```python
# 最佳实践：区分停牌和数据缺失
def process_stock_data(df):
    """处理股票数据，区分停牌和数据问题"""
    df = df.copy()

    # 创建停牌标记
    df['is_pause'] = df['volume'] == 0

    # 价格列用前值填充
    price_cols = ['open', 'high', 'low', 'close']
    for col in price_cols:
        if col in df.columns:
            df[col] = df[col].fillna(method='ffill')

    # 成交量用0填充
    if 'volume' in df.columns:
        df['volume'] = df['volume'].fillna(0)

    return df

df = process_stock_data(df)
```

## 异常值处理

### 检测异常值

```python
import scipy.stats as stats

def detect_outliers_zscore(series, threshold=3):
    """Z-score方法检测异常值"""
    z_scores = np.abs(stats.zscore(series.dropna()))
    return z_scores > threshold

def detect_outliers_iqr(series, multiplier=1.5):
    """IQR方法检测异常值"""
    Q1 = series.quantile(0.25)
    Q3 = series.quantile(0.75)
    IQR = Q3 - Q1
    lower = Q1 - multiplier * IQR
    upper = Q3 + multiplier * IQR
    return (series < lower) | (series > upper)

# 检测收益率异常值
returns = df['close'].pct_change().dropna()

outliers_zscore = detect_outliers_zscore(returns)
outliers_iqr = detect_outliers_iqr(returns)

print(f"Z-score异常值数量: {outliers_zscore.sum()}")
print(f"IQR异常值数量: {outliers_iqr.sum()}")

# 可视化
fig, axes = plt.subplots(1, 2, figsize=(14, 4))

# 箱线图
axes[0].boxplot(returns.dropna())
axes[0].set_title('Return Box Plot')

# 直方图
axes[1].hist(returns.dropna(), bins=50, edgecolor='black')
axes[1].set_title('Return Distribution')
axes[1].axvline(returns.mean() + 3*returns.std(), color='r', linestyle='--', label='3 Std')
axes[1].axvline(returns.mean() - 3*returns.std(), color='r', linestyle='--')
axes[1].legend()

plt.tight_layout()
plt.show()
```

### 处理异常值

```python
# 方法1：删除异常值
returns_cleaned = returns[~outliers_iqr]

# 方法2：替换为边界值
def cap_outliers(series, lower_percentile=0.01, upper_percentile=0.99):
    """将异常值替换为百分位边界值"""
    lower = series.quantile(lower_percentile)
    upper = series.quantile(upper_percentile)
    return series.clip(lower=lower, upper=upper)

returns_capped = cap_outliers(returns)

# 方法3：Winsorize（缩尾处理）
from scipy.stats import mstats
returns_winsorized = mstats.winsorize(returns.dropna(), limits=[0.01, 0.01])
```

## 数据验证

### 基本验证

```python
def validate_stock_data(df):
    """验证股票数据质量"""
    issues = []

    # 检查空数据
    if df.isnull().all().all():
        issues.append("所有数据都是缺失值")

    # 检查价格逻辑
    if (df['high'] < df['low']).any():
        issues.append("存在最高价<最低价的异常")

    if (df['high'] < df['close']).any():
        issues.append("存在最高价<收盘价的异常")

    if (df['low'] > df['open']).any():
        issues.append("存在最低价>开盘价的异常")

    # 检查成交量
    if (df['volume'] < 0).any():
        issues.append("存在负成交量")

    # 检查价格变化
    price_change = df['close'].pct_change()
    if (abs(price_change) > 0.5).any():
        issues.append("存在日涨跌超过50%的日期")

    return issues

issues = validate_stock_data(df)
if issues:
    print("数据问题:")
    for issue in issues:
        print(f"  - {issue}")
else:
    print("数据验证通过")
```

### 数据质量报告

```python
def data_quality_report(df):
    """生成数据质量报告"""
    report = {
        'total_rows': len(df),
        'date_range': f"{df.index.min()} to {df.index.max()}",
        'columns': list(df.columns),
        'missing_by_column': df.isnull().sum().to_dict(),
        'missing_percentage': (df.isnull().sum() / len(df) * 100).round(2).to_dict(),
        'duplicates': df.duplicated().sum()
    }
    return report

report = data_quality_report(df)
print("数据质量报告:")
for key, value in report.items():
    print(f"  {key}: {value}")
```
