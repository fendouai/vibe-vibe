# 3.3 数据重采样：日线、周线、分钟线转换

## 重采样概念

```python
# 重采样 (Resampling) = 改变数据的频率
# 上采样：低频 → 高频（添加更多数据点）
# 下采样：高频 → 低频（聚合数据点）

resampling_types = {
    'upsampling': '低频转高频',
    'downsampling': '高频转低频',
    'aggregation': '聚合（如分钟→日线）',
    'interpolation': '插值（如填充缺失的分钟数据）'
}
```

## Pandas resample 基础

```python
import pandas as pd
import numpy as np
import yfinance as yf

# 获取分钟数据示例
df = yf.download('AAPL', start='2023-01-03', end='2023-01-04', interval='5m')
df.columns = df.columns.get_level_values(0)
print(df.head(10))
#                              Open    High     Low   Close  Volume
# Datetime                                                ed
# 2023-01-03 09:30:00-05:00  126.1  126.25  126.00  126.25  102400
```

## 下采样（高频→低频）

### 分钟 → 日线

```python
# 分钟数据转为日线
daily = df.resample('D').agg({
    'open': 'first',      # 日开盘 = 当日第一笔
    'high': 'max',        # 日最高 = 当日最高
    'low': 'min',         # 日最低 = 当日最低
    'close': 'last',      # 日收盘 = 当日最后一笔
    'volume': 'sum'       # 日成交量 = 当日总和
})
print(daily.dropna())
```

### 日线 → 周线/月线

```python
# 获取日线数据
df = yf.download('AAPL', start='2022-01-01', end='2023-12-31')
df.columns = df.columns.get_level_values(0)

# 周线（以周日为周期结束）
weekly = df.resample('W').agg({
    'open': 'first',
    'high': 'max',
    'low': 'min',
    'close': 'last',
    'volume': 'sum'
})

# 月线（以月末为周期结束）
monthly = df.resample('M').agg({
    'open': 'first',
    'high': 'max',
    'low': 'min',
    'close': 'last',
    'volume': 'sum'
})

# 季度线
quarterly = df.resample('Q').agg({
    'open': 'first',
    'high': 'max',
    'low': 'min',
    'close': 'last',
    'volume': 'sum'
})
```

### OHLC 聚合规则

```python
# 常用OHLC聚合规则
ohlc_rules = {
    'first': '周期第一笔（开盘）',
    'last': '周期最后一笔（收盘）',
    'max': '周期最大值（最高）',
    'min': '周期最小值（最低）',
    'sum': '周期总和（成交量）',
    'mean': '周期平均值',
    'ohlc': '返回(OHLC)元组'
}

# 自定义OHLC聚合
def custom_ohlc(series):
    return pd.Series({
        'open': series.iloc[0],
        'high': series.max(),
        'low': series.min(),
        'close': series.iloc[-1]
    })

# 应用自定义聚合
result = df.resample('M').apply(custom_ohlc)
```

## 上采样（低频→高频）

### 日线 → 分钟线

```python
# 日线数据
df_daily = yf.download('AAPL', start='2023-01-03', end='2023-01-10')
df_daily.columns = df_daily.columns.get_level_values(0)
print(df_daily)
#             Open    High     Low   Close  Volume
# Date
# 2023-01-03  126.0  126.90  125.41  126.36  77.8M

# 上采样到小时（需要插值）
# 注意：上采样需要填充缺失值

# 方法1：前向填充
hourly_ffill = df_daily.resample('H').ffill()

# 方法2：插值（更准确）
hourly_interp = df_daily.resample('H').interpolate()
```

## 滚动窗口计算

### rolling 基础

```python
# 计算N日移动平均
df['ma5'] = df['close'].rolling(5).mean()
df['ma20'] = df['close'].rolling(20).mean()
df['ma60'] = df['close'].rolling(60).mean()

# 滚动标准差（波动率）
df['vol20'] = df['close'].rolling(20).std()

# 滚动最大值/最小值
df['high20'] = df['close'].rolling(20).max()
df['low20'] = df['close'].rolling(20).min()
```

### 滚动窗口统计

```python
# 滚动收益率
df['return_5d'] = df['close'].pct_change(5)

# 滚动夏普比率
trading_days = 252
df['sharpe_20'] = (df['return_5d'].rolling(20).mean() * 252 /
                   (df['return_5d'].rolling(20).std() * np.sqrt(252)))

# 滚动最大回撤
def rolling_max_drawdown(returns):
    """计算滚动最大回撤"""
    wealth = (1 + returns).cumprod()
    running_max = wealth.cummax()
    drawdown = (wealth - running_max) / running_max
    return drawdown.min()

df['dd_20'] = df['return_5d'].rolling(20).apply(rolling_max_drawdown)
```

### 滚动窗口与重采样结合

```python
# 计算每月的20日均线
# 先计算20日均线，再按月重采样取月末值
df['ma20'] = df['close'].rolling(20).mean()
monthly_ma = df['ma20'].resample('M').last()

# 计算每周的交易信号（基于日线数据）
df['ma_cross'] = 0
df.loc[df['ma5'] > df['ma20'], 'ma_cross'] = 1
df.loc[df['ma5'] < df['ma20'], 'ma_cross'] = -1

# 按周汇总信号
weekly_signals = df['ma_cross'].resample('W').last()
```

## 实战：完整的数据转换流程

```python
import pandas as pd
import numpy as np
import yfinance as yf

class DataResampler:
    """数据重采样工具类"""

    def __init__(self, df):
        self.df = df.copy()

    def to_daily(self):
        """转为日线"""
        daily = self.df.resample('D').agg({
            'open': 'first',
            'high': 'max',
            'low': 'min',
            'close': 'last',
            'volume': 'sum'
        }).dropna()
        return daily

    def to_weekly(self):
        """转为周线"""
        weekly = self.df.resample('W').agg({
            'open': 'first',
            'high': 'max',
            'low': 'min',
            'close': 'last',
            'volume': 'sum'
        }).dropna()
        return weekly

    def to_monthly(self):
        """转为月线"""
        monthly = self.df.resample('M').agg({
            'open': 'first',
            'high': 'max',
            'low': 'min',
            'close': 'last',
            'volume': 'sum'
        }).dropna()
        return monthly

    def add_indicators(self, df):
        """添加技术指标"""
        df = df.copy()
        df['returns'] = df['close'].pct_change()
        df['ma5'] = df['close'].rolling(5).mean()
        df['ma20'] = df['close'].rolling(20).mean()
        df['vol20'] = df['returns'].rolling(20).std() * np.sqrt(252)
        return df

# 使用示例
data = yf.download('AAPL', start='2022-01-01', end='2023-12-31', interval='60m')
data.columns = data.columns.get_level_values(0)

resampler = DataResampler(data)
daily = resampler.to_daily()
daily_with_indicators = resampler.add_indicators(daily)

print(f"60分钟数据: {len(data)} 行")
print(f"日线数据: {len(daily)} 行")
print(f"技术指标:\n{daily_with_indicators.tail()}")
```
