## yfinance 获取财务数据

### 分红数据

```python
import yfinance as yf

ticker = yf.Ticker('AAPL')

# 获取分红历史
dividends = ticker.dividends
print("分红历史:")
print(dividends.tail(10))

# 分红数据重采样（年化）
annual_dividends = dividends.resample('Y').sum()
print(f"\n年度分红:\n{annual_dividends.tail(5)}")

# 计算分红收益率
stock_price = ticker.info.get('currentPrice', 0)
if dividends is not None and len(dividends) > 0:
    latest_annual_div = annual_dividends.iloc[-1]
    dividend_yield = latest_annual_div / stock_price
    print(f"\n当前股息率: {dividend_yield:.2%}")
```

## 调整因子

### 股票拆分调整

```python
import yfinance as yf
import pandas as pd

def adjust_for_splits(df, splits):
    """根据拆股历史调整价格数据"""
    df_adjusted = df.copy()

    for date, ratio in splits.items():
        if ratio != 1 and date in df_adjusted.index:
            # 拆股前的价格需要调整
            mask = df_adjusted.index < date
            for col in ['open', 'high', 'low', 'close', 'adj close']:
                if col in df_adjusted.columns:
                    df_adjusted.loc[mask, col] *= ratio

    return df_adjusted

ticker = yf.Ticker('AAPL')
splits = ticker.splits

df = yf.download('AAPL', start='2020-01-01', end='2023-12-31')
df.columns = df.columns.get_level_values(0)

# 使用调整后的价格
df['adj_close'] = df['close']  # yfinance已经提供调整后价格
```

## 财务数据在策略中的应用

### 价值因子

```python
import yfinance as yf

def get_value_metrics(symbol):
    """获取价值因子"""
    ticker = yf.Ticker(symbol)
    info = ticker.info

    metrics = {
        'pe': info.get('trailingPE', None),
        'pb': info.get('priceToBook', None),
        'ps': info.get('priceToSalesTrailing12Months', None),
        'pcf': info.get('operatingCashflow', None) / info.get('marketCap', 1),  # PCF ratio
    }
    return metrics

# 计算多只股票的价值因子
symbols = ['AAPL', 'GOOGL', 'MSFT', 'AMZN']
value_df = pd.DataFrame([get_value_metrics(s) for s in symbols], index=symbols)
print(value_df)
```

### 盈利能力因子

```python
def get_profitability_metrics(symbol):
    """获取盈利能力因子"""
    ticker = yf.Ticker(symbol)
    income = ticker.income_stmt
    balance = ticker.balance_sheet

    if len(income) == 0 or len(balance) == 0:
        return {}

    # ROE = 净利润 / 股东权益
    net_income = income.loc['Net Income'].iloc[-1] if 'Net Income' in income.index else 0
    shareholders_equity = balance.loc['Stockholders Equity'].iloc[-1] if 'Stockholders Equity' in balance.index else 0
    roe = net_income / shareholders_equity if shareholders_equity else 0

    # ROA = 净利润 / 总资产
    total_assets = balance.loc['Total Assets'].iloc[-1] if 'Total Assets' in balance.index else 0
    roa = net_income / total_assets if total_assets else 0

    return {
        'ROE': roe,
        'ROA': roa,
        'Net_Margin': net_income / income.loc['Revenue'].iloc[-1] if 'Revenue' in income.index and income.loc['Revenue'].iloc[-1] else 0
    }
```

### 高股息策略

```python
def screen_dividend_stocks(symbols, min_yield=0.03):
    """筛选高股息股票"""
    results = []

    for symbol in symbols:
        ticker = yf.Ticker(symbol)
        info = ticker.info
        div_yield = info.get('dividendYield', 0) or 0
        price = info.get('currentPrice', 0)

        if div_yield >= min_yield and price > 0:
            dividends = ticker.dividends
            annual_div = dividends.resample('Y').sum().iloc[-1] if len(dividends) > 0 else 0

            results.append({
                'symbol': symbol,
                'price': price,
                'dividend_yield': div_yield,
                'annual_dividend': annual_div
            })

    return pd.DataFrame(results).sort_values('dividend_yield', ascending=False)
```
