## yfinance - 美股数据

### 基本使用

```python
import yfinance as yf

# 获取单只股票
aapl = yf.download('AAPL', start='2023-01-01', end='2023-12-31')
print(aapl.head())

# 获取多只股票
stocks = yf.download(['AAPL', 'GOOGL', 'MSFT'], start='2023-01-01', end='2023-12-31')
print(stocks['Close'].head())  # 收盘价

# 获取单个字段
aapl_close = yf.download('AAPL', start='2023-01-01', end='2023-03-31')['Close']
print(aapl_close.head())
```

### 获取信息数据

```python
# 获取股票信息
ticker = yf.Ticker('AAPL')
info = ticker.info

print(f"公司名: {info.get('longName')}")
print(f"当前价格: {info.get('currentPrice')}")
print(f"市值: {info.get('marketCap')}")
print(f"PE比率: {info.get('trailingPE')}")
print(f"52周高: {info.get('fiftyTwoWeekHigh')}")
print(f"52周低: {info.get('fiftyTwoWeekLow')}")
```

### 获取其他数据

```python
ticker = yf.Ticker('AAPL')

# 历史持仓
# 注意：仅对持仓过的股东有效
# historical = ticker.history(start='2023-01-01', end='2023-12-31')

# 分红历史
dividends = ticker.dividends
print(dividends.tail())

# 股票拆分历史
splits = ticker.splits
print(splits.tail())

# 财务报表

balance_sheet = ticker.balance_sheet  # 资产负债表
cashflow = ticker.cashflow  # 现金流量表
```

### 盘中数据

```python
# 盘中实时数据（当日）

print(day_data.head())
```

## Binance API - 数字货币

### 安装与基本使用

```bash
pip install python-binance
```

```python
from binance import Client
import pandas as pd

# 无需API密钥即可获取历史数据
client = Client()

# 获取K线数据
klines = client.get_klines(symbol='BTCUSDT', interval=Client.KLINE_INTERVAL_1DAY,
                            start_str='2023-01-01', end_str='2023-12-31')

# 转换为DataFrame
df = pd.DataFrame(klines, columns=['open_time', 'open', 'high', 'low', 'close',
                                     'volume', 'close_time', 'quote_volume',
                                     'trades', 'taker_buy_base', 'taker_buy_quote', 'ignore'])

# 转换数据类型
df['open'] = df['open'].astype(float)
df['high'] = df['high'].astype(float)
df['low'] = df['low'].astype(float)
df['close'] = df['close'].astype(float)
df['volume'] = df['volume'].astype(float)

print(df.head())
```

## 数据存储

### 保存为CSV

```python
import yfinance as yf
import pandas as pd

# 获取数据
df = yf.download('AAPL', start='2023-01-01', end='2023-12-31')

# 保存为CSV
df.to_csv('data/aapl_2023.csv')

# 读取CSV
df = pd.read_csv('data/aapl_2023.csv', index_col='Date', parse_dates=True)
```

### 保存为Parquet（推荐）

```python
# Parquet格式更高效，适合大数据
df.to_parquet('data/aapl_2023.parquet')

# 读取
df = pd.read_parquet('data/aapl_2023.parquet')
```

### 数据库存储

```python
import sqlite3

# 创建SQLite数据库
conn = sqlite3.connect('data/stocks.db')

# 保存到数据库
df.to_sql('aapl', conn, if_exists='replace')

# 从数据库读取
df = pd.read_sql('SELECT * FROM aapl', conn, index_col='Date')

conn.close()
```

## 数据获取工具类

```python
import yfinance as yf
import pandas as pd
from typing import List, Optional
from datetime import datetime

class StockDataFetcher:
    """股票数据获取工具"""

    def __init__(self):
        self.cache = {}

    def get_stock_data(self, symbol: str, start: str, end: str,
                       interval: str = '1d') -> pd.DataFrame:
        """
        获取股票数据

        Args:
            symbol: 股票代码（如 'AAPL', '000858.SS'）
            start: 开始日期 ('YYYY-MM-DD')
            end: 结束日期 ('YYYY-MM-DD')
            interval: K线周期 ('1d', '1h', '5m', etc.)
        """
        cache_key = f"{symbol}_{start}_{end}_{interval}"
        if cache_key in self.cache:
            return self.cache[cache_key]

        df = yf.download(symbol, start=start, end=end, interval=interval)

        if len(df) == 0:
            print(f"警告: {symbol} 无数据")
            return pd.DataFrame()

        # 扁平化列名
        if isinstance(df.columns, pd.MultiIndex):
            df.columns = df.columns.get_level_values(0)

        self.cache[cache_key] = df
        return df

    def get_multiple_stocks(self, symbols: List[str], start: str, end: str) -> pd.DataFrame:
        """获取多只股票的收盘价"""
        data = yf.download(symbols, start=start, end=end)['Close']
        return data

    def get_returns(self, symbol: str, start: str, end: str) -> pd.Series:
        """获取收益率序列"""
        df = self.get_stock_data(symbol, start, end)
        returns = df['Close'].pct_change().dropna()
        return returns

# 使用示例
fetcher = StockDataFetcher()
aapl_data = fetcher.get_stock_data('AAPL', '2023-01-01', '2023-12-31')
aapl_returns = fetcher.get_returns('AAPL', '2023-01-01', '2023-12-31')
```
