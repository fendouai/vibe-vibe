# 7.2 协整检验：找到合适的配对

## 协整的概念

```python
"""
协整 (Cointegration)：

两个时间序列，尽管各自可能是不平稳的，
但它们的线性组合可能是平稳的。

例如：
- A股票价格：随机游走 (不平稳)
- B股票价格：随机游走 (不平稳)
- 但 A - k×B 可能是平稳的

这意味着两个股票存在长期的"均衡关系"
"""

import pandas as pd
import numpy as np
import yfinance as yf
from statsmodels.tsa.stattools import coint, adfuller
```

## Engle-Granger 两步法

```python
def cointegration_test(series1, series2, name1='Series1', name2='Series2'):
    """
    协整检验（Engle-Granger两步法）

    H0: 两个序列不存在协整关系
    H1: 两个序列存在协整关系

    Returns:
        is_cointegrated: 是否协整
        hedge_ratio: 对冲比率
    """
    from statsmodels.regression.linear_model import OLS

    # 第一步：用OLS回归得到对冲比率
    model = OLS(series1, series2).fit()
    hedge_ratio = model.params[0]

    # 第二步：检验残差的平稳性
    spread = series1 - hedge_ratio * series2
    result = adfuller(spread, maxlag=1)

    is_cointegrated = result[1] < 0.05

    print(f"协整检验 {name1} vs {name2}:")
    print(f"  对冲比率: {hedge_ratio:.4f}")
    print(f"  ADF统计量: {result[0]:.4f}")
    print(f"  p值: {result[1]:.4f}")
    print(f"  结论: {'存在协整关系' if is_cointegrated else '不存在协整关系'}")

    return is_cointegrated, hedge_ratio

# 示例
stock1 = yf.download('AAPL', start='2022-01-01', end='2023-12-31')['Close']
stock2 = yf.download('MSFT', start='2022-01-01', end='2023-12-31')['Close']
stock1.columns = stock1.columns.get_level_values(0)
stock2.columns = stock2.columns.get_level_values(0)

is_coint, ratio = cointegration_test(stock1, stock2, 'AAPL', 'MSFT')
```

## 自动筛选配对

```python
def find_cointegrated_pairs(stocks_data, significance=0.05):
    """
    在多只股票中自动筛选协整的配对
    """
    n = len(stocks_data.columns)
    pairs = []

    for i in range(n):
        for j in range(i+1, n):
            stock1 = stocks_data.iloc[:, i]
            stock2 = stocks_data.iloc[:, j]

            # Engle-Granger检验
            score, pvalue, _ = coint(stock1, stock2)

            if pvalue < significance:
                pairs.append({
                    'stock1': stocks_data.columns[i],
                    'stock2': stocks_data.columns[j],
                    'pvalue': pvalue,
                    'score': score
                })

    return pd.DataFrame(pairs).sort_values('pvalue')
```
