# 7.4 Python实现：配对交易策略回测

## 完整回测框架

```python
class PairsTradingBacktester:
    """配对交易策略回测"""

    def __init__(self, stock1, stock2, name1='Stock1', name2='Stock2'):
        self.stock1 = stock1
        self.stock2 = stock2
        self.name1 = name1
        self.name2 = name2

        self.hedge_ratio = None
        self.spread = None
        self.zscore = None
        self.signals = None
        self.returns = None

    def calculate_spread(self, window=20):
        """计算价差"""
        model = OLS(self.stock1, self.stock2).fit()
        self.hedge_ratio = model.params[0]
        self.spread = self.stock1 - self.hedge_ratio * self.stock2
        self.zscore = (self.spread - self.spread.rolling(window).mean()) / self.spread.rolling(window).std()
        return self

    def generate_signals(self, entry=2, exit_threshold=0.5, window=20):
        """生成交易信号"""
        if self.zscore is None:
            self.calculate_spread(window)

        signals = pd.Series(0, index=self.stock1.index)

        # 做空价差
        signals[self.zscore > entry] = -1

        # 做价差
        signals[self.zscore < -entry] = 1

        # 平仓
        signals[self.zscore.abs() < exit_threshold] = 0

        self.signals = signals.shift(1).fillna(0)
        return self

    def run_backtest(self):
        """运行回测"""
        # 计算收益率
        ret1 = self.stock1.pct_change()
        ret2 = self.stock2.pct_change()

        # 策略收益 = signal * (ret1 - hedge_ratio * ret2)
        # 即：当signal=1时，买stock1卖stock2；当signal=-1时，反过来
        strategy_returns = self.signals * (ret1 - self.hedge_ratio * ret2)

        self.returns = strategy_returns.dropna()
        return self

    def performance(self):
        """计算绩效"""
        if self.returns is None:
            return

        total = (1 + self.returns).prod() - 1
        annual = self.returns.mean() * 252
        vol = self.returns.std() * np.sqrt(252)
        sharpe = annual / vol if vol > 0 else 0

        # 最大回撤
        cumret = (1 + self.returns).cumprod()
        maxdd = (cumret / cumret.cummax()).min() - 1

        print(f"\n{'='*50}")
        print(f"配对交易回测结果: {self.name1} vs {self.name2}")
        print(f"{'='*50}")
        print(f"对冲比率: {self.hedge_ratio:.4f}")
        print(f"总收益: {total:.2%}")
        print(f"年化收益: {annual:.2%}")
        print(f"夏普比率: {sharpe:.2f}")
        print(f"最大回撤: {maxdd:.2%}")

        return {
            'total_return': total,
            'annual_return': annual,
            'sharpe': sharpe,
            'max_drawdown': maxdd
        }

# 使用示例
stock1 = yf.download('AAPL', start='2022-01-01', end='2023-12-31')['Close']
stock2 = yf.download('MSFT', start='2022-01-01', end='2023-12-31')['Close']
stock1.columns = stock1.columns.get_level_values(0)
stock2.columns = stock2.columns.get_level_values(0)

backtester = PairsTradingBacktester(stock1, stock2, 'AAPL', 'MSFT')
backtester.generate_signals(entry=2, exit_threshold=0.5).run_backtest().performance()
```
