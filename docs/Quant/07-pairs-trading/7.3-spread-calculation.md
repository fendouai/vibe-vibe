# 7.3 价差计算：对冲比率、Z-Score

## 价差计算

```python
import pandas as pd
import numpy as np
import yfinance as yf
from statsmodels.regression.linear_model import OLS

def calculate_spread(stock1, stock2, hedge_ratio=None):
    """
    计算配对价差

    Args:
        stock1: 股票1价格
        stock2: 股票2价格
        hedge_ratio: 对冲比率（可选，如果不提供则计算）
    """
    if hedge_ratio is None:
        # 计算最优对冲比率
        model = OLS(stock1, stock2).fit()
        hedge_ratio = model.params[0]

    # 价差 = stock1 - hedge_ratio * stock2
    spread = stock1 - hedge_ratio * stock2

    return spread, hedge_ratio

def calculate_zscore(spread, window=20):
    """
    计算价差的Z-Score
    """
    mean = spread.rolling(window).mean()
    std = spread.rolling(window).std()
    zscore = (spread - mean) / std
    return zscore

# 示例
stock1 = yf.download('AAPL', start='2022-01-01', end='2023-12-31')['Close']
stock2 = yf.download('MSFT', start='2022-01-01', end='2023-12-31')['Close']
stock1.columns = stock1.columns.get_level_values(0)
stock2.columns = stock2.columns.get_level_values(0)

spread, ratio = calculate_spread(stock1, stock2)
zscore = calculate_zscore(spread, window=20)

print(f"对冲比率: {ratio:.4f}")
print(f"价差均值: {spread.mean():.4f}")
print(f"价差标准差: {spread.std():.4f}")
```

## 配对交易信号

```python
def pairs_trading_signals(stock1, stock2, entry_threshold=2, exit_threshold=0.5, window=20):
    """
    配对交易信号

    规则：
    - Z-score > entry_threshold → 做空价差（卖stock1，买stock2）
    - Z-score < -entry_threshold → 做价差（买stock1，卖stock2）
    - |Z-score| < exit_threshold → 平仓
    """
    spread, ratio = calculate_spread(stock1, stock2)
    zscore = calculate_zscore(spread, window)

    signals = pd.Series(0, index=stock1.index)

    # 做空价差：Z-score高，价格偏高
    signals[zscore > entry_threshold] = -1

    # 做价差：Z-score低，价格偏低
    signals[zscore < -entry_threshold] = 1

    # 平仓
    signals[(zscore.abs() < exit_threshold)] = 0

    return signals.shift(1).fillna(0), zscore

# 测试
signals, zscore = pairs_trading_signals(stock1, stock2)
print(f"做多次数: {(signals == 1).sum()}")
print(f"做空次数: {(signals == -1).sum()}")
```
