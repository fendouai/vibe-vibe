# 5.4 Python实现：动量策略回测

## 完整回测框架

```python
import numpy as np
import pandas as pd
import yfinance as yf
import matplotlib.pyplot as plt

class MomentumBacktester:
    """动量策略回测框架"""

    def __init__(self, tickers, start, end, lookback=20, holding=10,
                 strategy='tsmom', n_long=2, n_short=2):
        self.tickers = tickers
        self.start = start
        self.end = end
        self.lookback = lookback
        self.holding = holding
        self.strategy = strategy
        self.n_long = n_long
        self.n_short = n_short

        self.data = None
        self.signals = None
        self.returns = None

    def load_data(self):
        """加载数据"""
        self.data = yf.download(self.tickers, start=self.start,
                                  end=self.end)['Close']
        self.data.columns = self.data.columns.get_level_values(0)
        print(f"加载了 {len(self.data)} 天的数据")
        return self

    def generate_signals(self):
        """生成交易信号"""
        if self.strategy == 'tsmom':
            self._tsmom_signals()
        elif self.strategy == 'csmom':
            self._csmom_signals()
        return self

    def _tsmom_signals(self):
        """时间序列动量信号"""
        momentum = self.data.pct_change(self.lookback)
        self.signals = np.where(momentum > 0, 1, -1)
        self.signals = pd.DataFrame(self.signals, index=self.data.index,
                                      columns=self.data.columns)
        # 持有期
        self.signals = self.signals.resample(f'{self.holding}d').last()
        self.signals = self.signals.reindex(self.data.index).ffill().shift(1).fillna(0)

    def _csmom_signals(self):
        """横截面动量信号"""
        momentum = self.data.pct_change(self.lookback)
        self.signals = pd.DataFrame(0, index=self.data.index, columns=self.data.columns)

        for date in momentum.index[self.lookback::self.holding]:
            mom = momentum.loc[:date].iloc[-1]
            ranked = mom.sort_values(ascending=False)

            for i, ticker in enumerate(ranked.index):
                if i < self.n_long:
                    self.signals.loc[date, ticker] = 1
                elif i >= len(ranked) - self.n_short:
                    self.signals.loc[date, ticker] = -1

        self.signals = self.signals.ffill().shift(1).fillna(0)

    def run_backtest(self):
        """运行回测"""
        returns = self.data.pct_change()

        # 策略收益
        strategy_returns = (self.signals * returns).mean(axis=1)

        # 去除调仓日的收益（避免噪声）
        self.returns = strategy_returns

        return self

    def performance(self):
        """计算绩效指标"""
        if self.returns is None:
            print("请先运行回测")
            return

        total = (1 + self.returns).prod() - 1
        annual = self.returns.mean() * 252
        vol = self.returns.std() * np.sqrt(252)
        sharpe = annual / vol if vol > 0 else 0

        # 最大回撤
        cumret = (1 + self.returns).cumprod()
        maxdd = (cumret / cumret.cummax()).min() - 1

        # 胜率
        win_rate = (self.returns > 0).mean()

        print(f"\n{'='*50}")
        print(f"动量策略回测结果 ({self.strategy.upper()})")
        print(f"{'='*50}")
        print(f"回测期间: {self.start} - {self.end}")
        print(f"回看期: {self.lookback}天, 持有期: {self.holding}天")
        print(f"总收益: {total:.2%}")
        print(f"年化收益: {annual:.2%}")
        print(f"年化波动率: {vol:.2%}")
        print(f"夏普比率: {sharpe:.2f}")
        print(f"最大回撤: {maxdd:.2%}")
        print(f"胜率: {win_rate:.2%}")

        return {
            'total_return': total,
            'annual_return': annual,
            'volatility': vol,
            'sharpe': sharpe,
            'max_drawdown': maxdd,
            'win_rate': win_rate
        }

    def plot(self):
        """可视化"""
        fig, axes = plt.subplots(2, 1, figsize=(12, 8))

        # 累计收益
        cumret = (1 + self.returns).cumprod()
        cumret_bh = (1 + self.data.pct_change().mean(axis=1)).cumprod()

        axes[0].plot(cumret.index, cumret, label='Strategy')
        axes[0].plot(cumret_bh.index, cumret_bh, label='Buy & Hold', alpha=0.7)
        axes[0].set_title('Cumulative Returns')
        axes[0].legend()
        axes[0].grid(True, alpha=0.3)

        # 滚动夏普
        rolling_sharpe = self.returns.rolling(60).mean() / self.returns.rolling(60).std() * np.sqrt(252)
        axes[1].plot(rolling_sharpe.index, rolling_sharpe)
        axes[1].set_title('60-Day Rolling Sharpe Ratio')
        axes[1].axhline(0, color='red', linestyle='--')
        axes[1].grid(True, alpha=0.3)

        plt.tight_layout()
        plt.show()

# 使用示例
backtester = MomentumBacktester(
    tickers=['AAPL', 'GOOGL', 'MSFT', 'AMZN', 'NVDA'],
    start='2022-01-01',
    end='2023-12-31',
    lookback=60,
    holding=20,
    strategy='csmom'
)

backtester.load_data().generate_signals().run_backtest().performance()
backtester.plot()
```
