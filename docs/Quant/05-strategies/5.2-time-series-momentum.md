# 5.2 时间序列动量：收益率排序、趋势确认

## 时间序列动量 (TSMOM) 原理

```python
"""
时间序列动量策略：

1. 计算资产过去N日的收益率
2. 如果收益率 > 0 → 做多
3. 如果收益率 < 0 → 做空
4. 持有M日后平仓

核心思想：趋势确认后顺势而为
"""

import numpy as np
import pandas as pd
import yfinance as yf

# 参数
lookback = 20    # 回看期（天）
holding = 10     # 持有期（天）
```

## Python 实现

```python
def tsmom_strategy(prices, lookback=20, holding=10):
    """
    时间序列动量策略

    Args:
        prices: 价格序列
        lookback: 回看期
        holding: 持有期

    Returns:
        signals: 交易信号 (1=做多, -1=做空, 0=空仓)
    """
    # 计算动量信号
    returns = prices.pct_change(lookback)

    # 生成信号：正收益做多，负收益做空
    signals = np.where(returns > 0, 1, -1)

    # 每holding天交易一次
    # 将信号重采样到holding频率
    signals = pd.Series(signals, index=prices.index)
    signals = signals.resample(f'{holding}d').ffill()

    # 移位避开前视偏差
    signals = signals.shift(1).fillna(0)

    return signals

# 测试策略
df = yf.download('AAPL', start='2022-01-01', end='2023-12-31')
df.columns = df.columns.get_level_values(0)
prices = df['close']

signals = tsmom_strategy(prices, lookback=20, holding=10)
print(f"做多次数: {(signals > 0).sum()}")
print(f"做空次数: {(signals < 0).sum()}")
```

## 策略表现评估

```python
def evaluate_strategy(prices, signals):
    """评估策略表现"""
    # 计算收益率
    returns = prices.pct_change()

    # 策略收益 = 信号 * 下一期收益
    strategy_returns = signals.shift(1) * returns

    # 去除NaN和初始信号
    valid_mask = (signals.shift(1) != 0) & (~strategy_returns.isna())
    strategy_returns = strategy_returns[valid_mask]

    # 统计
    total_return = (1 + strategy_returns).prod() - 1
    annual_return = strategy_returns.mean() * 252
    annual_vol = strategy_returns.std() * np.sqrt(252)
    sharpe = annual_return / annual_vol if annual_vol > 0 else 0
    max_dd = (1 + strategy_returns).cumprod().div((1 + strategy_returns).cumprod().cummax()).min() - 1

    print(f"总收益: {total_return:.2%}")
    print(f"年化收益: {annual_return:.2%}")
    print(f"年化波动率: {annual_vol:.2%}")
    print(f"夏普比率: {sharpe:.2f}")
    print(f"最大回撤: {max_dd:.2%}")

    return {
        'total_return': total_return,
        'annual_return': annual_return,
        'annual_vol': annual_vol,
        'sharpe': sharpe,
        'max_drawdown': max_dd
    }

results = evaluate_strategy(prices, signals)
```
