## LSTM时间序列模型

```python
class LSTMModel(nn.Module):
    """LSTM模型用于时间序列"""

    def __init__(self, input_dim, hidden_dim=64, num_layers=2, dropout=0.2):
        super().__init__()
        self.hidden_dim = hidden_dim
        self.num_layers = num_layers

        self.lstm = nn.LSTM(
            input_size=input_dim,
            hidden_size=hidden_dim,
            num_layers=num_layers,
            batch_first=True,
            dropout=dropout if num_layers > 1 else 0
        )

        self.fc = nn.Sequential(
            nn.Linear(hidden_dim, hidden_dim // 2),
            nn.ReLU(),
            nn.Dropout(dropout),
            nn.Linear(hidden_dim // 2, 1),
            nn.Sigmoid()
        )

    def forward(self, x):
        # x shape: (batch, seq_len, input_dim)
        lstm_out, _ = self.lstm(x)
        # 取最后一个时间步的输出
        last_output = lstm_out[:, -1, :]
        return self.fc(last_output)

def prepare_sequences(X, y, seq_length=20):
    """准备序列数据"""
    X_seq, y_seq = [], []

    for i in range(len(X) - seq_length):
        X_seq.append(X[i:i+seq_length])
        y_seq.append(y[i+seq_length])

    return np.array(X_seq), np.array(y_seq)

# 准备序列数据
seq_length = 20
X_seq, y_seq = prepare_sequences(X.values, y.values, seq_length)

# 转换为tensor
X_seq_tensor = torch.FloatTensor(X_seq).to(device)
y_seq_tensor = torch.FloatTensor(y_seq).unsqueeze(1).to(device)

# 创建模型
lstm_model = LSTMModel(input_dim=X.shape[1], hidden_dim=64, num_layers=2).to(device)

# 训练
criterion = nn.BCELoss()
optimizer = optim.Adam(lstm_model.parameters(), lr=0.001)

epochs = 50
for epoch in range(epochs):
    lstm_model.train()
    total_loss = 0

    # Mini-batch训练
    for i in range(0, len(X_seq_tensor), 32):
        X_batch = X_seq_tensor[i:i+32]
        y_batch = y_seq_tensor[i:i+32]

        optimizer.zero_grad()
        outputs = lstm_model(X_batch)
        loss = criterion(outputs, y_batch)
        loss.backward()
        optimizer.step()

        total_loss += loss.item()

    if (epoch + 1) % 10 == 0:
        print(f"Epoch {epoch+1}/{epochs}, Loss: {total_loss:.4f}")
```

## 注意力机制

```python
class AttentionLayer(nn.Module):
    """简单的注意力层"""

    def __init__(self, hidden_dim):
        super().__init__()
        self.attention = nn.Linear(hidden_dim, 1)

    def forward(self, lstm_output):
        # lstm_output: (batch, seq_len, hidden_dim)
        attention_weights = torch.softmax(self.attention(lstm_output), dim=1)
        # 加权求和
        context = torch.sum(attention_weights * lstm_output, dim=1)
        return context, attention_weights

class AttentionLSTM(nn.Module):
    """带注意力机制的LSTM"""

    def __init__(self, input_dim, hidden_dim=64, num_layers=2, dropout=0.2):
        super().__init__()

        self.lstm = nn.LSTM(
            input_size=input_dim,
            hidden_size=hidden_dim,
            num_layers=num_layers,
            batch_first=True,
            dropout=dropout if num_layers > 1 else 0
        )

        self.attention = AttentionLayer(hidden_dim)

        self.fc = nn.Sequential(
            nn.Linear(hidden_dim, hidden_dim // 2),
            nn.ReLU(),
            nn.Dropout(dropout),
            nn.Linear(hidden_dim // 2, 1),
            nn.Sigmoid()
        )

    def forward(self, x):
        lstm_out, _ = self.lstm(x)
        context, attention_weights = self.attention(lstm_out)
        return self.fc(context), attention_weights
```

## 强化学习基础

```python
import gym
from collections import deque
import random

class TradingEnvironment(gym.Env):
    """简单的交易环境（类似OpenAI Gym风格）"""

    def __init__(self, prices, initial_balance=10000, transaction_fee=0.001):
        super().__init__()
        self.prices = prices
        self.initial_balance = initial_balance
        self.transaction_fee = transaction_fee

        # 状态空间: (balance, position, price, returns...)
        self.n_features = 5
        self.action_space = gym.spaces.Discrete(3)  # 0=持有, 1=买入, 2=卖出
        self.observation_space = gym.spaces.Box(low=-np.inf, high=np.inf,
                                                shape=(self.n_features,))

        self.reset()

    def reset(self):
        self.current_step = 0
        self.balance = self.initial_balance
        self.position = 0  # 持仓股数
        self.total_profit = 0
        self.trade_history = []
        return self._get_state()

    def _get_state(self):
        """获取当前状态"""
        price = self.prices[self.current_step]
        position_value = self.position * price
        portfolio_value = self.balance + position_value

        # 计算收益率
        if self.current_step > 0:
            returns = (price - self.prices[self.current_step-1]) / self.prices[self.current_step-1]
        else:
            returns = 0

        return np.array([
            self.balance / self.initial_balance,  # 资金比例
            position_value / self.initial_balance,  # 持仓比例
            price / self.prices[0] if self.current_step > 0 else 1.0,  # 价格相对起点
            returns,  # 当期收益
            self.total_profit / self.initial_balance  # 累计收益比例
        ])

    def step(self, action):
        price = self.prices[self.current_step]
        done = self.current_step >= len(self.prices) - 1

        reward = 0

        if action == 1:  # 买入
            if self.balance > price * (1 + self.transaction_fee):
                shares = self.balance * (1 - self.transaction_fee) / price
                self.position += shares
                self.balance -= shares * price * (1 + self.transaction_fee)

        elif action == 2:  # 卖出
            if self.position > 0:
                self.balance += self.position * price * (1 - self.transaction_fee)
                self.position = 0

        # 计算奖励（持仓价值变化）
        new_portfolio_value = self.balance + self.position * price
        reward = (new_portfolio_value - self.initial_balance) / self.initial_balance

        self.current_step += 1
        return self._get_state(), reward, done, {}
```

## Q-Learning

```python
class DQNAgent:
    """Deep Q-Network agent"""

    def __init__(self, state_dim, action_dim, learning_rate=0.001,
                 gamma=0.99, epsilon=1.0, epsilon_decay=0.995, epsilon_min=0.01):
        self.state_dim = state_dim
        self.action_dim = action_dim
        self.gamma = gamma
        self.epsilon = epsilon
        self.epsilon_decay = epsilon_decay
        self.epsilon_min = epsilon_min

        # Q网络
        self.q_network = nn.Sequential(
            nn.Linear(state_dim, 64),
            nn.ReLU(),
            nn.Linear(64, 64),
            nn.ReLU(),
            nn.Linear(64, action_dim)
        ).to(device)

        self.optimizer = optim.Adam(self.q_network.parameters(), lr=learning_rate)
        self.memory = deque(maxlen=10000)  # 经验回放

    def act(self, state):
        """选择动作（epsilon-greedy）"""
        if random.random() < self.epsilon:
            return random.randint(0, self.action_dim - 1)

        state_tensor = torch.FloatTensor(state).unsqueeze(0).to(device)
        q_values = self.q_network(state_tensor)
        return q_values.argmax().item()

    def remember(self, state, action, reward, next_state, done):
        """存储经验"""
        self.memory.append((state, action, reward, next_state, done))

    def replay(self, batch_size=32):
        """经验回放"""
        if len(self.memory) < batch_size:
            return

        batch = random.sample(self.memory, batch_size)

        states = torch.FloatTensor(np.array([e[0] for e in batch])).to(device)
        actions = torch.LongTensor([e[1] for e in batch]).to(device)
        rewards = torch.FloatTensor([e[2] for e in batch]).to(device)
        next_states = torch.FloatTensor(np.array([e[3] for e in batch])).to(device)
        dones = torch.FloatTensor([e[4] for e in batch]).to(device)

        # 计算当前Q值
        current_q = self.q_network(states).gather(1, actions.unsqueeze(1))

        # 计算目标Q值
        next_q = self.q_network(next_states).max(1)[0]
        target_q = rewards + (1 - dones) * self.gamma * next_q

        # 损失函数
        loss = nn.MSELoss()(current_q.squeeze(), target_q)

        # 反向传播
        self.optimizer.zero_grad()
        loss.backward()
        self.optimizer.step()

        # 降低epsilon
        self.epsilon = max(self.epsilon_min, self.epsilon * self.epsilon_decay)

        return loss.item()

# 训练DQN agent
agent = DQNAgent(state_dim=5, action_dim=3)

episodes = 100
for episode in range(episodes):
    state = env.reset()
    total_reward = 0

    while True:
        action = agent.act(state)
        next_state, reward, done, _ = env.step(action)
        agent.remember(state, action, reward, next_state, done)

        total_reward += reward
        state = next_state

        # 学习
        loss = agent.replay()

        if done:
            break

    if (episode + 1) % 10 == 0:
        print(f"Episode {episode+1}: Total Reward = {total_reward:.4f}, Epsilon = {agent.epsilon:.4f}")
```

## 何时使用深度学习

| 情况               | 建议                        |
| ------------------ | --------------------------- |
| 数据量小（<10000） | 使用传统ML（如RF、XGBoost） |
| 数据量大，特征复杂 | 深度学习可能有效            |
| 需要序列建模       | LSTM/Transformer            |
| 交易信号明确       | 监督学习即可                |
| 需要学习交易策略   | 强化学习                    |
| 数据噪声大         | 深度学习可能过拟合          |

## 注意事项

1. **过拟合风险**: 金融数据噪声大，深度学习容易过拟合
2. **数据质量**: 深度学习对数据质量敏感
3. **计算成本**: 训练和调优需要大量计算资源
4. **可解释性**: 深度学习模型难以解释

## 总结

深度学习是强大的工具，但：

- 不是银弹
- 需要大量数据和计算
- 更容易过拟合
- 可解释性差
- 传统ML往往效果相当
