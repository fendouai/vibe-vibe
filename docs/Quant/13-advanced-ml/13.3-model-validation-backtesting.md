## 时间序列交叉验证

金融数据具有时间依赖性，不能使用随机打乱的交叉验证：

```python
from sklearn.model_selection import TimeSeriesSplit
import numpy as np

def time_series_cv(X, y, n_splits=5, test_size=None):
    """
    时间序列交叉验证

    每次用更早的数据训练，用更晚的数据测试
    """
    tscv = TimeSeriesSplit(n_splits=n_splits, test_size=test_size)

    for train_idx, test_idx in tscv.split(X):
        X_train, X_test = X.iloc[train_idx], X.iloc[test_idx]
        y_train, y_test = y.iloc[train_idx], y.iloc[test_idx]

        yield X_train, X_test, y_train, y_test

# 使用示例
for fold, (X_train, X_test, y_train, y_test) in enumerate(time_series_cv(X, y, n_splits=5)):
    print(f"Fold {fold+1}:")
    print(f"  Train: {len(X_train)} samples ({X_train.index[0]} to {X_train.index[-1]})")
    print(f"  Test: {len(X_test)} samples ({X_test.index[0]} to {X_test.index[-1]})")
```

## Walk-Forward Analysis

```python
def walk_forward_analysis(model_class, X, y, train_size=252, test_size=21,
                         n_steps=None, **model_params):
    """
    Walk-Forward分析

    模拟真实交易场景：
    - 用历史数据训练模型
    - 在未来数据上测试
    - 定期重新训练模型
    """
    results = []
    n_samples = len(X)

    step = 0
    while train_size + test_size <= n_samples:
        train_end = train_size
        test_end = train_end + test_size

        # 分割数据
        X_train = X.iloc[:train_end]
        X_test = X.iloc[train_end:test_end]
        y_train = y.iloc[:train_end]
        y_test = y.iloc[train_end:test_end]

        # 训练模型
        model = model_class(**model_params)
        model.fit(X_train, y_train)

        # 测试
        y_pred = model.predict(X_test)
        y_pred_proba = model.predict_proba(X_test)[:, 1] if hasattr(model, 'predict_proba') else None

        # 计算指标
        accuracy = accuracy_score(y_test, y_pred)

        # 记录结果
        results.append({
            'step': step,
            'train_start': X_train.index[0],
            'train_end': X_train.index[-1],
            'test_start': X_test.index[0],
            'test_end': X_test.index[-1],
            'accuracy': accuracy,
            'train_size': len(X_train),
            'test_size': len(X_test)
        })

        # 滑动窗口
        train_size += test_size
        step += 1

        if n_steps and step >= n_steps:
            break

    return pd.DataFrame(results)

# 使用示例
results = walk_forward_analysis(
    RandomForestClassifier,
    X, y,
    train_size=252,  # 约1年训练数据
    test_size=21,    # 约1个月测试
    n_estimators=100,
    max_depth=10,
    random_state=42
)

print(results)
print(f"\n平均准确率: {results['accuracy'].mean():.4f}")
print(f"准确率标准差: {results['accuracy'].std():.4f}")
```

```python
def label_by_horizon(prices: pd.Series, horizon: int = 1, threshold: float = 0.0):
    """
    固定时间范围标签

    Parameters:
    - horizon: 预测时间步数
    - threshold: 涨跌阈值
    """
    future_returns = prices.shift(-horizon) / prices - 1

    # 三分类：涨、跌、震荡
    labels = pd.Series(0, index=prices.index)  # 震荡
    labels[future_returns > threshold] = 1       # 涨
    labels[future_returns < -threshold] = -1     # 跌

    return labels, future_returns
```

### Triple Barrier 标签

```python
def triple_barrier_labels(prices: pd.Series, horizon: int = 20,
                         take_profit: float = 0.05, stop_loss: float = 0.02):
    """
    三重障碍标签（常见于衍生品和日内交易）

    - 止盈障碍
    - 止损障碍
    - 时间障碍
    """
    labels = pd.Series(0, index=prices.index)
    horizons = pd.Series(horizon, index=prices.index)

    for i in range(len(prices) - horizon):
        entry_price = prices.iloc[i]

        # 检查是否触及止盈
        tp_hit = (prices.iloc[i+1:i+horizon+1] >= entry_price * (1 + take_profit)).any()
        sl_hit = (prices.iloc[i+1:i+horizon+1] <= entry_price * (1 - stop_loss)).any()

        if tp_hit and not sl_hit:
            labels.iloc[i] = 1  # 止盈出局
        elif sl_hit and not tp_hit:
            labels.iloc[i] = -1  # 止损出局
        else:
            # 时间到期，根据最终收益判断
            final_return = (prices.iloc[i+horizon] / entry_price) - 1
            labels.iloc[i] = 1 if final_return > 0 else -1

    return labels, horizons

# 示例
labels_tb, horizons_tb = triple_barrier_labels(data['close'], horizon=20, take_profit=0.05, stop_loss=0.02)
print(f"Triple Barrier标签分布:\n{labels_tb.value_counts()}")
```

### 样本外测试

```python
def out_of_sample_test(model, X_train, X_test, y_train, y_test):
    """严格的样本外测试"""
    # 训练集上评估
    train_pred = model.predict(X_train)
    train_accuracy = accuracy_score(y_train, train_pred)

    # 测试集上评估
    test_pred = model.predict(X_test)
    test_accuracy = accuracy_score(y_test, test_pred)

    # 过拟合程度
    overfitting_gap = train_accuracy - test_accuracy

    print(f"训练集准确率: {train_accuracy:.4f}")
    print(f"测试集准确率: {test_accuracy:.4f}")
    print(f"过拟合差距: {overfitting_gap:.4f}")

    if overfitting_gap > 0.1:
        print("⚠️ 警告: 存在明显过拟合")

    return {
        'train_accuracy': train_accuracy,
        'test_accuracy': test_accuracy,
        'overfitting_gap': overfitting_gap
    }
```

```python
import matplotlib.pyplot as plt
from sklearn.model_selection import learning_curve

def plot_learning_curve(model, X, y, train_sizes=np.linspace(0.1, 1.0, 10)):
    """绘制学习曲线"""
    train_sizes_abs, train_scores, test_scores = learning_curve(
        model, X, y,
        train_sizes=train_sizes,
        cv=5,
        scoring='accuracy',
        n_jobs=-1
    )

    train_scores_mean = train_scores.mean(axis=1)
    train_scores_std = train_scores.std(axis=1)
    test_scores_mean = test_scores.mean(axis=1)
    test_scores_std = test_scores.std(axis=1)

    plt.figure(figsize=(10, 6))
    plt.fill_between(train_sizes_abs, train_scores_mean - train_scores_std,
                     train_scores_mean + train_scores_std, alpha=0.1, color='r')
    plt.fill_between(train_sizes_abs, test_scores_mean - test_scores_std,
                     test_scores_mean + test_scores_std, alpha=0.1, color='g')
    plt.plot(train_sizes_abs, train_scores_mean, 'o-', color='r', label='Training score')
    plt.plot(train_sizes_abs, test_scores_mean, 'o-', color='g', label='Cross-validation score')
    plt.xlabel('Training examples')
    plt.ylabel('Accuracy')
    plt.legend()
    plt.title('Learning Curve')
    plt.grid(True)
    plt.tight_layout()
    plt.savefig('learning_curve.png', dpi=150)
    plt.show()
```

```python
def feature_importance_stability(model, X, y, n_bootstrap=10):
    """
    检验特征重要性稳定性

    如果特征在不同数据子集上重要性差异大，说明不稳定
    """
    importance_records = []

    for i in range(n_bootstrap):
        # Bootstrap采样
        indices = np.random.choice(len(X), len(X), replace=True)
        X_boot = X.iloc[indices]
        y_boot = y.iloc[indices]

        # 训练模型
        model_copy = RandomForestClassifier(n_estimators=50, random_state=i)
        model_copy.fit(X_boot, y_boot)

        # 记录特征重要性
        importance_records.append(
            pd.Series(model_copy.feature_importances_, index=X.columns)
        )

    # 汇总
    importance_df = pd.DataFrame(importance_records)
    stability = pd.DataFrame({
        'mean': importance_df.mean(),
        'std': importance_df.std(),
        'cv': importance_df.std() / (importance_df.mean() + 1e-10)  # 变异系数
    }).sort_values('mean', ascending=False)

    print("特征重要性稳定性分析:")
    print(stability.head(10))

    return stability

# 稳定性分析
stability = feature_importance_stability(RandomForestClassifier(n_estimators=100, random_state=42), X, y)
```

## 蒙特卡洛模拟

```python
def monte_carlo_simulation(predictions, actual, n_simulations=1000):
    """
    蒙特卡洛模拟评估预测不确定性

    predictions: 模型预测概率
    actual: 实际结果
    """
    n = len(actual)
    base_accuracy = accuracy_score(actual, (predictions > 0.5).astype(int))

    accuracies = []
    for _ in range(n_simulations):
        # Bootstrap采样
        indices = np.random.choice(n, n, replace=True)
        boot_pred = predictions[indices]
        boot_actual = actual.iloc[indices]

        boot_acc = accuracy_score(boot_actual, (boot_pred > 0.5).astype(int))
        accuracies.append(boot_acc)

    print(f"基础准确率: {base_accuracy:.4f}")
    print(f"Monte Carlo 95%置信区间: [{np.percentile(accuracies, 2.5):.4f}, "
          f"{np.percentile(accuracies, 97.5):.4f}]")
    print(f"标准误差: {np.std(accuracies):.4f}")

    return np.array(accuracies)

# 使用示例
```

## 完整ML回测框架

```python
class MLBacktester:
    """机器学习策略回测器"""

    def __init__(self, model_class, features, labels, train_size=252, test_size=21):
        self.model_class = model_class
        self.features = features
        self.labels = labels
        self.train_size = train_size
        self.test_size = test_size

        self.model = None
        self.results = []

    def run(self, **model_params):
        """运行walk-forward回测"""
        n = len(self.features)
        position = 0

        for i in range(self.train_size, n - self.test_size, self.test_size):
            # 训练数据
            X_train = self.features.iloc[i - self.train_size:i]
            y_train = self.labels.iloc[i - self.train_size:i]

            # 测试数据
            X_test = self.features.iloc[i:i + self.test_size]
            y_test = self.labels.iloc[i:i + self.test_size]

            # 训练模型
            self.model = self.model_class(**model_params)
            self.model.fit(X_train, y_train)

            # 预测
            predictions = self.model.predict(X_test)
            probabilities = self.model.predict_proba(X_test)[:, 1]

            # 记录结果
            for j, (idx, pred, prob) in enumerate(zip(X_test.index, predictions, probabilities)):
                self.results.append({
                    'date': idx,
                    'prediction': pred,
                    'probability': prob,
                    'actual': y_test.iloc[j],
                    'correct': pred == y_test.iloc[j]
                })

        return pd.DataFrame(self.results)

    def evaluate(self):
        """评估回测结果"""
        df = pd.DataFrame(self.results)

        accuracy = df['correct'].mean()

        # 按概率分组查看准确率
        df['prob_bin'] = pd.cut(df['probability'], bins=[0, 0.3, 0.5, 0.7, 1.0])
        grouped = df.groupby('prob_bin')['correct'].agg(['mean', 'count'])

        print(f"整体准确率: {accuracy:.4f}")
        print(f"\n按预测概率分组的准确率:")
        print(grouped)

        return df

# 使用示例
backtester = MLBacktester(
    RandomForestClassifier,
    X, y,
    train_size=252,
    test_size=21
)

results = backtester.run(n_estimators=100, max_depth=10, random_state=42)
df_results = backtester.evaluate()
```

## 总结

| 验证方法       | 说明         | 注意事项         |
| -------------- | ------------ | ---------------- |
| 时间序列CV     | 保持时间顺序 | 不能随机打乱     |
| Walk-Forward   | 模拟真实交易 | 最接近实盘       |
| Triple Barrier | 更现实的标签 | 需要仔细设定参数 |
| Monte Carlo    | 不确定性量化 | 计算量大         |
