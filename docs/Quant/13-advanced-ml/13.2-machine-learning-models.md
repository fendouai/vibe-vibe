# 13.2 机器学习模型

## ML在量化交易中的应用

机器学习可以用于：

- **Alpha预测**: 预测未来收益方向或幅度
- **因子模型**: 发现新的Alpha因子
- **风险管理**: 预测极端损失
- **交易执行**: 优化下单策略

## 回归 vs 分类

```python
# 回归：预测连续值（收益率）
# y = 未来收益率

# 分类：预测离散值（涨跌）
# y = 1 (涨) or 0 (跌)

from sklearn.model_selection import train_test_split
from sklearn.metrics import mean_squared_error, accuracy_score, classification_report

# 准备数据
X = features.fillna(0)  # 特征
y_regression = returns.shift(-1).fillna(0)  # 回归目标：下期收益率
y_classification = (returns.shift(-1) > 0).astype(int).fillna(0)  # 分类目标：涨跌

# 分割数据
X_train, X_test, y_train, y_test = train_test_split(
    X, y_classification, test_size=0.2, random_state=42
)
```

## 线性模型

### 线性回归

```python
from sklearn.linear_model import LinearRegression, Ridge, Lasso, ElasticNet

# 普通线性回归
lr = LinearRegression()
lr.fit(X_train, y_train)
y_pred = lr.predict(X_test)

mse = mean_squared_error(y_test, y_pred)
print(f"Linear Regression MSE: {mse:.6f}")

# Ridge回归（L2正则化）- 处理多重共线性
ridge = Ridge(alpha=1.0)
ridge.fit(X_train, y_train)

# Lasso回归（L1正则化）- 特征选择
lasso = Lasso(alpha=0.01)
lasso.fit(X_train, y_train)

# ElasticNet（L1+L2组合）
elastic = ElasticNet(alpha=0.01, l1_ratio=0.5)
elastic.fit(X_train, y_train)

# 查看系数
coef_df = pd.Series(lr.coef_, index=X.columns).sort_values(ascending=False)
print("\nLinear Regression Coefficients:")
print(coef_df.head(10))
```

### 逻辑回归（分类）

```python
from sklearn.linear_model import LogisticRegression

# 逻辑回归用于二分类
lr_clf = LogisticRegression(C=1.0, max_iter=1000)
lr_clf.fit(X_train, y_train)

y_pred_proba = lr_clf.predict_proba(X_test)[:, 1]
y_pred = lr_clf.predict(X_test)

accuracy = accuracy_score(y_test, y_pred)
print(f"Logistic Regression Accuracy: {accuracy:.4f}")

# 查看分类报告
print("\nClassification Report:")
print(classification_report(y_test, y_pred))
```

## 模型比较

```python
def compare_models(X_train, X_test, y_train, y_test, task='classification'):
    """比较多个模型的效果"""
    from sklearn.neighbors import KNeighborsClassifier
    from sklearn.svm import SVC
    from sklearn.neural_network import MLPClassifier

    if task == 'classification':
        models = {
            'Logistic Regression': LogisticRegression(max_iter=1000),
            'Random Forest': RandomForestClassifier(n_estimators=100, max_depth=10),
            'Gradient Boosting': GradientBoostingClassifier(n_estimators=100),
            'XGBoost': xgb.XGBClassifier(n_estimators=100, use_label_encoder=False, eval_metric='logloss'),
            'LightGBM': lgb.LGBMClassifier(n_estimators=100, verbose=-1),
            'KNN': KNeighborsClassifier(n_neighbors=5),
            'SVM': SVC(probability=True),
            'MLP': MLPClassifier(hidden_layer_sizes=(100, 50), max_iter=500)
        }
        metric = accuracy_score
    else:
        models = {
            'Linear Regression': LinearRegression(),
            'Ridge': Ridge(alpha=1.0),
            'Random Forest': RandomForestRegressor(n_estimators=100),
            'Gradient Boosting': GradientBoostingRegressor(n_estimators=100),
            'XGBoost': xgb.XGBRegressor(n_estimators=100)
        }
        metric = mean_squared_error

    results = {}
    for name, model in models.items():
        model.fit(X_train, y_train)
        y_pred = model.predict(X_test)

        if task == 'classification':
            results[name] = metric(y_test, y_pred)
        else:
            results[name] = metric(y_test, y_pred)

    # 按性能排序
    sorted_results = sorted(results.items(), key=lambda x: x[1], reverse=(task=='classification'))

    print(f"\n{'='*50}")
    print(f"Model Comparison ({task})")
    print(f"{'='*50}")
    for name, score in sorted_results:
        if task == 'classification':
            print(f"{name:25s}: {score:.4f}")
        else:
            print(f"{name:25s}: {score:.6f}")

    return results

# 比较分类模型
results = compare_models(X_train, X_test, y_train, y_test, 'classification')
```

## 超参数调优

```python
from sklearn.model_selection import GridSearchCV, RandomizedSearchCV

def tune_hyperparameters(model, param_grid, X_train, y_train, cv=5):
    """
    超参数调优

    Parameters:
    - model: 基础模型
    - param_grid: 参数网格
    - cv: 交叉验证折数
    """
    grid_search = GridSearchCV(
        model,
        param_grid,
        cv=cv,
        scoring='accuracy',
        n_jobs=-1,
        verbose=1
    )

    grid_search.fit(X_train, y_train)

    print(f"\nBest parameters: {grid_search.best_params_}")
    print(f"Best CV score: {grid_search.best_score_:.4f}")

    return grid_search.best_estimator_, grid_search.best_params_

# 示例：调优随机森林
param_grid = {
    'n_estimators': [50, 100, 200],
    'max_depth': [5, 10, 15, None],
    'min_samples_leaf': [10, 20, 50]
}

rf_tuned, best_params = tune_hyperparameters(
    RandomForestClassifier(random_state=42),
    param_grid,
    X_train, y_train
)

# 使用最佳模型预测
y_pred_tuned = rf_tuned.predict(X_test)
print(f"\nTuned Random Forest Accuracy: {accuracy_score(y_test, y_pred_tuned):.4f}")
```

## 模型集成

```python
class EnsembleModel:
    """集成模型"""

    def __init__(self, models: list, weights: list = None, voting='soft'):
        self.models = models
        self.weights = weights or [1/len(models)] * len(models)
        self.voting = voting

    def fit(self, X, y):
        for model in self.models:
            model.fit(X, y)
        return self

    def predict_proba(self, X):
        """预测概率"""
        predictions = np.zeros((len(X), 2))

        for model, weight in zip(self.models, self.weights):
            predictions += weight * model.predict_proba(X)

        return predictions

    def predict(self, X):
        """预测类别"""
        if self.voting == 'soft':
            proba = self.predict_proba(X)
            return np.argmax(proba, axis=1)
        else:
            # 硬投票
            predictions = np.array([m.predict(X) for m in self.models])
            from scipy.stats import mode
            return mode(predictions, axis=0)[0].flatten()

# 创建集成模型
ensemble = EnsembleModel([
    RandomForestClassifier(n_estimators=100, random_state=42),
    GradientBoostingClassifier(n_estimators=100, random_state=42),
    xgb.XGBClassifier(n_estimators=100, use_label_encoder=False, eval_metric='logloss', random_state=42)
], weights=[0.3, 0.3, 0.4])

ensemble.fit(X_train, y_train)
y_pred_ensemble = ensemble.predict(X_test)
print(f"Ensemble Accuracy: {accuracy_score(y_test, y_pred_ensemble):.4f}")
```

## 总结

| 模型类型          | 优点               | 缺点       | 适用场景       |
| ----------------- | ------------------ | ---------- | -------------- |
| 线性模型          | 可解释、简单       | 线性假设   | 基础baseline   |
| 决策树            | 可解释、处理非线性 | 容易过拟合 | 特征重要性分析 |
| 随机森林          | 抗过拟合、并行     | 慢、难解释 | 分类、回归     |
| Gradient Boosting | 精度高             | 容易过拟合 | 竞赛、实战     |
| XGBoost/LightGBM  | 高效、精度高       | 参数多     | 大规模数据     |
