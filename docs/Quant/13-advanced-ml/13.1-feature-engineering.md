# 13.1 特征工程

## 什么是特征工程？

特征工程是将原始数据转换为模型可用的特征的过程。在量化交易中，好的特征（Alpha因子）是策略成功的关键。

```
原始数据 → 特征工程 → 特征 → ML模型 → 预测 → 交易信号
```

## 技术指标特征

### 趋势类指标

```python
import pandas as pd
import numpy as np

def calculate_technical_features(df: pd.DataFrame) -> pd.DataFrame:
    """计算技术指标特征"""
    features = pd.DataFrame(index=df.index)

    # 移动平均线
    features['sma_5'] = df['close'].rolling(5).mean()
    features['sma_20'] = df['close'].rolling(20).mean()
    features['sma_60'] = df['close'].rolling(60).mean()

    # 价格与均线的关系
    features['price_to_sma20'] = df['close'] / features['sma_20']
    features['price_to_sma60'] = df['close'] / features['sma_60']

    # 均线交叉信号
    features['sma_cross'] = (features['sma_5'] - features['sma_20']) / features['sma_20']

    # 指数移动平均
    features['ema_12'] = df['close'].ewm(span=12).mean()
    features['ema_26'] = df['close'].ewm(span=26).mean()

    # MACD
    features['macd'] = features['ema_12'] - features['ema_26']
    features['macd_signal'] = features['macd'].ewm(span=9).mean()
    features['macd_hist'] = features['macd'] - features['macd_signal']

    return features

# 示例数据
np.random.seed(42)
data = pd.DataFrame({
    'open': np.random.randn(100).cumsum() + 100,
    'high': np.random.randn(100).cumsum() + 102,
    'low': np.random.randn(100).cumsum() + 98,
    'close': np.random.randn(100).cumsum() + 100,
    'volume': np.random.randint(1000000, 10000000, 100)
}, index=pd.date_range('2024-01-01', periods=100))

features = calculate_technical_features(data)
print(features.tail(10))
```

### 动量类指标

```python
def calculate_momentum_features(df: pd.DataFrame) -> pd.DataFrame:
    """计算动量特征"""
    features = pd.DataFrame(index=df.index)

    # 收益率
    features['returns'] = df['close'].pct_change()

    # 动量（历史收益率）
    for period in [5, 10, 20, 60]:
        features[f'momentum_{period}'] = df['close'].pct_change(period)

    # RSI (Relative Strength Index)
    delta = df['close'].diff()
    gain = delta.where(delta > 0, 0).rolling(14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(14).mean()
    rs = gain / loss
    features['rsi_14'] = 100 - (100 / (1 + rs))

    # 随机指标 (Stochastic)
    low_14 = df['low'].rolling(14).min()
    high_14 = df['high'].rolling(14).max()
    features['stoch_k'] = 100 * (df['close'] - low_14) / (high_14 - low_14)
    features['stoch_d'] = features['stoch_k'].rolling(3).mean()

    # CCI (Commodity Channel Index)
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    sma_tp = typical_price.rolling(20).mean()
    mean_deviation = typical_price.rolling(20).apply(lambda x: np.abs(x - x.mean()).mean())
    features['cci_20'] = (typical_price - sma_tp) / (0.015 * mean_deviation)

    # ADX (Average Directional Index)
    features['adx_14'] = calculate_adx(df, 14)

    return features

def calculate_adx(df: pd.DataFrame, period: int = 14) -> pd.Series:
    """计算ADX"""
    high = df['high']
    low = df['low']

    # True Range
    tr1 = high - low
    tr2 = abs(high - df['close'].shift())
    tr3 = abs(low - df['close'].shift())
    tr = pd.concat([tr1, tr2, tr3], axis=1).max(axis=1)

    # Directional Movement
    plus_dm = high.diff()
    minus_dm = -low.diff()
    plus_dm[plus_dm < 0] = 0
    minus_dm[minus_dm < 0] = 0

    # Smoothed
    smoothed_plus_dm = plus_dm.rolling(period).mean()
    smoothed_minus_dm = minus_dm.rolling(period).mean()
    smoothed_tr = tr.rolling(period).mean()

    # DI
    plus_di = 100 * smoothed_plus_dm / smoothed_tr
    minus_di = 100 * smoothed_minus_dm / smoothed_tr

    # DX
    dx = 100 * abs(plus_di - minus_di) / (plus_di + minus_di)

    # ADX
    adx = dx.rolling(period).mean()

    return adx

momentum_features = calculate_momentum_features(data)
```

### 波动率类指标

```python
def calculate_volatility_features(df: pd.DataFrame) -> pd.DataFrame:
    """计算波动率特征"""
    features = pd.DataFrame(index=df.index)

    # 历史波动率
    for period in [10, 20, 60]:
        features[f'volatility_{period}'] = df['close'].pct_change().rolling(period).std()

    # 布林带
    sma20 = df['close'].rolling(20).mean()
    std20 = df['close'].rolling(20).std()
    features['bb_upper'] = sma20 + 2 * std20
    features['bb_lower'] = sma20 - 2 * std20
    features['bb_width'] = (features['bb_upper'] - features['bb_lower']) / sma20
    features['bb_position'] = (df['close'] - features['bb_lower']) / (features['bb_upper'] - features['bb_lower'])

    # ATR (Average True Range)
    high_low = df['high'] - df['low']
    high_close = abs(df['high'] - df['close'].shift())
    low_close = abs(df['low'] - df['close'].shift())
    tr = pd.concat([high_low, high_close, low_close], axis=1).max(axis=1)
    features['atr_14'] = tr.rolling(14).mean()
    features['atr_percent'] = features['atr_14'] / df['close']  # ATR占价格比例

    # 波动率比率
    features['vol_ratio'] = features['volatility_10'] / features['volatility_60']

    return features
```

## 成交量特征

```python
def calculate_volume_features(df: pd.DataFrame) -> pd.DataFrame:
    """计算成交量特征"""
    features = pd.DataFrame(index=df.index)

    # 成交量移动平均
    features['volume_sma_20'] = df['volume'].rolling(20).mean()
    features['volume_ratio'] = df['volume'] / features['volume_sma_20']

    # OBV (On-Balance Volume)
    obv = (np.sign(df['close'].diff()) * df['volume']).fillna(0).cumsum()
    features['obv'] = obv
    features['obv_ma'] = obv.rolling(20).mean()

    # VWAP (Volume Weighted Average Price)
    features['vwap'] = (df['close'] * df['volume']).cumsum() / df['volume'].cumsum()

    # 成交量与价格背离
    price_change = df['close'].diff()
    features['volume_price_divergence'] = np.where(
        price_change > 0, df['volume'], -df['volume']
    )

    # 累积/派发线 (A/D Line)
    accumulation = ((df['close'] - df['low']) - (df['high'] - df['close'])) / (df['high'] - df['low'])
    accumulation = accumulation.fillna(0)
    features['ad_line'] = (accumulation * df['volume']).cumsum()

    return features
```

## 特征标准化

```python
from sklearn.preprocessing import StandardScaler, MinMaxScaler, RobustScaler

def normalize_features(features: pd.DataFrame, method: str = 'standard') -> pd.DataFrame:
    """
    标准化特征

    Parameters:
    - features: 原始特征
    - method: 'standard' (Z-score), 'minmax', 'robust'
    """
    features_clean = features.dropna()

    if method == 'standard':
        scaler = StandardScaler()
    elif method == 'minmax':
        scaler = MinMaxScaler()
    elif method == 'robust':
        scaler = RobustScaler()
    else:
        raise ValueError(f"Unknown method: {method}")

    normalized = pd.DataFrame(
        scaler.fit_transform(features_clean),
        columns=features_clean.columns,
        index=features_clean.index
    )

    return normalized

# 示例
normalized_features = normalize_features(features, 'standard')
print(normalized_features.describe())
```

## 特征选择

```python
from sklearn.feature_selection import SelectKBest, f_regression, mutual_info_regression
from sklearn.ensemble import RandomForestRegressor

def select_features(X: pd.DataFrame, y: pd.Series, k: int = 10) -> list:
    """
    特征选择

    Parameters:
    - X: 特征矩阵
    - y: 目标变量
    - k: 选择前k个特征
    """
    # 方法1: 相关性分析
    correlations = X.corrwith(y).abs().sort_values(ascending=False)
    print("Top 10 correlated features:")
    print(correlations.head(10))

    # 方法2: SelectKBest
    selector = SelectKBest(f_regression, k=k)
    X_selected = selector.fit_transform(X.fillna(0), y)
    selected_features = X.columns[selector.get_support()].tolist()
    print(f"\nSelected features (SelectKBest): {selected_features}")

    # 方法3: 随机森林特征重要性
    rf = RandomForestRegressor(n_estimators=100, random_state=42)
    rf.fit(X.fillna(0), y)
    importance = pd.Series(rf.feature_importances_, index=X.columns).sort_values(ascending=False)
    print(f"\nTop 10 important features (RF):")
    print(importance.head(10))

    return selected_features
```

## 总结

| 特征类型 | 示例            | 用途         |
| -------- | --------------- | ------------ |
| 趋势类   | SMA, EMA, MACD  | 捕捉市场方向 |
| 动量类   | RSI, Stochastic | 识别超买超卖 |
| 波动率类 | Bollinger, ATR  | 衡量风险     |
| 成交量类 | OBV, VWAP       | 确认趋势强度 |
