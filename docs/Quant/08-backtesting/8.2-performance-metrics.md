## Sharpe Ratio

```python
def sharpe_ratio(returns, risk_free_rate=0.0, periods=252):
    """
    计算夏普比率

    Sharpe = (策略收益 - 无风险收益) / 策略波动率

    解读：
    - Sharpe > 1: 好
    - Sharpe > 2: 很好
    - Sharpe > 3: 优秀
    """
    excess_returns = returns - risk_free_rate / periods
    if returns.std() == 0:
        return 0
    return np.sqrt(periods) * excess_returns.mean() / returns.std()

# 示例
returns = yf.download('AAPL', start='2022-01-01', end='2023-12-31')['Close'].pct_change().dropna()
sr = sharpe_ratio(returns)
print(f"夏普比率: {sr:.2f}")
```

## Sortino Ratio

```python
def sortino_ratio(returns, risk_free_rate=0.0, periods=252, target=0):
    """
    计算索提诺比率

    Sortino = (策略收益 - 目标收益) / 下行波动率

    只考虑下行风险，不考虑上行波动
    """
    excess_returns = returns - risk_free_rate / periods
    downside_returns = returns[returns < target]

    if len(downside_returns) == 0 or downside_returns.std() == 0:
        return np.nan

    downside_std = downside_returns.std() * np.sqrt(periods)
    return (returns.mean() * periods - risk_free_rate) / downside_std

# 示例
sor = sortino_ratio(returns)
print(f"索提诺比率: {sor:.2f}")
```

## 最大回撤

```python
def max_drawdown(returns):
    """
    计算最大回撤

    最大回撤 = 历史最高点到最低点的最大跌幅
    """
    wealth = (1 + returns).cumprod()
    peak = wealth.cummax()
    drawdown = (wealth - peak) / peak
    return drawdown.min()

def max_drawdown_duration(returns):
    """
    计算最大回撤持续时间
    """
    wealth = (1 + returns).cumprod()
    peak = wealth.cummax()
    drawdown = wealth - peak

    # 找到最大回撤的起止时间
    dd_series = drawdown[drawdown < 0]
    if len(dd_series) == 0:
        return 0, None, None

    max_dd_idx = drawdown.idxmin()

    # 找到回升到峰值的时间
    peak_before_dd = peak[:drawdown.idxmin()].idxmax()

    return abs(drawdown.min()), peak_before_dd, max_dd_idx

# 示例
mdd = max_drawdown(returns)
print(f"最大回撤: {mdd:.2%}")
```

## 完整性能报告

```python
def performance_report(returns, strategy_name='Strategy'):
    """生成完整性能报告"""
    import pandas as pd

    total_return = (1 + returns).prod() - 1
    annual_return = returns.mean() * 252
    annual_vol = returns.std() * np.sqrt(252)
    sharpe = sharpe_ratio(returns)
    sortino = sortino_ratio(returns)
    max_dd = max_drawdown(returns)
    win_rate = (returns > 0).mean()

    # 盈亏比
    avg_win = returns[returns > 0].mean() if (returns > 0).any() else 0
    avg_loss = abs(returns[returns < 0].mean()) if (returns < 0).any() else 0
    profit_factor = avg_win / avg_loss if avg_loss != 0 else np.nan

    results = pd.Series({
        '策略': strategy_name,
        '总收益率': f"{total_return:.2%}",
        '年化收益率': f"{annual_return:.2%}",
        '年化波动率': f"{annual_vol:.2%}",
        '夏普比率': f"{sharpe:.2f}",
        '索提诺比率': f"{sortino:.2f}",
        '最大回撤': f"{max_dd:.2%}",
        '胜率': f"{win_rate:.2%}",
        '盈亏比': f"{profit_factor:.2f}"
    })

    return results

# 示例
report = performance_report(returns, 'AAPL Buy & Hold')
print(report)
```
