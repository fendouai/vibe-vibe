# 2.4 订单类型：市价单、限价单、止损单

## 订单类型概览

```python
# 量化交易中的主要订单类型
order_types = {
    'market': '市价单 - 以市场价格立即成交',
    'limit': '限价单 - 指定价格或更优价格成交',
    'stop': '止损单 - 达到触发价后以市价成交',
    'stop_limit': '止损限价单 - 达到触发价后以限价成交',
    'trailing_stop': '追踪止损单 - 跟踪价格变化的止损'
}
```

## 市价单 (Market Order)

### 概念与特点

```python
# 市价单以当前市场价格立即成交
# 优点：保证成交，速度快
# 缺点：价格不确定，可能有滑点

market_order = {
    'type': 'MARKET',
    'symbol': 'AAPL',
    'quantity': 100,
    'side': 'BUY',  # 或 'SELL'
    'description': '以市场价买入100股AAPL'
}

# 执行逻辑
execute_logic = """
1. 提交订单 → 交易所撮合引擎
2. 按当前最优报价成交
3. 买入：按 ask 价格成交
   卖出：按 bid 价格成交
4. 成交后返回成交报告
"""
```

### 滑点 (Slippage)

```python
# 滑点 = 预期成交价 vs 实际成交价 的差异
# 原因：市场波动、流动性不足、大单冲击

def calculate_slippage(order_price, execution_price, quantity):
    """计算滑点"""
    expected_cost = order_price * quantity
    actual_cost = execution_price * quantity
    slippage = (actual_cost - expected_cost) / expected_cost
    return slippage

# 示例
expected_price = 100.00
execution_price = 100.05
slippage = calculate_slippage(expected_price, execution_price, 100)
print(f"滑点: {slippage:.2%}")  # 0.05%
```

## 限价单 (Limit Order)

### 概念与特点

```python
# 限价单指定成交价格上限（买入）或下限（卖出）
# 优点：价格确定，无滑点风险
# 缺点：可能不成交

limit_order = {
    'type': 'LIMIT',
    'symbol': 'AAPL',
    'quantity': 100,
    'price': 99.50,      # 最高买入价（买入单）或最低卖出价（卖出单）
    'side': 'BUY',
    'time_in_force': 'DAY'  # 当日有效
}

# 限价单规则
limit_rules = """
买入限价单：成交价 <= 指定的limit price
卖出限价单：成交价 >= 指定的limit price

例如：
- 买入限价单 price=99.50 → 可能在99.50或更低成交
- 卖出限价单 price=100.50 → 可能在100.50或更高成交
"""
```

### 有效期选项

```python
# TIME IN FORCE 选项
time_in_force = {
    'DAY': '当日有效，收盘后自动取消',
    'GTC': 'Good Till Cancelled，一直有效直到手动取消',
    'IOC': 'Immediate Or Cancel，不能成交部分立即取消',
    'FOK': 'Fill Or Kill，必须全部成交否则取消'
}
```

## 止损单 (Stop Order)

### 概念与特点

```python
# 止损单是"被动触发"的订单
# 当价格达到触发价时，自动转为市价单或限价单

# 买入止损单：价格上穿触发价时买入（用于止损卖出/做空平仓）
# 卖出止损单：价格下穿触发价时卖出（用于止损）

stop_order = {
    'type': 'STOP',
    'symbol': 'AAPL',
    'quantity': 100,
    'stop_price': 95.00,  # 触发价
    'side': 'SELL',        # 触发后卖出
    'description': '当价格跌到95时，自动卖出100股'
}

# 触发逻辑
trigger_logic = """
止损卖出：
- 持有100股AAPL，成本100
- 设置止损价95
- 当市场价格<=95时，触发卖出
- 作用：限制最大亏损在5%

买入止损（做空平仓）：
- 做空100股AAPL，开仓价100
- 设置买入止损价105
- 当价格>=105时，平仓
- 作用：限制做空损失
"""
```

### 追踪止损 (Trailing Stop)

```python
# 追踪止损是一种动态止损
# 止损价格随市场价格"追踪"调整

trailing_stop = {
    'type': 'TRAILING_STOP',
    'symbol': 'AAPL',
    'quantity': 100,
    'trailing_percent': 0.10,  # 追踪10%
    'side': 'SELL',
    'description': '从最高点回撤10%时止损'
}

# 示例
initial_price = 100
highest_price = 120
current_price = 115

trailing_stop_price = highest_price * (1 - 0.10)  # 108
print(f"当前追踪止损价: {trailing_stop_price}")

# 如果价格继续上涨到125
highest_price = 125
trailing_stop_price = highest_price * (1 - 0.10)  # 112.5
print(f"新的追踪止损价: {trailing_stop_price}")
```

## 止损限价单 (Stop-Limit Order)

```python
# 止损限价单 = 止损触发 + 限价成交
# 触发后以限价成交，而非市价

stop_limit_order = {
    'type': 'STOP_LIMIT',
    'symbol': 'AAPL',
    'quantity': 100,
    'stop_price': 95.00,   # 触发价
    'limit_price': 94.50,  # 限价（触发后以此价格成交）
    'side': 'SELL',
    'description': '价格<=95时触发，触发后以94.50或更高卖出'
}

# 风险
# 如果价格快速下跌，可能无法以limit_price成交
# 这种情况称为"止损穿仓"
```

## 订单有效期 (Time in Force)

```python
# 完整示例
def create_order(order_type, symbol, quantity, side, **kwargs):
    order = {
        'type': order_type,
        'symbol': symbol,
        'quantity': quantity,
        'side': side,
        'time_in_force': kwargs.get('time_in_force', 'DAY')
    }

    if order_type == 'LIMIT':
        order['price'] = kwargs['price']
    elif order_type == 'STOP':
        order['stop_price'] = kwargs['stop_price']
    elif order_type == 'STOP_LIMIT':
        order['stop_price'] = kwargs['stop_price']
        order['limit_price'] = kwargs['limit_price']

    return order

# 创建不同类型订单
market_buy = create_order('MARKET', 'AAPL', 100, 'BUY')
limit_sell = create_order('LIMIT', 'AAPL', 100, 'SELL', price=105.00)
stop_loss = create_order('STOP', 'AAPL', 100, 'SELL', stop_price=95.00)
```

## 实践：订单管理系统

```python
class SimpleOrderManager:
    """简单的订单管理系统"""

    def __init__(self):
        self.orders = []
        self.fills = []

    def place_order(self, order):
        """下单"""
        order['order_id'] = f"ORD{len(self.orders) + 1:04d}"
        order['status'] = 'PENDING'
        order['filled_qty'] = 0
        self.orders.append(order)
        return order['order_id']

    def cancel_order(self, order_id):
        """取消订单"""
        for order in self.orders:
            if order['order_id'] == order_id:
                order['status'] = 'CANCELLED'
                return True
        return False

    def get_order_status(self, order_id):
        """查询订单状态"""
        for order in self.orders:
            if order['order_id'] == order_id:
                return order['status']
        return None

    def simulate_fill(self, order_id, fill_price, fill_qty):
        """模拟成交"""
        for order in self.orders:
            if order['order_id'] == order_id:
                order['filled_qty'] += fill_qty
                if order['filled_qty'] >= order['quantity']:
                    order['status'] = 'FILLED'
                else:
                    order['status'] = 'PARTIALLY_FILLED'

                self.fills.append({
                    'order_id': order_id,
                    'price': fill_price,
                    'quantity': fill_qty
                })
                return True
        return False

# 使用示例
manager = SimpleOrderManager()

# 下单
order_id = manager.place_order({
    'type': 'LIMIT',
    'symbol': 'AAPL',
    'quantity': 100,
    'side': 'BUY',
    'price': 99.50
})
print(f"订单ID: {order_id}")

# 查询状态
print(f"订单状态: {manager.get_order_status(order_id)}")
```
