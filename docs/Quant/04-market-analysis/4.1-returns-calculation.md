# 4.1 收益率计算：对数收益率、简单收益率

## 收益率的重要性

```python
# 收益率是量化交易的核心
# 策略比较、风险评估、组合优化都依赖收益率

import numpy as np
import pandas as pd
import yfinance as yf
```

## 简单收益率 (Simple Return)

### 定义

```python
# 简单收益率 = (P_t - P_{t-1}) / P_{t-1}
# 也称为百分比收益率

def simple_return(prices):
    """计算简单收益率"""
    return prices.pct_change()

# 示例
prices = pd.Series([100, 105, 102, 108, 107])
returns = simple_return(prices)
print("简单收益率:")
print(returns)
# 0       NaN
# 1    0.050
# 2   -0.0286
# 3    0.0588
# 4   -0.0093
```

### 累加性质

```python
# 简单收益率可以累加（但不是线性）
# 最终收益率 ≠ 各期收益率之和

initial = 100
returns_period = [0.05, -0.02, 0.03]  # 5%, -2%, 3%

# 错误计算
wrong_final = initial * (1 + sum(returns_period))
print(f"错误计算: {wrong_final}")  # 106

# 正确计算
correct_final = initial
for r in returns_period:
    correct_final *= (1 + r)
print(f"正确计算: {correct_final}")  # 105.88
```

## 对数收益率 (Log Return)

### 定义

```python
# 对数收益率 = ln(P_t / P_{t-1})
# 也称为连续复利收益率

def log_return(prices):
    """计算对数收益率"""
    return np.log(prices / prices.shift(1))

# 或
def log_return2(prices):
    """计算对数收益率"""
    return np.log(prices).diff()

prices = pd.Series([100, 105, 102, 108, 107])
log_returns = log_return(prices)
print("对数收益率:")
print(log_returns)
# 0         NaN
# 1    0.04879
# 2   -0.02887
# 3    0.05735
# 4   -0.00935
```

### 可累加性质

```python
# 对数收益率可以累加！
# 总对数收益 = 各期对数收益之和

total_log_return = log_returns.sum()
print(f"总对数收益: {total_log_return:.4f}")

# 验证
final_price = prices.iloc[0] * np.exp(total_log_return)
print(f"最终价格验证: {final_price:.2f}")

# 这使得对数收益率在统计分析中更方便
```

## 两种收益率的比较

```python
# 对数收益率 vs 简单收益率
returns_comparison = pd.DataFrame({
    '简单收益率': simple_return(prices),
    '对数收益率': log_return(prices),
    '对数≈简单(小值时)': np.log(1 + simple_return(prices))
})

print(returns_comparison)

# 结论：
# 1. 对于小收益率，两者近似相等
# 2. 对于大收益率，差异显著
# 3. 统计分析用对数收益更方便
# 4. 计算实际盈亏用简单收益更直观
```

## 多周期收益率

```python
# 日收益率 → 周收益率 → 月收益率

df = yf.download('AAPL', start='2023-01-01', end='2023-12-31')
df.columns = df.columns.get_level_values(0)
close = df['close']

# 日收益率
daily_returns = close.pct_change().dropna()

# 周收益率（5天）
weekly_returns = close.pct_change(5).dropna()

# 月收益率（20天）
monthly_returns = close.pct_change(20).dropna()

print(f"日收益均值: {daily_returns.mean():.4f}")
print(f"周收益均值: {weekly_returns.mean():.4f}")
print(f"月收益均值: {monthly_returns.mean():.4f}")
```

## 年化收益率

```python
# 年化收益率 = 平均日收益率 × 252
# 或使用对数收益率

# 方法1：日收益均值年化
annual_return_mean = daily_returns.mean() * 252

# 方法2：对数收益年化
annual_return_log = daily_returns.apply(lambda x: np.log(1+x)).mean() * 252

# 方法3：几何年化
annual_return_geo = (1 + daily_returns).prod() ** (252 / len(daily_returns)) - 1

print(f"年化收益率 (均值法): {annual_return_mean:.2%}")
print(f"年化收益率 (对数法): {annual_return_log:.2%}")
print(f"年化收益率 (几何法): {annual_return_geo:.2%}")
```

## 收益率分布

```python
import matplotlib.pyplot as plt

# 绘制收益率分布
fig, axes = plt.subplots(1, 2, figsize=(14, 5))

# 简单收益率分布
axes[0].hist(daily_returns.dropna(), bins=50, edgecolor='black', alpha=0.7)
axes[0].set_title('Simple Returns Distribution')
axes[0].set_xlabel('Return')
axes[0].set_ylabel('Frequency')
axes[0].axvline(daily_returns.mean(), color='red', linestyle='--', label=f'Mean: {daily_returns.mean():.4f}')
axes[0].legend()

# 对数收益率分布
log_rets = np.log(1 + daily_returns)
axes[1].hist(log_rets.dropna(), bins=50, edgecolor='black', alpha=0.7, color='orange')
axes[1].set_title('Log Returns Distribution')
axes[1].set_xlabel('Log Return')
axes[1].set_ylabel('Frequency')
axes[1].axvline(log_rets.mean(), color='red', linestyle='--', label=f'Mean: {log_rets.mean():.4f}')
axes[1].legend()

plt.tight_layout()
plt.show()
```

## 实战：计算资产组合收益率

```python
# 假设组合配置
portfolio = {
    'AAPL': 0.4,    # 40%
    'GOOGL': 0.3,   # 30%
    'MSFT': 0.3     # 30%
}

# 获取数据
symbols = list(portfolio.keys())
data = yf.download(symbols, start='2023-01-01', end='2023-12-31')['Close']
data.columns = data.columns.get_level_values(0)

# 计算各资产收益率
returns = data.pct_change().dropna()

# 计算组合收益率（每日）
portfolio_returns = pd.Series({
    date: sum(returns.loc[date, symbol] * weight
              for symbol, weight in portfolio.items())
    for date in returns.index
})

print(f"组合日收益均值: {portfolio_returns.mean():.4f}")
print(f"组合年化收益: {portfolio_returns.mean() * 252:.2%}")
print(f"组合波动率: {portfolio_returns.std() * np.sqrt(252):.2%}")
```
