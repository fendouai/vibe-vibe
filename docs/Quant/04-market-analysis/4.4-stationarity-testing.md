# 4.4 平稳性检验：ADF检验

## 平稳性的概念

```python
# 平稳性是时间序列分析的核心概念
# 平稳序列：均值、方差、自协方差不随时间变化

import numpy as np
import pandas as pd
import yfinance as yf
from statsmodels.tsa.stattools import adfuller, kpss
```

## 平稳性类型

```python
# 1. 严格平稳 (Strictly Stationary)
# 所有统计性质完全相同，移位不变

# 2. 弱平稳/宽平稳 (Weakly Stationary / Covariance Stationary)
# 均值恒定
# 方差有限
# 自协方差只依赖于滞后期，不依赖于时间

# 大多数金融时间序列是非平稳的
# 价格序列通常是非平稳（随机游走）
# 收益率序列通常是平稳的
```

## ADF检验 (Augmented Dickey-Fuller)

### 原理

```python
# ADF检验原假设：序列存在单位根（非平稳）
# 备择假设：序列是平稳的

# 如果p值 < 0.05，拒绝原假设，认为序列平稳
```

### Python实现

```python
def adf_test(series, name='Series'):
    """ADF平稳性检验"""
    result = adfuller(series.dropna(), autolag='AIC')

    print(f'ADF检验结果 - {name}:')
    print(f'  ADF统计量: {result[0]:.4f}')
    print(f'  p值: {result[1]:.4f}')
    print(f'  滞后数: {result[2]}')
    print(f'  临界值:')
    for key, value in result[4].items():
        print(f'    {key}: {value:.4f}')

    if result[1] < 0.05:
        print(f'  结论: 拒绝原假设，{name}是平稳的')
    else:
        print(f'  结论: 不能拒绝原假设，{name}是非平稳的')

    return result[1] < 0.05

# 示例：检验价格序列
df = yf.download('AAPL', start='2022-01-01', end='2023-12-31')
df.columns = df.columns.get_level_values(0)

prices = df['close']
returns = prices.pct_change().dropna()

print("=== 价格序列 ===")
adf_test(prices, 'AAPL Price')

print("\n=== 收益率序列 ===")
adf_test(returns, 'AAPL Returns')
```

## KPSS检验

```python
# KPSS检验原假设：序列是平稳的
# 与ADF互补

def kpss_test(series, name='Series'):
    """KPSS平稳性检验"""
    result = kpss(series.dropna(), regression='c', nlags='auto')

    print(f'KPSS检验结果 - {name}:')
    print(f'  KPSS统计量: {result[0]:.4f}')
    print(f'  p值: {result[1]:.4f}')
    print(f'  滞后数: {result[2]}')
    print(f'  临界值:')
    for key, value in result[3].items():
        print(f'    {key}: {value:.4f}')

    if result[1] < 0.05:
        print(f'  结论: 拒绝原假设，{name}是非平稳的')
    else:
        print(f'  结论: 不能拒绝原假设，{name}是平稳的')

    return result[1] >= 0.05

print("=== 价格序列 ===")
kpss_test(prices, 'AAPL Price')

print("\n=== 收益率序列 ===")
kpss_test(returns, 'AAPL Returns')
```

## 综合判断

```python
# ADF和KPSS综合使用
def stationarity_test(series, name='Series'):
    """综合平稳性检验"""
    adf_result = adfuller(series.dropna(), autolag='AIC')
    kpss_result = kpss(series.dropna(), regression='c', nlags='auto')

    is_stationary = adf_result[1] < 0.05 and kpss_result[1] >= 0.05

    conclusion = "平稳" if is_stationary else "非平稳"

    print(f'\n{name}: {conclusion}')
    print(f'  ADF p值: {adf_result[1]:.4f} (平稳需要 < 0.05)')
    print(f'  KPSS p值: {kpss_result[1]:.4f} (平稳需要 >= 0.05)')

    return is_stationary

# 测试多个序列
print("=== 综合检验 ===")
stationarity_test(prices, '价格序列')
stationarity_test(returns, '收益率序列')
```

## 差分与平稳性

```python
# 对非平稳序列进行差分可以使其平稳

# 1阶差分
diff1 = prices.diff().dropna()
print("\n1阶差分:")
stationarity_test(diff1, '1阶差分')

# 2阶差分（收益率通常已经平稳）
diff2 = prices.diff().diff().dropna()
print("\n2阶差分:")
stationarity_test(diff2, '2阶差分')
```

## 实战：协整检验（配对交易基础）

```python
from statsmodels.tsa.stattools import coint

# 协整检验：检验两个序列是否存在长期均衡关系
# 协整的序列对适合配对交易

def cointegration_test(series1, series2, name1='Series1', name2='Series2'):
    """协整检验"""
    score, pvalue, _ = coint(series1, series2)

    print(f'协整检验 {name1} vs {name2}:')
    print(f'  统计量: {score:.4f}')
    print(f'  p值: {pvalue:.4f}')

    if pvalue < 0.05:
        print(f'  结论: 存在协整关系 (p < 0.05)')
    else:
        print(f'  结论: 不存在协整关系 (p >= 0.05)')

    return pvalue < 0.05

# 示例：检验两只股票是否协整
stocks = yf.download(['AAPL', 'MSFT'], start='2022-01-01', end='2023-12-31')['Close']
stocks.columns = stocks.columns.get_level_values(0)

cointegration_test(stocks['AAPL'], stocks['MSFT'], 'AAPL', 'MSFT')
```
