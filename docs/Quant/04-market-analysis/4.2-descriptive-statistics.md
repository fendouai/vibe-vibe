# 4.2 分布统计：均值、方差、偏度、峰度

## 描述性统计基础

```python
import numpy as np
import pandas as pd
import yfinance as yf
from scipy import stats

# 获取收益率
df = yf.download('AAPL', start='2022-01-01', end='2023-12-31')
df.columns = df.columns.get_level_values(0)
returns = df['close'].pct_change().dropna()
```

## 均值 (Mean)

```python
# 算术均值
mean_simple = returns.mean()

# 几何均值（用于计算平均增长率）
# 几何均值 = (∏(1+r_i))^(1/n) - 1
geometric_mean = (1 + returns).prod() ** (1/len(returns)) - 1

print(f"算术均值: {mean_simple:.6f}")
print(f"几何均值: {geometric_mean:.6f}")

# 注意：对于有负收益的情况，几何均值 < 算术均值
# 这是因为复利效应
```

## 方差与标准差

```python
# 方差 (Variance) - 收益率离散程度的度量
variance = returns.var()

# 标准差 (Standard Deviation) - 与均值同单位，更直观
std = returns.std()

# 年化
annual_std = std * np.sqrt(252)

print(f"日方差: {variance:.6f}")
print(f"日标准差: {std:.6f}")
print(f"年化标准差: {annual_std:.2%}")

# 注意：年化标准差 ≈ 年化波动率
```

## 分位数

```python
# 分位数 - 收益率的分布位置
quantiles = returns.quantile([0.01, 0.05, 0.25, 0.50, 0.75, 0.95, 0.99])
print("分位数:")
print(quantiles)

# VaR计算（Value at Risk）
var_95 = returns.quantile(0.05)  # 95%置信度
var_99 = returns.quantile(0.01)  # 99%置信度
print(f"\nVaR (95%): {var_95:.2%}")
print(f"VaR (99%): {var_99:.2%}")
```

## 偏度 (Skewness)

```python
# 偏度 - 分布不对称程度
# 正偏度：右尾长（上涨概率更大，但幅度小）
# 负偏度：左尾长（下跌概率更大，但幅度大）

skewness = stats.skew(returns)
print(f"偏度: {skewness:.4f}")

# 解读
if skewness > 0:
    print("正偏度：右尾长，存在大幅上涨的可能性")
elif skewness < 0:
    print("负偏度：左尾长，存在大幅下跌的风险")
else:
    print("对称分布")
```

## 峰度 (Kurtosis)

```python
# 峰度 - 分布尖峰和肥尾程度
# 超额峰度 = 峰度 - 3
# 正峰度：尖峰肥尾（"黑天鹅"事件更多）

kurtosis = stats.kurtosis(returns)  # 超额峰度
print(f"超额峰度: {kurtosis:.4f}")

# 解读
if kurtosis > 0:
    print("尖峰肥尾：极端事件比正态分布更频繁")
else:
    print("平峰薄尾：极端事件比正态分布更少")
```

## 正态性检验

```python
# Jarque-Bera 检验
jb_stat, jb_pvalue = stats.jarque_bera(returns)
print(f"JB统计量: {jb_stat:.4f}")
print(f"JB p值: {jb_pvalue:.4f}")

if jb_pvalue < 0.05:
    print("拒绝正态分布假设：收益率不服从正态分布")
else:
    print("不能拒绝正态分布假设")

# Shapiro-Wilk 检验（对小样本更准确）
if len(returns) <= 5000:
    stat, pvalue = stats.shapiro(returns.sample(min(5000, len(returns))))
    print(f"Shapiro p值: {pvalue:.4f}")
```

## 可视化分布

```python
import matplotlib.pyplot as plt
from scipy.stats import norm

fig, axes = plt.subplots(2, 2, figsize=(14, 10))

# 1. 直方图 vs 正态分布
ax = axes[0, 0]
ax.hist(returns, bins=50, density=True, alpha=0.7, edgecolor='black')
x = np.linspace(returns.min(), returns.max(), 100)
ax.plot(x, norm.pdf(x, returns.mean(), returns.std()), 'r-', lw=2, label='Normal')
ax.set_title('Return Distribution vs Normal')
ax.legend()

# 2. Q-Q图
ax = axes[0, 1]
stats.probplot(returns, dist="norm", plot=ax)
ax.set_title('Q-Q Plot (Normal)')

# 3. 箱线图
ax = axes[1, 0]
ax.boxplot(returns)
ax.set_title('Return Box Plot')
ax.set_ylabel('Return')

# 4. 累积分布函数
ax = axes[1, 1]
sorted_returns = np.sort(returns)
cdf = np.arange(1, len(sorted_returns) + 1) / len(sorted_returns)
ax.plot(sorted_returns, cdf)
ax.set_title('Cumulative Distribution Function')
ax.set_xlabel('Return')
ax.set_ylabel('Cumulative Probability')

plt.tight_layout()
plt.show()
```

## 收益率分布特征总结

```python
def return_characteristics(returns):
    """计算收益率的完整统计特征"""
    return pd.Series({
        'Count': len(returns),
        'Mean': returns.mean(),
        'Std': returns.std(),
        'Min': returns.min(),
        'Max': returns.max(),
        'Skewness': stats.skew(returns),
        'Kurtosis': stats.kurtosis(returns),
        'VaR_95': returns.quantile(0.05),
        'VaR_99': returns.quantile(0.01),
        'CVaR_95': returns[returns <= returns.quantile(0.05)].mean(),
        'Annual_Return': returns.mean() * 252,
        'Annual_Vol': returns.std() * np.sqrt(252),
        'Sharpe': (returns.mean() * 252) / (returns.std() * np.sqrt(252))
    })

char = return_characteristics(returns)
print("收益率统计特征:")
print(char.round(4))
```
