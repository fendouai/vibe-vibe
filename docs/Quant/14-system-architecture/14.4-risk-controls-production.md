# 14.4 生产风控框架

## 生产风控的重要性

真实的量化交易系统面临各种风险：

| 风险类型   | 描述               | 后果       |
| ---------- | ------------------ | ---------- |
| 市场风险   | 价格不利变动       | 亏损       |
| 流动性风险 | 无法以合理价格成交 | 滑点增大   |
| 操作风险   | 人为错误           | 错误下单   |
| 技术风险   | 系统故障           | 交易中断   |
| 模型风险   | 模型失效           | 亏损       |
| 尾部风险   | 黑天鹅事件         | 灾难性亏损 |

## 风控架构

## Pre-Trade风控

```python
class PreTradeRiskManager:
    """下单前风控"""

    def __init__(self, config: dict):
        self.config = config
        self.logger = setup_logger('risk', 'logs/risk.log')

        # 风控参数
        self.max_position_size = config.get('max_position_size', 0.1)  # 最大仓位比例
        self.max_single_order = config.get('max_single_order', 100000)  # 单笔最大金额
        self.max_daily_loss = config.get('max_daily_loss', 0.05)  # 日最大亏损
        self.max_leverage = config.get('max_leverage', 2.0)  # 最大杠杆
        self.min_cash_ratio = config.get('min_cash_ratio', 0.1)  # 最小现金比例

        # 状态
        self.daily_pnl = 0.0
        self.daily_trades = 0
        self.circuit_breaker_triggered = False

    def check_order(self, order: Order, portfolio: dict) -> tuple[bool, str]:
        """
        检查订单是否通过风控

        Returns:
            (is_approved, rejection_reason)
        """
        # 1. 检查熔断机制
        if self.circuit_breaker_triggered:
            return False, "Circuit breaker triggered - trading disabled"

        # 2. 检查日亏损限制
        if self.daily_pnl < -self.max_daily_loss * portfolio['total_value']:
            self.circuit_breaker_triggered = True
            self.logger.critical(f"Daily loss limit reached: {self.daily_pnl:.2f}")
            return False, f"Daily loss limit reached: {self.daily_pnl:.2f}"

        # 3. 检查仓位限制
        position_value = portfolio['positions'].get(order.symbol, 0) * order.price
        total_value = portfolio['total_value']

        if order.side == 'buy':
            new_position_value = position_value + order.quantity * order.price
        else:
            new_position_value = abs(position_value - order.quantity * order.price)

        position_ratio = new_position_value / total_value
        if position_ratio > self.max_position_size:
            return False, f"Position size exceeded: {position_ratio:.2%} > {self.max_position_size:.2%}"

        # 4. 检查单笔订单金额
        order_value = order.quantity * order.price
        if order_value > self.max_single_order:
            return False, f"Single order value exceeded: {order_value:.2f} > {self.max_single_order}"

        # 5. 检查现金比例
        if order.side == 'buy':
            required_cash = order_value * (1 + 0.001)  # 考虑手续费
            if portfolio['cash'] < required_cash:
                return False, f"Insufficient cash: {portfolio['cash']:.2f} < {required_cash:.2f}"

            new_cash_ratio = (portfolio['cash'] - required_cash) / total_value
            if new_cash_ratio < self.min_cash_ratio:
                return False, f"Min cash ratio violated: {new_cash_ratio:.2%} < {self.min_cash_ratio:.2%}"

        # 6. 检查杠杆
        total_exposure = sum(abs(v) for v in portfolio['positions'].values()) * portfolio.get('avg_price', 0)
        leverage = total_exposure / total_value
        if leverage > self.max_leverage:
            return False, f"Leverage exceeded: {leverage:.2f}x > {self.max_leverage}x"

        # 7. 检查交易频率
        if self.daily_trades >= 100:  # 假设日最大交易次数100
            return False, f"Daily trade limit reached: {self.daily_trades}"

        # 8. 价格合理性检查
        if order.price:
            deviation = abs(order.price - portfolio.get('current_price', {}).get(order.symbol, order.price)) / order.price
            if deviation > 0.05:  # 偏离超过5%
                self.logger.warning(f"Large price deviation for {order.symbol}: {deviation:.2%}")

        return True, "Approved"

    def update_daily_pnl(self, pnl: float):
        """更新日盈亏"""
        self.daily_pnl += pnl

    def increment_trade_count(self):
        """增加交易计数"""
        self.daily_trades += 1

    def reset_daily(self):
        """重置日度状态（每个交易日开始时调用）"""
        self.daily_pnl = 0.0
        self.daily_trades = 0
        self.circuit_breaker_triggered = False
```

## Post-Trade监控

```python
class PostTradeMonitor:
    """下单后监控"""

    def __init__(self, risk_manager: PreTradeRiskManager):
        self.risk_manager = risk_manager
        self.logger = setup_logger('post_trade', 'logs/post_trade.log')
        self.pending_orders = {}
        self.filled_orders = {}

    def on_order_submitted(self, order: Order):
        """订单提交事件"""
        self.pending_orders[order.order_id] = {
            'order': order,
            'submitted_at': datetime.now(),
            'status': 'submitted'
        }
        self.logger.info(f"Order submitted: {order.order_id} {order.symbol} {order.side} {order.quantity}")

    def on_order_filled(self, order_id: str, fill_price: float, fill_quantity: float):
        """订单成交事件"""
        if order_id not in self.pending_orders:
            self.logger.warning(f"Filled order not in pending: {order_id}")
            return

        order_info = self.pending_orders.pop(order_id)
        order = order_info['order']

        # 记录成交
        self.filled_orders[order_id] = {
            'order': order,
            'fill_price': fill_price,
            'fill_quantity': fill_quantity,
            'filled_at': datetime.now()
        }

        # 更新风控状态
        pnl = 0
        if order.side == 'sell':
            pnl = (fill_price - order.price) * fill_quantity
        self.risk_manager.update_daily_pnl(pnl)
        self.risk_manager.increment_trade_count()

        self.logger.info(f"Order filled: {order_id} {order.symbol} @ {fill_price}")

        # 检查是否需要告警
        self._check_alerts(order, fill_price)

    def on_order_rejected(self, order_id: str, reason: str):
        """订单被拒绝事件"""
        if order_id in self.pending_orders:
            self.pending_orders[order_id]['status'] = 'rejected'
            self.pending_orders[order_id]['rejection_reason'] = reason

        self.logger.warning(f"Order rejected: {order_id} - {reason}")

    def _check_alerts(self, order: Order, fill_price: float):
        """检查告警条件"""
        # 成交价格偏离订单价格
        if order.price:
            price_deviation = abs(fill_price - order.price) / order.price
            if price_deviation > 0.01:  # 超过1%
                self.logger.warning(
                    f"Large slippage: {order.symbol} "
                    f"expected {order.price} filled @ {fill_price} "
                    f"(deviation: {price_deviation:.2%})"
                )

        # 大额订单
        order_value = order.quantity * fill_price
        if order_value > 100000:
            self.logger.info(f"Large order executed: {order.symbol} {order_value:.2f}")

    def get_pending_orders(self) -> list:
        """获取待成交订单"""
        return list(self.pending_orders.values())

    def get_today_fills(self) -> list:
        """获取今日成交"""
        return list(self.filled_orders.values())
```

## 仓位限制

```python
class PositionLimits:
    """仓位限制管理器"""

    def __init__(self):
        # 各标的的最大仓位
        self.max_positions = {}  # symbol -> max_shares
        self.max_position_pct = 0.1  # 单个标的最大占比10%

        # 全局仓位限制
        self.max_total_long = 1.0  # 最大做多比例
        self.max_total_short = 0.3  # 最大做空比例
        self.max_net_exposure = 0.8  # 最大净暴露

    def check_position_limits(self, symbol: str, new_position: float,
                             total_portfolio_value: float) -> tuple[bool, str]:
        """检查仓位限制"""
        # 检查绝对仓位
        if symbol in self.max_positions:
            if abs(new_position) > self.max_positions[symbol]:
                return False, f"Absolute position limit exceeded for {symbol}"

        # 检查仓位占比
        position_value = abs(new_position)
        position_pct = position_value / total_portfolio_value
        if position_pct > self.max_position_pct:
            return False, f"Position percentage exceeded: {position_pct:.2%} > {self.max_position_pct:.2%}"

        return True, "OK"

    def check_exposure_limits(self, positions: dict, total_value: float) -> tuple[bool, str]:
        """检查暴露度限制"""
        total_long = 0.0
        total_short = 0.0

        for symbol, position in positions.items():
            if position > 0:
                total_long += position
            elif position < 0:
                total_short += abs(position)

        long_pct = total_long / total_value
        short_pct = total_short / total_value
        net_exposure = (total_long - total_short) / total_value

        if long_pct > self.max_total_long:
            return False, f"Long exposure exceeded: {long_pct:.2%} > {self.max_total_long:.2%}"

        if short_pct > self.max_total_short:
            return False, f"Short exposure exceeded: {short_pct:.2%} > {self.max_total_short:.2%}"

        if abs(net_exposure) > self.max_net_exposure:
            return False, f"Net exposure exceeded: {net_exposure:.2%} > {self.max_net_exposure:.2%}"

        return True, "OK"
```

## 紧急止损机制

```python
class EmergencyStop:
    """紧急止损机制"""

    def __init__(self, broker, risk_manager: PreTradeRiskManager):
        self.broker = broker
        self.risk_manager = risk_manager
        self.logger = setup_logger('emergency', 'logs/emergency.log')
        self.emergency_stopped = False

    def trigger_emergency_stop(self, reason: str):
        """触发紧急止损"""
        if self.emergency_stopped:
            return

        self.emergency_stopped = True
        self.logger.critical(f"EMERGENCY STOP TRIGGERED: {reason}")

        # 1. 取消所有待成交订单
        self._cancel_all_pending_orders()

        # 2. 平掉所有仓位
        self._close_all_positions()

        # 3. 发送告警
        self._send_alert(reason)

    def _cancel_all_pending_orders(self):
        """取消所有待成交订单"""
        try:
            pending = self.risk_manager.post_trade_monitor.get_pending_orders()
            for order_info in pending:
                order_id = order_info['order'].order_id
                try:
                    self.broker.cancel_order(order_id)
                    self.logger.info(f"Cancelled order: {order_id}")
                except Exception as e:
                    self.logger.error(f"Failed to cancel order {order_id}: {e}")
        except Exception as e:
            self.logger.error(f"Error cancelling pending orders: {e}")

    def _close_all_positions(self):
        """平掉所有仓位"""
        try:
            positions = self.broker.get_positions()
            for position in positions:
                symbol = position['symbol']
                quantity = position['quantity']

                if quantity > 0:
                    # 卖出平多
                    self.broker.place_order(symbol, quantity, 'sell', 'market')
                    self.logger.info(f"Closing long position: {symbol} {quantity}")
                elif quantity < 0:
                    # 买入平空
                    self.broker.place_order(symbol, abs(quantity), 'buy', 'market')
                    self.logger.info(f"Closing short position: {symbol} {abs(quantity)}")
        except Exception as e:
            self.logger.error(f"Error closing positions: {e}")

    def _send_alert(self, reason: str):
        """发送告警"""
        # 可以集成邮件、短信、Slack等通知
        message = f"EMERGENCY STOP: {reason}"
        print(f"🚨 ALERT: {message}")
        # email_client.send("emergency@example.com", "EMERGENCY STOP", message)

    def check_stop_conditions(self, portfolio_value: float, initial_value: float):
        """检查是否需要触发紧急止损"""
        drawdown = (initial_value - portfolio_value) / initial_value

        # 回撤超过20%
        if drawdown > 0.20:
            self.trigger_emergency_stop(f"Drawdown {drawdown:.2%} > 20%")

        # 回撤超过15%（警告）
        elif drawdown > 0.15:
            self.logger.warning(f"Large drawdown warning: {drawdown:.2%}")
```

## 完整的生产风控系统

```python
class ProductionRiskSystem:
    """完整的生产风控系统"""

    def __init__(self, config: dict, broker):
        self.config = config
        self.broker = broker

        # 初始化各组件
        self.pre_trade = PreTradeRiskManager(config['risk'])
        self.post_trade = PostTradeMonitor(self.pre_trade)
        self.position_limits = PositionLimits()
        self.emergency_stop = EmergencyStop(broker, self.pre_trade)

        # 注册broker事件回调
        self.broker.on_order_filled = self.post_trade.on_order_filled
        self.broker.on_order_rejected = self.post_trade.on_order_rejected

    def check_order(self, order: Order, portfolio: dict) -> tuple[bool, str]:
        """下单前检查"""
        # Pre-trade检查
        approved, reason = self.pre_trade.check_order(order, portfolio)
        if not approved:
            self.pre_trade.logger.warning(f"Order rejected: {reason}")
            return False, reason

        # 仓位限制检查
        approved, reason = self.position_limits.check_position_limits(
            order.symbol,
            portfolio['positions'].get(order.symbol, 0),
            portfolio['total_value']
        )
        if not approved:
            return False, reason

        return True, "Approved"

    def submit_order(self, order: Order, portfolio: dict) -> str:
        """提交订单"""
        # 风控检查
        approved, reason = self.check_order(order, portfolio)
        if not approved:
            raise OrderRejectError(f"Order rejected by risk: {reason}")

        # 提交订单
        order_id = self.broker.submit_order(order)

        # 记录
        self.post_trade.on_order_submitted(order)

        return order_id

    def run_end_of_day(self, portfolio_value: float, initial_value: float):
        """日终处理"""
        # 检查紧急止损条件
        self.emergency_stop.check_stop_conditions(portfolio_value, initial_value)

        # 重置日度状态
        self.pre_trade.reset_daily()
```

## 总结

| 风控类型   | 检查时机 | 关键指标         |
| ---------- | -------- | ---------------- |
| Pre-Trade  | 下单前   | 仓位、亏损、杠杆 |
| Post-Trade | 下单后   | 滑点、成交率     |
| Position   | 实时     | 各标的仓位占比   |
| Exposure   | 实时     | 多空暴露度       |
| Emergency  | 阈值触发 | 回撤、异常       |
