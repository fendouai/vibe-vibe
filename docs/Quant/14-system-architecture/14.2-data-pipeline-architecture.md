# 14.2 数据管道架构

## 数据管道的组成

量化交易系统离不开完善的数据管道：

```
数据源 → 采集 → 清洗 → 存储 → 处理 → 特征 → 模型 → 策略
```

### 数据工厂

```python
class DataSourceFactory:
    """数据源工厂"""

    _sources = {}

    @classmethod
    def register(cls, name: str, source_class: type):
        cls._sources[name] = source_class

    @classmethod
    def create(cls, name: str, **kwargs) -> MarketDataSource:
        if name not in cls._sources:
            raise ValueError(f"Unknown data source: {name}")
        return cls._sources[name](**kwargs)

# 注册数据源
DataSourceFactory.register('alpaca', AlpacaDataSource)
```

## 数据清洗

```python
def clean_market_data(df: pd.DataFrame) -> pd.DataFrame:
    """清洗市场数据"""
    df = df.copy()

    # 1. 去除重复行
    df = df.drop_duplicates()

    # 2. 去除缺失值
    # 向前填充缺失的开盘/收盘价
    df['open'] = df['open'].ffill()
    df['close'] = df['close'].ffill()
    df['high'] = df['high'].ffill()
    df['low'] = df['low'].ffill()
    df['volume'] = df['volume'].fillna(0)

    # 3. 处理异常值
    # 剔除价格为0或负数的行
    df = df[df['close'] > 0]

    # 剔除成交量为0的行
    df = df[df['volume'] >= 0]

    # 4. 验证OHLC关系
    # 最高价 >= 最低价
    # 最高价 >= 开盘价、收盘价
    # 最低价 <= 开盘价、收盘价
    valid = (
        (df['high'] >= df['low']) &
        (df['high'] >= df['open']) &
        (df['high'] >= df['close']) &
        (df['low'] <= df['open']) &
        (df['low'] <= df['close'])
    )
    df = df[valid]

    # 5. 按时间排序
    if 'timestamp' in df.columns:
        df = df.sort_values('timestamp')

    return df

def validate_data_quality(df: pd.DataFrame) -> dict:
    """数据质量检查"""
    report = {
        'total_rows': len(df),
        'missing_values': df.isnull().sum().to_dict(),
        'zero_prices': (df['close'] == 0).sum(),
        'negative_prices': (df['close'] < 0).sum(),
        'invalid_ohlc': ((df['high'] < df['low'])).sum(),
        'duplicates': df.duplicated().sum()
    }
    return report
```

```python
import timescaledb  # PostgreSQL + TimescaleDB
# 或者使用 InfluxDB

class TimeSeriesDB:
    """时间序列数据库"""

    def __init__(self, connection_string: str):
        self.conn = psycopg2.connect(connection_string)

    def write_bars(self, symbol: str, bars: pd.DataFrame):
        """写入K线数据"""
        bars = bars.copy()
        bars['symbol'] = symbol
        bars['time'] = bars.index

        # 使用UPSERT
        query = """
        INSERT INTO market_bars (symbol, time, open, high, low, close, volume)
        VALUES (%(symbol)s, %(time)s, %(open)s, %(high)s, %(low)s, %(close)s, %(volume)s)
        ON CONFLICT (symbol, time) DO UPDATE SET
            open = EXCLUDED.open,
            high = EXCLUDED.high,
            low = EXCLUDED.low,
            close = EXCLUDED.close,
            volume = EXCLUDED.volume
        """

        with self.conn.cursor() as cur:
            for _, row in bars.iterrows():
                cur.execute(query, row.to_dict())
        self.conn.commit()

    def read_bars(self, symbol: str, start: datetime, end: datetime) -> pd.DataFrame:
        """读取K线数据"""
        query = """
        SELECT * FROM market_bars
        WHERE symbol = %s AND time BETWEEN %s AND %s
        ORDER BY time
        """
        return pd.read_sql(query, self.conn, params=(symbol, start, end))
```

### Redis缓存

```python
import redis
import json

class DataCache:
    """数据缓存"""

    def __init__(self, host='localhost', port=6379, ttl=300):
        self.redis = redis.Redis(host=host, port=port, decode_responses=True)
        self.ttl = ttl  # 默认TTL 5分钟

    def set(self, key: str, value: any, ttl: int = None):
        """写入缓存"""
        ttl = ttl or self.ttl
        self.redis.setex(key, ttl, json.dumps(value))

    def get(self, key: str) -> any:
        """读取缓存"""
        data = self.redis.get(key)
        if data:
            return json.loads(data)
        return None

    def delete(self, key: str):
        """删除缓存"""
        self.redis.delete(key)

    def get_market_data(self, symbol: str) -> dict:
        """获取市场数据缓存"""
        return self.get(f"market:{symbol}")

    def set_market_data(self, symbol: str, data: dict, ttl: int = 5):
        """设置市场数据缓存（5秒TTL）"""
        self.set(f"market:{symbol}", data, ttl)

# 使用
cache = DataCache()
cache.set_market_data('AAPL', {'price': 175.5, 'volume': 1000000})
data = cache.get_market_data('AAPL')
```

## 数据处理管道

```python
from typing import Callable
from dataclasses import dataclass

@dataclass
class PipelineConfig:
    """管道配置"""
    name: str
    batch_size: int = 100
    max_retries: int = 3
    timeout: int = 60

class DataPipeline:
    """数据处理管道"""

    def __init__(self, config: PipelineConfig):
        self.config = config
        self.steps = []
        self.logger = setup_logger('pipeline', 'logs/pipeline.log')

    def add_step(self, name: str, processor: Callable, **kwargs):
        """添加处理步骤"""
        self.steps.append({
            'name': name,
            'processor': processor,
            'params': kwargs
        })
        return self

    def run(self, data: any) -> any:
        """运行管道"""
        result = data

        for step in self.steps:
            self.logger.info(f"执行步骤: {step['name']}")
            try:
                result = step['processor'](result, **step['params'])
            except Exception as e:
                self.logger.error(f"步骤 {step['name']} 失败: {e}")
                raise

        return result

# 创建管道
```

## 实时数据流

```python
import asyncio
from collections import deque

class MarketDataStream:
    """市场数据流"""

    def __init__(self, data_source: MarketDataSource):
        self.source = data_source
        self.subscribers = {}
        self.buffer = deque(maxlen=1000)  # 缓冲区
        self.running = False

    def subscribe(self, symbol: str, callback: Callable):
        """订阅数据"""
        if symbol not in self.subscribers:
            self.subscribers[symbol] = []
        self.subscribers[symbol].append(callback)

    async def start(self):
        """启动数据流"""
        self.running = True
        while self.running:
            for symbol in self.subscribers.keys():
                try:
                    data = self.source.get_realtime(symbol)
                    self.buffer.append(data)

                    # 推送给订阅者
                    for callback in self.subscribers[symbol]:
                        await callback(data)
                except Exception as e:
                    print(f"获取数据失败 {symbol}: {e}")

            await asyncio.sleep(1)  # 1秒更新

    def stop(self):
        """停止数据流"""
        self.running = False

# 使用
async def on_data(data):
    print(f"收到数据: {data}")

stream = MarketDataStream(data_source)
stream.subscribe('AAPL', on_data)
```

## 特征存储

```python
class FeatureStore:
    """特征存储"""

    def __init__(self, db_connection):
        self.db = db_connection

    def compute_features(self, df: pd.DataFrame) -> pd.DataFrame:
        """计算特征"""
        features = pd.DataFrame(index=df.index)

        # 技术指标
        features['returns'] = df['close'].pct_change()
        features['sma_20'] = df['close'].rolling(20).mean()
        features['volatility_20'] = df['close'].pct_change().rolling(20).std()
        features['rsi_14'] = calculate_rsi(df['close'], 14)

        # 收益率特征
        for i in [1, 5, 10, 20]:
            features[f'return_{i}d'] = df['close'].pct_change(i)

        return features

    def store_features(self, symbol: str, features: pd.DataFrame):
        """存储特征到数据库"""
        features = features.copy()
        features['symbol'] = symbol
        features['timestamp'] = features.index

        # 存储到特征表
        features.to_sql('features', self.db, if_exists='append', index=False)

    def get_latest_features(self, symbol: str, n: int = 1) -> pd.DataFrame:
        """获取最新特征"""
        query = f"""
        SELECT * FROM (
            SELECT *, ROW_NUMBER() OVER (ORDER BY timestamp DESC) as rn
            FROM features WHERE symbol = '{symbol}'
        ) t WHERE rn <= {n}
        """
        return pd.read_sql(query, self.db)
```

## 总结

数据管道是量化系统的基础设施：

| 组件   | 功能         | 技术选型             |
| ------ | ------------ | -------------------- |
| 数据源 | 采集市场数据 | Alpaca, Binance API  |
| 缓存   | 加速数据访问 | Redis                |
| 清洗   | 数据质量控制 | Pandas               |
| 存储   | 持久化存储   | PostgreSQL, InfluxDB |
| 特征   | 特征工程     | 自定义, TA-Lib       |
| 流处理 | 实时处理     | Asyncio              |
