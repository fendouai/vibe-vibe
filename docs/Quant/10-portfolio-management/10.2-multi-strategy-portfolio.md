# 10.2 多策略组合

## 为什么需要多策略组合？

单一策略面临的问题：

- **策略失效**: 市场风格切换时，依赖单一因子的策略可能亏损
- **集中风险**: 单一策略可能有明显的尾部风险
- **容量限制**: 大资金无法全仓投入单一策略
- **收益不稳定**: 单一策略在不同市场环境下表现差异大

多策略组合的优势：

- **分散风险**: 不同策略在不同市场环境下互补
- **收益平滑**: 高波动策略与低波动策略组合
- **容量增加**: 各策略独立运行，子策略容量叠加
- **更稳定的alpha**: 多个alpha源组合后更可靠

## 相关性阈值处理

```python
def remove_highly_correlated(strategies, threshold=0.7):
    """移除高度相关的策略，保留夏普比率最高的"""
    corr_matrix = strategies.corr()

    # 找到高度相关的策略对
    to_remove = set()
    strategy_sharpes = {}

    for col in strategies.columns:
        strategy_sharpes[col] = (
            strategies[col].mean() / strategies[col].std() * np.sqrt(252)
        )

    for i in range(len(corr_matrix.columns)):
        for j in range(i+1, len(corr_matrix.columns)):
            if abs(corr_matrix.iloc[i, j]) > threshold:
                # 保留夏普比率更高的
                if strategy_sharpes[corr_matrix.columns[i]] > \
                   strategy_sharpes[corr_matrix.columns[j]]:
                    to_remove.add(corr_matrix.columns[j])
                else:
                    to_remove.add(corr_matrix.columns[i])

    return strategies.drop(columns=list(to_remove)), list(to_remove)

selected_strategies, removed = remove_highly_correlated(strategies, threshold=0.7)
print(f"\n移除的策略: {removed}")
print(f"保留的策略: {list(selected_strategies.columns)}")
```

## 策略权重分配方法

### 1. 等权重分配 (Equal Weight)

最简单的分配方式，每个策略权重相同。

```python
def equal_weightAllocation(strategies):
    """等权重分配"""
    n_strategies = len(strategies.columns)
    weights = np.ones(n_strategies) / n_strategies
    return weights

weights = equal_weightAllocation(strategies)
print("等权重分配:", dict(zip(strategies.columns, weights)))
```

### 2. 风险平价 (Risk Parity)

每个策略对组合总风险的贡献相等。

```python
def risk_parity_allocation(strategies):
    """风险平价分配"""
    cov_matrix = strategies.cov() * 252  # 年化协方差

    # 简化版本：使用波动率的倒数
    volatilities = strategies.std() * np.sqrt(252)
    weights = 1 / volatilities
    weights /= weights.sum()

    return weights

rp_weights = risk_parity_allocation(strategies)
print("风险平价分配:", dict(zip(strategies.columns, rp_weights.round(4))))
```

### 3. 基于历史表现的动态权重

```python
def historical_performance_weighting(strategies, lookback=252, min_weight=0.05, max_weight=0.4):
    """基于历史表现的动态权重"""
    # 计算历史夏普比率
    sharpes = {}
    for col in strategies.columns:
        mean_ret = strategies[col].iloc[-lookback:].mean() * 252
        std_ret = strategies[col].iloc[-lookback:].std() * np.sqrt(252)
        sharpes[col] = mean_ret / std_ret if std_ret > 0 else 0

    # 转换为权重（使用softmax风格的归一化）
    sharpe_values = np.array(list(sharpes.values()))
    exp_sharpes = np.exp(sharpe_values - np.max(sharpe_values))  # 数值稳定性
    raw_weights = exp_sharpes / exp_sharpes.sum()

    # 应用权重限制
    weights = np.clip(raw_weights, min_weight, max_weight)
    weights /= weights.sum()  # 重新归一化

    return dict(zip(strategies.columns, weights))

dynamic_weights = historical_performance_weighting(strategies)
print("动态权重分配:", {k: f"{v:.3f}" for k, v in dynamic_weights.items()})
```

## 组合业绩评估

```python
def evaluate_portfolio(strategies, weights):
    """评估组合表现"""
    weights = np.array([weights[col] for col in strategies.columns])

    # 组合收益
    portfolio_returns = (strategies * weights).sum(axis=1)

    # 统计指标
    total_return = (1 + portfolio_returns).prod() - 1
    annual_return = portfolio_returns.mean() * 252
    annual_volatility = portfolio_returns.std() * np.sqrt(252)
    sharpe_ratio = annual_return / annual_volatility if annual_volatility > 0 else 0
    max_drawdown = (portfolio_returns.cumsum() -
                    portfolio_returns.cumsum().cummax()).min()

    # 最大回撤持续时间
    cumulative = portfolio_returns.cumsum()
    running_max = cumulative.cummax()
    drawdown = cumulative - running_max

    print(f"组合表现评估:")
    print(f"  总收益: {total_return:.2%}")
    print(f"  年化收益: {annual_return:.2%}")
    print(f"  年化波动率: {annual_volatility:.2%}")
    print(f"  夏普比率: {sharpe_ratio:.3f}")
    print(f"  最大回撤: {max_drawdown:.2%}")

    return portfolio_returns

# 比较不同分配方法
print("\n=== 等权重组合 ===")
equal_returns = evaluate_portfolio(strategies, equal_weightAllocation(strategies))

print("\n=== 风险平价组合 ===")

print("\n=== 动态权重组合 ===")
```

## 策略仓位计算

```python
def calculate_strategy_positions(capital, strategies_weights, strategy_returns):
    """计算每个策略的实际仓位"""
    total_capital = capital
    positions = {}

    for strategy_name in strategies_weights.keys():
        allocation = total_capital * strategies_weights[strategy_name]

        # 计算需要多少资金来匹配策略的收益波动
        strategy_vol = strategy_returns[strategy_name].std()
        if strategy_vol > 0:
            # 目标波动率（假设每个策略年化波动率10%）
            target_vol = 0.10 / np.sqrt(252)
            leverage = strategy_vol / target_vol
        else:
            leverage = 1.0

        positions[strategy_name] = {
            'allocation': allocation,
            'leverage': leverage,
            'notional': allocation * leverage
        }

    return positions

# 示例
capital = 1_000_000  # 100万初始资金
positions = calculate_strategy_positions(capital, dict(zip(strategies.columns, rp_weights)), strategies)

print("\n策略仓位分配:")
for name, pos in positions.items():
    print(f"  {name}:")
    print(f"    分配资金: ¥{pos['allocation']:,.0f}")
    print(f"    杠杆: {pos['leverage']:.2f}x")
    print(f"    名义金额: ¥{pos['notional']:,.0f}")
```

## 实际组合管理框架

```python
class MultiStrategyPortfolio:
    def __init__(self, initial_capital, max_weight=0.4, min_weight=0.05):
        self.capital = initial_capital
        self.max_weight = max_weight
        self.min_weight = min_weight
        self.strategies = {}
        self.weights = {}

    def add_strategy(self, name, returns, target_vol=0.10):
        """添加策略到组合"""
        self.strategies[name] = {
            'returns': returns,
            'target_vol': target_vol
        }

    def calculate_weights(self, method='risk_parity'):
        """计算策略权重"""
        if method == 'equal':
            n = len(self.strategies)
            self.weights = {name: 1/n for name in self.strategies.keys()}

        elif method == 'risk_parity':
            volatilities = {
                name: data['returns'].std() * np.sqrt(252)
                for name, data in self.strategies.items()
            }
            inv_vols = {name: 1/vol for name, vol in volatilities.items()}
            total = sum(inv_vols.values())
            self.weights = {name: vol/total for name, vol in inv_vols.items()}

        elif method == 'sharpe_weighted':
            sharpes = {}
            for name, data in self.strategies.items():
                mean_ret = data['returns'].mean() * 252
                vol = data['returns'].std() * np.sqrt(252)
                sharpes[name] = mean_ret / vol if vol > 0 else 0

            # 归一化并应用权重限制
            total_sharpe = sum(max(s, 0) for s in sharpes.values())
            raw_weights = {name: max(s, 0)/total_sharpe for name, s in sharpes.items()}

            self.weights = {}
            for name, w in raw_weights.items():
                self.weights[name] = np.clip(w, self.min_weight, self.max_weight)

            # 重新归一化
            total = sum(self.weights.values())
            self.weights = {name: w/total for name, w in self.weights.items()}

        return self.weights

    def get_portfolio_returns(self):
        """获取组合收益"""
        if not self.weights:
            raise ValueError("先计算权重")

        result = pd.Series(0, index=list(self.strategies.values())[0]['returns'].index)

        for name, weight in self.weights.items():
            result += self.strategies[name]['returns'] * weight

        return result

    def rebalance(self, method='risk_parity'):
        """再平衡组合"""
        self.calculate_weights(method)
        return self.weights

# 使用示例
portfolio = MultiStrategyPortfolio(initial_capital=1_000_000)

for col in strategies.columns:
    portfolio.add_strategy(col, strategies[col])

print("风险平价权重:")
print(portfolio.rebalance('risk_parity'))

print("\n夏普加权权重:")
print(portfolio.rebalance('sharpe_weighted'))
```

## 多策略组合注意事项

| 风险类型   | 描述                       | 应对方法               |
| ---------- | -------------------------- | ---------------------- |
| 相关性突变 | 危机时刻策略相关性急剧上升 | 降低总杠杆，保持现金   |
| 策略容量   | 大资金无法有效执行小策略   | 优先选择容量大的策略   |
| 执行延迟   | 多策略同时下单可能延迟     | 使用算法执行           |
| 资金利用率 | 低波动策略占用资金但收益低 | 适当提高低波动策略杠杆 |

## 总结

多策略组合的核心是**分散化**：

- 不要把所有资金放在一个策略上
- 关注策略间的低相关性
- 定期评估和再平衡
