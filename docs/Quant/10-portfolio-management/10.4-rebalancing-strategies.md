# 10.4 仓位再平衡

## 什么是仓位再平衡？

仓位再平衡（Rebalancing）是调整投资组合中各资产权重，使其回归目标权重的过程。随着市场波动，原本均衡的权重会偏离目标，再平衡就是"恢复出厂设置"。

## 为什么要再平衡？

1. **控制风险**: 防止单一资产权重过高导致风险集中
2. **实现高卖低买**: 自动减持涨多的资产，增持跌多的资产
3. **保持目标配置**: 确保组合始终符合投资目标

## 再平衡触发条件

### 1. 时间驱动再平衡

按固定时间周期再平衡（每月、每季度）

```python
def time_based_rebalancing(target_weights, current_prices, rebalance_freq='monthly'):
    """
    时间驱动再平衡

    Parameters:
    - target_weights: 目标权重 dict
    - current_prices: 当前价格 dict
    - rebalance_freq: 再平衡频率 ('weekly', 'monthly', 'quarterly')
    """
    # 计算目标持仓
    total_value = sum(current_prices.values())
    target_positions = {
        asset: (weight * total_value) / price
        for asset, (weight, price) in zip(target_weights.keys(),
                                           zip(target_weights.values(),
                                               current_prices.values()))
    }

    return target_positions

# 示例
target_weights = {'BTC': 0.4, 'ETH': 0.3, 'USDT': 0.3}
current_prices = {'BTC': 50000, 'ETH': 3000, 'USDT': 1}

target_positions = time_based_rebalancing(target_weights, current_prices)
print("目标持仓:")
for asset, units in target_positions.items():
    value = units * current_prices[asset]
    print(f"  {asset}: {units:.6f} 单位 (价值 ${value:,.2f})")
```

### 2. 阈值驱动再平衡

当资产权重偏离目标超过阈值时触发

```python
def threshold_based_rebalancing(portfolio, target_weights, threshold=0.05):
    """
    阈值驱动再平衡

    当某个资产权重偏离目标超过阈值时触发
    """
    total_value = portfolio['total_value']
    current_prices = portfolio['current_prices']
    current_positions = portfolio['current_positions']

    # 计算当前权重
    current_weights = {
        asset: (units * current_prices[asset]) / total_value
        for asset, units in current_positions.items()
    }

    # 找出需要再平衡的资产
    rebalance_needed = []
    for asset in target_weights:
        current = current_weights.get(asset, 0)
        target = target_weights[asset]
        deviation = abs(current - target)

        if deviation > threshold:
            rebalance_needed.append({
                'asset': asset,
                'current_weight': current,
                'target_weight': target,
                'deviation': deviation
            })

    # 计算新的目标持仓
    new_positions = {}
    for asset in target_weights:
        target_value = total_value * target_weights[asset]
        new_positions[asset] = target_value / current_prices[asset]

    return {
        'rebalance_needed': len(rebalance_needed) > 0,
        'deviations': rebalance_needed,
        'new_positions': new_positions,
        'trades': calculate_trades(current_positions, new_positions)
    }

def calculate_trades(current, new):
    """计算机工交易"""
    trades = {}
    all_assets = set(current.keys()) | set(new.keys())

    for asset in all_assets:
        current_units = current.get(asset, 0)
        new_units = new.get(asset, 0)
        trades[asset] = new_units - current_units

    return trades

# 示例
portfolio = {
    'total_value': 100000,
    'current_prices': {'A': 100, 'B': 50, 'C': 25},
    'current_positions': {'A': 300, 'B': 400, 'C': 800}  # A:30000, B:20000, C:20000
}

target_weights = {'A': 0.33, 'B': 0.33, 'C': 0.34}
threshold = 0.03

result = threshold_based_rebalancing(portfolio, target_weights, threshold)

print(f"需要再平衡: {result['rebalance_needed']}")
print("\n偏离情况:")
for d in result['deviations']:
    print(f"  {d['asset']}: 当前 {d['current_weight']:.2%}, 目标 {d['target_weight']:.2%}, 偏离 {d['deviation']:.2%}")

print("\n交易指令:")
for asset, units in result['trades'].items():
    if units > 0:
        print(f"  买入 {asset}: +{units:.4f}")
    elif units < 0:
        print(f"  卖出 {asset}: {units:.4f}")
```

### 3. 波动率驱动再平衡

根据市场波动率动态调整再平衡周期

```python
class VolatilityTriggeredRebalancing:
    def __init__(self, target_weights, base_threshold=0.05, vol_threshold=0.02):
        self.target_weights = target_weights
        self.base_threshold = base_threshold
        self.vol_threshold = vol_threshold

    def calculate_current_weights(self, portfolio_value, positions, prices):
        """计算当前权重"""
        weights = {}
        for asset in positions:
            asset_value = positions[asset] * prices[asset]
            weights[asset] = asset_value / portfolio_value
        return weights

    def should_rebalance(self, current_weights, recent_volatility=None):
        """判断是否需要再平衡"""
        deviations = []

        for asset, target_weight in self.target_weights.items():
            current_weight = current_weights.get(asset, 0)
            deviation = abs(current_weight - target_weight)
            deviations.append(deviation)

        max_deviation = max(deviations) if deviations else 0

        # 基础阈值
        threshold = self.base_threshold

        # 如果波动率高，降低阈值（更频繁再平衡）
        if recent_volatility and recent_volatility > self.vol_threshold:
            threshold = threshold * 0.8  # 降低20%

        return max_deviation > threshold, max_deviation, threshold

# 使用示例
rebalancer = VolatilityTriggeredRebalancing(
    target_weights={'A': 0.4, 'B': 0.3, 'C': 0.3},
    base_threshold=0.05,
    vol_threshold=0.015
)

current_weights = {'A': 0.35, 'B': 0.32, 'C': 0.33}
should, deviation, threshold = rebalancer.should_rebalance(
    current_weights,
    recent_volatility=0.018
)

print(f"最大偏离: {deviation:.2%}, 阈值: {threshold:.2%}")
print(f"需要再平衡: {should}")
```

## 再平衡的成本

### 交易成本考虑

```python
def rebalance_with_costs(current_positions, new_positions, prices,
                         commission_rate=0.001, commission_min=5):
    """
    计算考虑交易成本后的再平衡
    """
    trades = {}
    total_cost = 0

    for asset in set(current_positions.keys()) | set(new_positions.keys()):
        current = current_positions.get(asset, 0)
        new = new_positions.get(asset, 0)
        trade = new - current

        if abs(trade) > 0:
            trade_value = abs(trade * prices[asset])
            commission = max(trade_value * commission_rate, commission_min)

            trades[asset] = {
                'current_units': current,
                'new_units': new,
                'trade_units': trade,
                'trade_value': trade_value,
                'commission': commission
            }
            total_cost += commission

    return {
        'trades': trades,
        'total_commission': total_cost,
        'net_value_change': -total_cost  # 交易成本是净损失
    }

# 示例
result = rebalance_with_costs(
    current_positions={'A': 300, 'B': 400, 'C': 800},
    new_positions={'A': 330, 'B': 330, 'C': 680},
    prices={'A': 100, 'B': 50, 'C': 25}
)

print("交易明细:")
for asset, trade in result['trades'].items():
    print(f"\n  {asset}:")
    print(f"    交易量: {trade['trade_units']:+.0f} 单位")
    print(f"    交易额: ${trade['trade_value']:,.2f}")
    print(f"    手续费: ${trade['commission']:.2f}")

print(f"\n总手续费: ${result['total_commission']:.2f}")
```

### 税负考虑

```python
def rebalance_with_taxes(current_positions, new_positions, prices,
                         cost_basis, tax_rate=0.20):
    """
    计算考虑资本利得税后的再平衡
    """
    trades = {}
    total_realized_gain = 0
    total_tax = 0

    for asset in set(current_positions.keys()) & set(new_positions.keys()):
        current = current_positions[asset]
        new = new_positions[asset]

        # 只考虑卖出（减少持仓）
        if new < current:
            shares_sold = current - new
            proceeds = shares_sold * prices[asset]
            cost = shares_sold * cost_basis[asset]
            gain = proceeds - cost
            tax = gain * tax_rate if gain > 0 else 0

            trades[asset] = {
                'shares_sold': shares_sold,
                'proceeds': proceeds,
                'cost_basis_realized': cost,
                'realized_gain': gain,
                'tax': tax
            }

            total_realized_gain += gain
            total_tax += tax

    return {
        'trades': trades,
        'total_realized_gain': total_realized_gain,
        'estimated_tax': total_tax,
        'after_tax_proceeds': total_realized_gain - total_tax
    }

# 示例
tax_result = rebalance_with_taxes(
    current_positions={'A': 300, 'B': 400, 'C': 800},
    new_positions={'A': 250, 'B': 350, 'C': 900},
    prices={'A': 100, 'B': 50, 'C': 25},
    cost_basis={'A': 80, 'B': 45, 'C': 20},  # 原始买入价格
    tax_rate=0.20
)

print("资本利得税计算:")
for asset, data in tax_result['trades'].items():
    print(f"\n  {asset}:")
    print(f"    卖出股数: {data['shares_sold']}")
    print(f"    实现收益: ${data['realized_gain']:,.2f}")
    print(f"    估计税额: ${data['tax']:,.2f}")

print(f"\n总实现收益: ${tax_result['total_realized_gain']:,.2f}")
print(f"总估计税额: ${tax_result['estimated_tax']:,.2f}")
print(f"税后收益: ${tax_result['after_tax_proceeds']:,.2f}")
```

## 最优再平衡策略

### 阈值优化的科学方法

```python
def optimize_rebalance_threshold(returns_df, target_weights,
                                 transaction_cost_rate=0.001,
                                 lookback=252):
    """
    通过回测寻找最优再平衡阈值
    """
    results = []

    for threshold in np.arange(0.01, 0.20, 0.01):
        portfolio_values = [1.0]  # 初始资金为1
        current_weights = target_weights.copy()
        n_rebalances = 0

        for i in range(lookback, len(returns_df)):
            # 更新当前权重（根据收益自然变化）
            for asset in current_weights:
                idx = returns_df.columns.get_loc(asset)
                ret = returns_df.iloc[i-1, idx]
                current_weights[asset] *= (1 + ret)

            # 归一化
            total = sum(current_weights.values())
            current_weights = {k: v/total for k, v in current_weights.items()}

            # 检查是否需要再平衡
            max_deviation = max(
                abs(current_weights.get(k, 0) - v)
                for k, v in target_weights.items()
            )

            if max_deviation > threshold:
                # 再平衡
                n_rebalances += 1
                portfolio_value = portfolio_values[-1]

                # 扣除交易成本
                trade_value = sum(
                    abs(current_weights.get(k, 0) - v) * portfolio_value
                    for k, v in target_weights.items()
                )
                cost = trade_value * transaction_cost_rate
                portfolio_values.append(portfolio_value - cost)

                # 重置为目标权重
                current_weights = target_weights.copy()
            else:
                # 继续持有
                portfolio_value = portfolio_values[-1] * (
                    sum(current_weights[k] * returns_df.iloc[i-1, returns_df.columns.get_loc(k)]
                        for k in current_weights) + 1
                )
                portfolio_values.append(portfolio_value)

        final_value = portfolio_values[-1]
        annual_return = (final_value - 1) / (len(portfolio_values) / 252)

        results.append({
            'threshold': threshold,
            'final_value': final_value,
            'annual_return': annual_return,
            'n_rebalances': n_rebalances,
            'total_cost': (1 - final_value) / (1 - len(returns_df)/252 * transaction_cost_rate)
        })

    return pd.DataFrame(results)
```

## 完整再平衡系统

```python
class PortfolioRebalancer:
    def __init__(self, initial_capital, target_weights,
                 rebalance_method='threshold',
                 threshold=0.05,
                 frequency='monthly',
                 transaction_cost_rate=0.001,
                 tax_rate=0.20):
        self.capital = initial_capital
        self.target_weights = target_weights
        self.rebalance_method = rebalance_method
        self.threshold = threshold
        self.frequency = frequency
        self.transaction_cost_rate = transaction_cost_rate
        self.tax_rate = tax_rate

        self.current_positions = {k: 0 for k in target_weights}
        self.cost_basis = {k: 0 for k in target_weights}
        self.portfolio_history = []

    def update_prices(self, prices):
        """更新当前价格"""
        self.current_prices = prices
        self.current_value = sum(
            self.current_positions[k] * prices[k]
            for k in self.current_positions
        )

    def calculate_weights(self):
        """计算当前权重"""
        weights = {}
        for asset in self.current_positions:
            value = self.current_positions[asset] * self.current_prices[asset]
            weights[asset] = value / self.current_value
        return weights

    def should_rebalance(self):
        """判断是否需要再平衡"""
        current_weights = self.calculate_weights()

        if self.rebalance_method == 'threshold':
            max_deviation = max(
                abs(current_weights.get(k, 0) - v)
                for k, v in self.target_weights.items()
            )
            return max_deviation > self.threshold

        elif self.rebalance_method == 'time':
            # 简化：每月再平衡
            return True  # 需要结合日期逻辑

        return False

    def execute_rebalance(self):
        """执行再平衡"""
        new_positions = {}

        for asset in self.target_weights:
            target_value = self.current_value * self.target_weights[asset]
            new_positions[asset] = target_value / self.current_prices[asset]

        # 计算交易
        trades = {
            k: new_positions[k] - self.current_positions[k]
            for k in self.target_weights
        }

        # 更新持仓
        for asset in self.current_positions:
            old_units = self.current_positions[asset]
            new_units = new_positions[asset]

            if new_units > old_units:
                # 买入
                self.current_positions[asset] = new_units
            elif new_units < old_units:
                # 卖出，更新成本基础
                shares_sold = old_units - new_units
                realized_gain = shares_sold * (
                    self.current_prices[asset] - self.cost_basis[asset]
                )
                # 更新成本基础（简单按比例）
                self.current_positions[asset] = new_units

        return trades

    def update_after_trade(self, trades):
        """更新交易后的状态"""
        trade_value = sum(
            abs(trades[k]) * self.current_prices[k]
            for k in trades
        )
        cost = trade_value * self.transaction_cost_rate

        self.capital -= cost
        self.portfolio_history.append({
            'value': self.current_value - cost,
            'positions': self.current_positions.copy()
        })

    def run(self, price_history):
        """运行再平衡模拟"""
        for i, (date, prices) in enumerate(price_history.items()):
            self.update_prices(prices)

            if self.should_rebalance():
                trades = self.execute_rebalance()
                self.update_after_trade(trades)
            else:
                # 更新组合价值
                returns = {
                    k: (prices[k] - price_history.iloc[i-1][k]) / price_history.iloc[i-1][k]
                    for k in self.current_positions
                }
                self.current_value *= (1 + sum(
                    self.target_weights[k] * returns[k]
                    for k in returns
                ))
                self.portfolio_history.append({
                    'value': self.current_value,
                    'positions': self.current_positions.copy()
                })

        return self.portfolio_history

# 使用示例
rebalancer = PortfolioRebalancer(
    initial_capital=100_000,
    target_weights={'A': 0.4, 'B': 0.3, 'C': 0.3},
    rebalance_method='threshold',
    threshold=0.08
)

print("再平衡系统已初始化")
print(f"目标权重: {rebalancer.target_weights}")
print(f"再平衡阈值: {rebalancer.threshold}")
```

## 再平衡策略对比

| 策略       | 优点         | 缺点           | 适用场景   |
| ---------- | ------------ | -------------- | ---------- |
| 固定时间   | 简单、可预测 | 可能错过机会   | 长期投资   |
| 阈值触发   | 成本可控     | 可能长期不调仓 | 主动管理   |
| 波动率触发 | 适应市场     | 逻辑复杂       | 高波动市场 |
| 混合策略   | 综合优点     | 实现复杂       | 专业投资者 |

## 总结

1. **再平衡是纪律**: 不要贪婪或恐惧，坚持目标权重
2. **成本意识**: 交易成本和税负会侵蚀收益
3. **选择合适方法**: 根据自己的投资目标和风格选择
4. **记录跟踪**: 保持交易记录，定期复盘
