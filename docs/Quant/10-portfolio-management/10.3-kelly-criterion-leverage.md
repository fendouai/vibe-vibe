# 10.3 凯利公式与杠杆

## 凯利公式的起源

凯利公式（Kelly Criterion）由约翰·凯利（John Kelly）于1956年在贝尔实验室提出，源于信息论。最初用于计算在噪声信道中传输信息的最优赌注比例，后来被巴菲特、比尔·阿克曼等投资大师广泛应用于投资组合管理。

## 经典凯利公式

在最简单的两项赌局中：

$$f^* = \frac{p \cdot b - q}{b}$$

其中：

- $f^*$: 投入资金的比例
- $p$: 获胜概率
- $q = 1-p$: 失败概率
- $b$: 赔率（赢了能获得本金的几倍，不含本金）

简化形式（当 $b=1$ 时）：

$$f^* = 2p - 1$$

这意味着只有当胜率超过50%时，才值得下注。

## 凯利公式在投资中的应用

### 离散收益版本

对于具有离散收益的投资：

$$f^* = \frac{E(R)}{Var(R)}$$

其中：

- $E(R)$: 策略预期收益率
- $Var(R)$: 策略收益率方差

### 连续收益版本（更常用）

$$f^* = \frac{\mu - r}{\sigma^2}$$

其中：

- $\mu$: 策略预期超额收益率
- $\sigma^2$: 超额收益率方差
- $r$: 无风险利率

```python
import numpy as np
import pandas as pd
import matplotlib.pyplot as plt

def kelly_criterion(win_rate, avg_win, avg_loss):
    """
    计算凯利公式

    Parameters:
    - win_rate: 胜率 (0-1)
    - avg_win: 平均盈利 (正数)
    - avg_loss: 平均亏损 (正数)

    Returns:
    - kelly_fraction: 建议的下注/投资比例
    """
    # 赔率 = 平均盈利 / 平均亏损
    b = avg_win / avg_loss

    # 凯利公式
    q = 1 - win_rate
    kelly = (b * win_rate - q) / b

    return max(kelly, 0)  # 凯利不能为负

# 示例：胜率60%，平均盈利$100，平均亏损$80
kelly = kelly_criterion(0.60, 100, 80)
print(f"凯利公式建议仓位: {kelly:.2%}")

# 如果使用2倍杠杆
print(f"使用2倍杠杆后实际仓位: {min(kelly * 2, 1.0):.2%}")
```

## 凯利公式的几何意义

凯利公式实际上是在最大化资金增长率的对数：

$$g(f) = p \ln(1 + fb) + q \ln(1 - f)$$

```python
def kelly_growth_rate(f, p, b):
    """计算给定仓位f的资金增长率"""
    return p * np.log(1 + f * b) + (1 - p) * np.log(1 - f)

def plot_kelly_curve(win_rate, avg_win, avg_loss):
    """绘制凯利曲线"""
    b = avg_win / avg_loss

    fractions = np.linspace(0.01, 0.99, 100)
    growth_rates = [kelly_growth_rate(f, win_rate, b) for f in fractions]

    optimal_kelly = kelly_criterion(win_rate, avg_win, avg_loss)
    optimal_growth = kelly_growth_rate(optimal_kelly, win_rate, b)

    plt.figure(figsize=(10, 6))
    plt.plot(fractions, growth_rates, 'b-', linewidth=2)
    plt.axvline(x=optimal_kelly, color='r', linestyle='--',
                label=f'Optimal Kelly = {optimal_kelly:.2%}')
    plt.axhline(y=optimal_growth, color='g', linestyle=':', alpha=0.5)
    plt.xlabel('Position Size (Fraction of Capital)')
    plt.ylabel('Expected Log Growth Rate')
    plt.title(f'Kelly Criterion (Win Rate={win_rate:.0%}, Odds={b:.2f})')
    plt.legend()
    plt.grid(True, alpha=0.3)
    plt.tight_layout()
    plt.savefig('kelly_curve.png', dpi=150)
    plt.show()

# 绘制不同胜率的凯利曲线
for wr in [0.4, 0.5, 0.6, 0.7]:
    kelly = kelly_criterion(wr, 100, 80)
    print(f"胜率 {wr:.0%}: 凯利仓位 = {kelly:.2%}")
```

## 半凯利与分数凯利

研究表明，实际应用中**半凯利**（Kelly的一半）更为稳健：

- **全凯利**: 理论最优，但波动极大
- **半凯利**: 降低约25%的波动，只损失约6%的收益率
- **四分之一凯利**: 进一步降低波动，适合保守投资者

```python
def compare_kelly_fractions(strategy_returns, kelly_fraction=0.5):
    """比较不同凯利系数的表现"""
    results = {}

    for fraction in [0.25, 0.5, 0.75, 1.0]:
        # 计算凯利仓位
        mu = strategy_returns.mean()
        sigma2 = strategy_returns.var()
        full_kelly = mu / sigma2 if sigma2 > 0 else 0

        # 应用凯利系数
        actual_fraction = full_kelly * fraction

        # 模拟收益序列
        simulated_returns = strategy_returns * actual_fraction

        # 计算统计指标
        cumulative = (1 + simulated_returns).cumprod()
        total_return = cumulative.iloc[-1] - 1
        annual_return = simulated_returns.mean() * 252
        annual_vol = simulated_returns.std() * np.sqrt(252)
        sharpe = annual_return / annual_vol if annual_vol > 0 else 0

        results[f'{int(fraction*100)}% Kelly'] = {
            'kelly_fraction': actual_fraction,
            'total_return': total_return,
            'annual_return': annual_return,
            'annual_vol': annual_vol,
            'sharpe': sharpe,
            'max_drawdown': ((cumulative - cumulative.cummax()) / cumulative.cummax()).min()
        }

    return pd.DataFrame(results).T

# 使用示例
np.random.seed(42)
strategy_returns = pd.Series(np.random.randn(252) * 0.02 + 0.0005)
comparison = compare_kelly_fractions(strategy_returns)
print(comparison)
```

## 杠杆的使用

### 杠杆的基本概念

$$实际收益 = \frac{策略收益}{自己资金的比例}$$

举例：

- 自己出50万，借50万（2倍杠杆）
- 策略盈利10%，实际盈利5万 = 10% / 0.5 = 20%的收益
- 策略亏损10%，实际亏损5万 = -20%的收益

```python
def calculate_leveraged_returns(strategy_return, leverage, borrowing_rate=0.03/252):
    """
    计算杠杆化收益

    Parameters:
    - strategy_return: 策略自身收益率
    - leverage: 杠杆倍数 (1 = 无杠杆)
    - borrowing_rate: 借贷日利率
    """
    # 资金成本
    equity_fraction = 1 / leverage
    borrowing_cost = (1 - equity_fraction) * borrowing_rate

    # 杠杆收益 = 策略收益 * 杠杆 - 借贷成本
    leveraged_return = strategy_return * leverage - borrowing_cost

    return leveraged_return

# 示例
leverage_levels = [1.0, 1.5, 2.0, 2.5, 3.0]
daily_return = 0.001  # 策略日收益0.1%

print("不同杠杆倍数下的年化收益:")
print(f"策略自身年化收益: {(1 + daily_return)**252 - 1:.2%}")
print()

for lev in leverage_levels:
    leveraged = calculate_leveraged_returns(daily_return, lev)
    leveraged_annual = (1 + leveraged)**252 - 1
    print(f"  {lev}x 杠杆: 年化收益 = {leveraged_annual:.2%}")
```

## 杠杆与风险

###杠杆放大收益也放大风险

```python
def plot_leverage_impact(initial_capital=100000, leverage=2.0, daily_return=0.001,
                         daily_vol=0.02, n_days=252):
    """模拟不同杠杆下的收益分布"""
    np.random.seed(42)

    # 生成模拟收益
    returns = np.random.normal(daily_return, daily_vol, n_days)

    # 计算不同杠杆下的收益
    results = {}
    for lev in [1.0, 1.5, 2.0, 2.5, 3.0]:
        leveraged_returns = calculate_leveraged_returns(returns, lev)
        equity_curve = initial_capital * np.cumprod(1 + leveraged_returns)

        results[f'{lev}x'] = {
            'equity_curve': equity_curve,
            'final_value': equity_curve[-1],
            'max_drawdown': ((equity_curve - np.maximum.accumulate(equity_curve)) /
                            np.maximum.accumulate(equity_curve)).min()
        }

    # 绘图
    plt.figure(figsize=(12, 8))
    for name, data in results.items():
        plt.plot(data['equity_curve'], label=name)

    plt.xlabel('Days')
    plt.ylabel('Portfolio Value ($)')
    plt.title('Impact of Leverage on Portfolio Growth')
    plt.legend()
    plt.grid(True, alpha=0.3)
    plt.tight_layout()
    plt.savefig('leverage_impact.png', dpi=150)
    plt.show()

    # 打印统计
    print("\n杠杆影响统计:")
    print(f"{'杠杆':<10} {'最终价值':>15} {'最大回撤':>12} {'收益率':>12}")
    print("-" * 50)
    for name, data in results.items():
        ret = (data['final_value'] / initial_capital - 1) * 100
        dd = data['max_drawdown'] * 100
        print(f"{name:<10} ${data['final_value']:>14,.0f} {dd:>11.1f}% {ret:>11.1f}%")

plot_leverage_impact()
```

## 凯利与止损的结合

实际交易中，**凯利公式 + 止损** 是经典组合：

```python
def kelly_with_stop_loss(win_rate, avg_win, avg_loss, max_loss_pct=0.5):
    """
    带止损的凯利公式

    止损限制了单次最大亏损，从而影响凯利计算
    """
    # 有效赔率（考虑止损后）
    effective_b = avg_win / (avg_loss * max_loss_pct)

    # 有效胜率（考虑止损触发后）
    # 假设止损不改变胜率，仅改变盈亏比
    effective_kelly = kelly_criterion(win_rate, avg_win, avg_loss * max_loss_pct)

    return effective_kelly

# 示例
print("标准凯利:", kelly_criterion(0.60, 100, 80))
print("带50%止损的凯利:", kelly_with_stop_loss(0.60, 100, 80, 0.5))
```

## 杠杆管理原则

| 原则         | 说明                               |
| ------------ | ---------------------------------- |
| 不要满仓全押 | 即使凯利公式建议仓位，也要留有余地 |
| 分散化       | 不同策略使用独立仓位计算           |
| 动态调整     | 根据市场波动率调整杠杆             |
| 设置硬止损   | 防止单一交易毁灭性损失             |
| 半凯利保守   | 推荐使用50%凯利系数                |

## Python实现：完整凯利仓位计算器

```python
class KellyPositionSizer:
    def __init__(self, initial_capital, kelly_fraction=0.5, max_leverage=3.0):
        self.initial_capital = initial_capital
        self.kelly_fraction = kelly_fraction
        self.max_leverage = max_leverage

    def calculate_position_from_trades(self, trades_df):
        """
        从交易记录计算凯利仓位

        trades_df 需要包含: profit, loss 列
        """
        win_rate = (trades_df['profit'] > 0).mean()
        avg_win = trades_df[trades_df['profit'] > 0]['profit'].mean()
        avg_loss = abs(trades_df[trades_df['loss'] < 0]['loss'].mean())

        kelly = kelly_criterion(win_rate, avg_win, avg_loss)
        adjusted_kelly = kelly * self.kelly_fraction

        return {
            'win_rate': win_rate,
            'avg_win': avg_win,
            'avg_loss': avg_loss,
            'kelly': kelly,
            'adjusted_kelly': adjusted_kelly,
            'position_size': self.initial_capital * adjusted_kelly
        }

    def calculate_position_from_returns(self, returns, risk_free_rate=0.02/252):
        """
        从收益率序列计算凯利仓位
        """
        excess_returns = returns - risk_free_rate

        mu = excess_returns.mean()
        sigma2 = excess_returns.var()

        if sigma2 == 0:
            return {'kelly': 0, 'position_size': 0}

        full_kelly = mu / sigma2
        adjusted_kelly = full_kelly * self.kelly_fraction

        # 应用杠杆限制
        effective_leverage = min(adjusted_kelly, self.max_leverage)
        actual_position = self.initial_capital * effective_leverage

        return {
            'mean_return': mu,
            'variance': sigma2,
            'full_kelly': full_kelly,
            'adjusted_kelly': adjusted_kelly,
            'effective_leverage': effective_leverage,
            'position_size': actual_position
        }

# 使用示例
np.random.seed(42)
simulated_returns = pd.Series(np.random.randn(252) * 0.015 + 0.0003)

sizer = KellyPositionSizer(initial_capital=100_000, kelly_fraction=0.5)
result = sizer.calculate_position_from_returns(simulated_returns)

print("凯利仓位计算结果:")
for k, v in result.items():
    if 'return' in k or 'variance' in k:
        print(f"  {k}: {v:.6f}")
    elif 'kelly' in k or 'leverage' in k:
        print(f"  {k}: {v:.4f}")
    else:
        print(f"  {k}: ¥{v:,.0f}")
```

## 总结

| 概念     | 公式                                          | 要点             |
| -------- | --------------------------------------------- | ---------------- |
| 经典凯利 | $f^* = \frac{pb - q}{b}$                      | 最大化对数增长率 |
| 收益率版 | $f^* = \frac{\mu}{\sigma^2}$                  | 使用超额收益     |
| 半凯利   | $f^*_{half} = \frac{f^*}{2}$                  | 更稳健的实操选择 |
| 杠杆     | $R_{leveraged} = R \times L - (L-1) \times r$ | 注意借贷成本     |

**核心原则**: 凯利公式告诉我们应该投多少，但实际应用中要打折使用（半凯利或更低）。
